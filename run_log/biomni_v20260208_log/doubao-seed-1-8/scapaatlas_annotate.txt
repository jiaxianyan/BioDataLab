# Load required libraries
library(GenomicRanges)
library(tidyverse)

# Read BED files
rbp_bed <- read_tsv("benchmark/dataset/scAPAatlas/rbp_binding_sites.bed", 
                    col_names = c("chr", "start", "end", "name", "score", "strand", 
                                  "signal", "pval", "qval", "peak_len"))
apa_bed <- read_tsv("benchmark/dataset/scAPAatlas/apa_regions.bed", 
                    col_names = c("chr", "start", "end", "name", "score", "strand"))

# Convert to GRanges
rbp_gr <- makeGRangesFromDataFrame(rbp_bed, keep.extra.columns = TRUE)
apa_gr <- makeGRangesFromDataFrame(apa_bed)

# Find overlaps
overlaps <- findOverlaps(rbp_gr, apa_gr)

# Get overlapping RBP sites
overlapping_rbp <- rbp_gr[queryHits(overlaps)]

# Convert back to data frame
result_df <- as.data.frame(overlapping_rbp) %>%
  # Reorder columns to match BED format
  select(seqnames, start, end, name, score, strand, signal, pval, qval, peak_len) %>%
  # Rename columns to match original BED
  rename(chr = seqnames)

# Save intermediate result
write_tsv(result_df, "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/scapaatlas_annotate/intermediate_overlaps.bed", 
          col_names = FALSE)

# Save final result
write_tsv(result_df, "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/scapaatlas_annotate.bed", 
          col_names = FALSE)

# Print result
cat("Found", nrow(result_df), "overlapping regions\n")
print(result_df)
