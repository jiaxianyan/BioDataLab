Given a single-cell dataset from `benchmark/dataset/DRMef/pbmc3k_cd4_fixed.h5ad`.

Read the dataset. It contains 'CD4 T cells' with a pre-annotated observation column named 'group', which has two labels: 'resistant' and 'sensitive'.

Perform a differential expression analysis (Wilcoxon rank-sum test) to find genes that are upregulated in the 'resistant' group compared to the 'sensitive' group.

Filter the results to identify significant genes meeting these criteria: Adjusted p-value (Benjamini-Hochberg) < 0.05; Log2 fold change > 1.0.

Save the names of the identified genes into a CSV file `/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/drmref_annotate.csv`.

Output example for reference:
gene 
gene1
gene2
...

All intermediate results should be save in `/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/drmref_annotate` directory.

<execute>
import scanpy as sc
import pandas as pd
import os

# Create necessary directories
tmp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/drmref_annotate"
output_dir = "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8"
os.makedirs(tmp_dir, exist_ok=True)
os.makedirs(output_dir, exist_ok=True)

# Read the dataset
adata = sc.read("benchmark/dataset/DRMef/pbmc3k_cd4_fixed.h5ad")
print(f"Dataset loaded successfully. Number of cells: {adata.n_obs}, Number of genes: {adata.n_vars}")
print(f"Group labels: {adata.obs['group'].unique()}")

# Perform differential expression analysis (Wilcoxon rank-sum test)
sc.tl.rank_genes_groups(adata, groupby='group', groups=['resistant'], reference='sensitive', method='wilcoxon', corr_method='benjamini-hochberg')

# Extract full results
full_results = sc.get.rank_genes_groups_df(adata, group='resistant')
print(f"Full DE results shape: {full_results.shape}")

# Save full results to tmp directory
full_results.to_csv(os.path.join(tmp_dir, "full_de_results.csv"), index=False)
print(f"Full results saved to {os.path.join(tmp_dir, 'full_de_results.csv')}")

# Filter significant genes
filtered_genes = full_results[(full_results['pvals_adj'] < 0.05) & (full_results['logfoldchanges'] > 1.0)]
print(f"Number of significant upregulated genes: {filtered_genes.shape[0]}")

# Prepare the output dataframe
output_df = filtered_genes[['names']].rename(columns={'names': 'gene'})

# Save filtered genes to the specified CSV
output_path = os.path.join(output_dir, "drmref_annotate.csv")
output_df.to_csv(output_path, index=False)
print(f"Filtered genes saved to {output_path}")

# Verify the file exists
if os.path.exists(output_path):
    print(f"Success: Output file {output_path} exists.")
else:
    print(f"Error: Output file {output_path} not found.")
