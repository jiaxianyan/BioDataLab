from bs4 import BeautifulSoup
import csv
import os

# Define paths
tmp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/amdb_extract"
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/amdb_extract.csv"

# Create output directory if it doesn't exist
os.makedirs(os.path.dirname(output_file), exist_ok=True)

# Initialize results list
results = []

# Process each sample
for sample_id in ["ERS1713185", "SRS4889584"]:
    xml_path = os.path.join(tmp_dir, f"{sample_id}_metadata.xml")
    
    # Parse XML
    with open(xml_path, 'r') as f:
        soup = BeautifulSoup(f, 'xml')
    
    # Extract host information
    host = "N/A"
    host_tags = soup.find_all("Attribute", {"attribute_name": "host"})
    if host_tags:
        host = host_tags[0].get_text(strip=True)
    else:
        # Try alternative tags
        organism_tags = soup.find_all("Organism")
        if organism_tags:
            host = organism_tags[0].get_text(strip=True)
    
    # For now, set diet to N/A, we'll look this up later
    results.append({
        "Sample Accesion": sample_id,
        "Host": host,
        "Diet": "N/A"
    })

# Save initial results to CSV
with open(output_file, 'w', newline='') as csvfile:
    fieldnames = ["Sample Accesion", "Host", "Diet"]
    writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
    writer.writeheader()
    writer.writerows(results)

print("Initial metadata extraction completed. Results saved to:", output_file)
