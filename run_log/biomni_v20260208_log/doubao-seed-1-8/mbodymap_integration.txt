import subprocess
import json
import os

# Configuration
GEO_SAMPLES = ["GSM3396082", "GSM3396083"]  # Using first two samples for testing
TEMP_DIR = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/mbodymap_integration"
OUTPUT_FILE = "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/mbodymap_integration.json"

# Create directories if they don't exist
os.makedirs(TEMP_DIR, exist_ok=True)
os.makedirs(os.path.dirname(OUTPUT_FILE), exist_ok=True)

# Function to get SRA runs from GEO sample
def get_sra_runs(geo_sample):
    try:
        # Use EDirect to link GEO to SRA
        cmd = f"esearch -db gds -query {geo_sample} | elink -target sra | efetch -format runinfo"
        result = subprocess.run(cmd, shell=True, capture_output=True, text=True, check=True)
        return result.stdout
    except subprocess.CalledProcessError as e:
        print(f"Error processing {geo_sample}: {e}")
        return None

# Process all samples
all_runs = []
for sample in GEO_SAMPLES:
    print(f"Processing {sample}...")
    runinfo = get_sra_runs(sample)
    if runinfo:
        # Save temp file
        temp_file = os.path.join(TEMP_DIR, f"{sample}_runinfo.csv")
        with open(temp_file, 'w') as f:
            f.write(runinfo)
        print(f"Saved run info to {temp_file}")
        
        # Parse CSV to get relevant runs
        lines = runinfo.strip().split('\n')
        if len(lines) > 1:
            headers = lines[0].split(',')
            for line in lines[1:]:
                fields = line.split(',')
                run_dict = dict(zip(headers, fields))
                # Filter for 16S rRNA and paired-end
                if ('16S' in run_dict.get('LibraryName', '') or '16S' in run_dict.get('Description', '')) and run_dict.get('LibraryLayout') == 'PAIRED':
