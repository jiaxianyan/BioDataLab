query = "(transcription factor[Title/Abstract] OR transcription factors[Title/Abstract]) AND \"Meta-Analysis\"[Publication Type] AND (\"2025/01/01\"[Date - Publication] : \"2025/01/05\"[Date - Publication])"

# Search PubMed
print(f"Searching PubMed with query: {query}")
handle = Entrez.esearch(db="pubmed", term=query, retmax=100)
record = Entrez.read(handle)
handle.close()

pmids = record["IdList"]
print(f"Found {len(pmids)} articles")

# Retrieve abstracts
articles = []
for pmid in pmids:
    print(f"Retrieving abstract for PMID: {pmid}")
    handle = Entrez.efetch(db="pubmed", id=pmid, rettype="abstract", retmode="text")
    abstract = handle.read()
    handle.close()
    
    articles.append({
        "PMID": pmid,
        "Abstract": abstract
    })

# Save to file
output_file = os.path.join(output_dir, "pubmed_results.json")
with open(output_file, "w") as f:
    json.dump(articles, f, indent=2)

print(f"Results saved to {output_file}")

import json
import os

# Load the retrieved articles
input_file = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/tf_marker_retrieval/pubmed_results.json"
with open(input_file, "r") as f:
    articles = json.load(f)

# Define keywords to identify functional/regulatory relationships between TFs and markers
marker_terms = ["biomarker", "genetic marker", "protein marker", "epigenetic marker", "marker"]
relationship_terms = ["regulates", "controls", "modulates", "interacts with", "associates with", "correlates with", "binds to", "targets", "is associated with", "is correlated with"]

# Analyze each abstract
results = []
for article in articles:
    pmid = article["PMID"]
    abstract = article["Abstract"].lower()
    
    # Check if abstract mentions any marker term and relationship term
    has_marker = any(term.lower() in abstract for term in marker_terms)
    has_relationship = any(term.lower() in abstract for term in relationship_terms)
    
    # Check if both are present and if they relate TFs to markers
    accepted = False
    if has_marker and has_relationship:
        # More specific check to ensure the relationship is between TFs and markers
        tf_terms = ["transcription factor", "transcription factors", "tf", "tfs"]
        has_tf = any(term.lower() in abstract for term in tf_terms)
        if has_tf:
            accepted = True
    
    results.append({
        "PMID": pmid,
        "Accepted": accepted
    })

# Save the final results
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/tf_marker_retrieval.json"
os.makedirs(os.path.dirname(output_file), exist_ok=True)
with open(output_file, "w") as f:
    json.dump(results, f, indent=2)

print(f"Analysis completed. Results saved to {output_file}")
