# Function to process each GEO dataset
process_geo_dataset <- function(gse_id) {
  cat("Processing dataset:", gse_id, "\n")
  
  # Download GEO data
  gse <- getGEO(gse_id, GSEMatrix = FALSE, AnnotGPL = TRUE)
  
  # Get sample information
  samples <- gse@gsms
  
  # Process each sample
  sample_data <- lapply(samples, function(sample) {
    # Get sample accession
    sample_acc <- sample@header$geo_accession
    
    # Extract metadata
    metadata <- sample@header
    
    # Extract age
    age <- metadata[grepl("age", names(metadata), ignore.case = TRUE)]
    age_val <- ifelse(length(age) > 0, age[[1]], "N/A")
    
    # Extract gender
    gender <- metadata[grepl("gender|sex", names(metadata), ignore.case = TRUE)]
    gender_val <- ifelse(length(gender) > 0, gender[[1]], "N/A")
    
    # Extract clinical phenotypes
    phenotype <- metadata[grepl("phenotype|disease|condition|status", names(metadata), ignore.case = TRUE)]
    phenotype_val <- ifelse(length(phenotype) > 0, paste(phenotype, collapse = "; "), "N/A")
    
    # Extract drug information
    drug <- metadata[grepl("drug|treatment|therapy|medication", names(metadata), ignore.case = TRUE)]
    drug_val <- ifelse(length(drug) > 0, paste(drug, collapse = "; "), "N/A")
    
    # Create result object
    list(
      sample = sample_acc,
      age = age_val,
      gender = gender_val,
      "clinical phenotypes" = phenotype_val,
      drug = drug_val
    )
  })
  
  # Convert to data frame
  df <- bind_rows(sample_data)
  
  return(df)
}

# Function to process each GEO dataset
process_geo_dataset <- function(gse_id) {
  cat("Processing dataset:", gse_id, "\n")
  
  # Download GEO data
  gse <- getGEO(gse_id, GSEMatrix = FALSE, AnnotGPL = TRUE)
  
  # Get sample information
  samples <- gse@gsms
  
  # Process each sample
  sample_data <- lapply(samples, function(sample) {
    # Get sample accession
    sample_acc <- sample@header$geo_accession
    
    # Extract metadata fields
    metadata <- sample@header
    
    # Extract age - check characteristics_ch1 first
    age <- "N/A"
    if (!is.null(metadata$characteristics_ch1)) {
      age_fields <- metadata$characteristics_ch1[grepl("age", metadata$characteristics_ch1, ignore.case = TRUE)]
      if (length(age_fields) > 0) {
        age <- sub("^age: ", "", age_fields[1], ignore.case = TRUE)
      }
    }
    
    # Extract gender - check characteristics_ch1 first
    gender <- "N/A"
    if (!is.null(metadata$characteristics_ch1)) {
      gender_fields <- metadata$characteristics_ch1[grepl("gender|sex", metadata$characteristics_ch1, ignore.case = TRUE)]
      if (length(gender_fields) > 0) {
        gender <- sub("^(gender|sex): ", "", gender_fields[1], ignore.case = TRUE)
      }
    }
    
    # Extract clinical phenotypes - check characteristics_ch1 and title
    phenotype <- "N/A"
    phenotype_fields <- c()
    
    if (!is.null(metadata$characteristics_ch1)) {
      phenotype_fields <- c(phenotype_fields, 
                           metadata$characteristics_ch1[grepl("phenotype|disease|condition|status|cell type|tissue", metadata$characteristics_ch1, ignore.case = TRUE)])
    }
    
    if (!is.null(metadata$title)) {
      phenotype_fields <- c(phenotype_fields, metadata$title)
    }
    
    if (length(phenotype_fields) > 0) {
      phenotype <- paste(unique(phenotype_fields), collapse = "; ")
    }
    
    # Extract drug information - check characteristics_ch1 and treatment_protocol_ch1
    drug <- "N/A"
    drug_fields <- c()
    
    if (!is.null(metadata$characteristics_ch1)) {
      drug_fields <- c(drug_fields, 
                      metadata$characteristics_ch1[grepl("drug|treatment|therapy|medication|ifn|interferon", metadata$characteristics_ch1, ignore.case = TRUE)])
    }
    
    if (!is.null(metadata$treatment_protocol_ch1)) {
      drug_fields <- c(drug_fields, metadata$treatment_protocol_ch1)
    }
    
    if (length(drug_fields) > 0) {
      drug <- paste(unique(drug_fields), collapse = "; ")
    }
    
    # Clean up fields
    age <- ifelse(age == "" | is.null(age), "N/A", age)
    gender <- ifelse(gender == "" | is.null(gender), "N/A", gender)
    phenotype <- ifelse(phenotype == "" | is.null(phenotype), "N/A", phenotype)
    drug <- ifelse(drug == "" | is.null(drug), "N/A", drug)
    
    # Create result object
    list(
      sample = sample_acc,
      age = age,
      gender = gender,
      "clinical phenotypes" = phenotype,
      drug = drug
    )
  })
  
  # Convert to data frame
  df <- bind_rows(sample_data)
  
  return(df)
}
