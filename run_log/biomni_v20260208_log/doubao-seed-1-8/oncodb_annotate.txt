# Load required libraries
library(survival)
library(tidyverse)

# Define file paths
input_file <- "./operation_env/database_lake/benchmark/dataset/OncoDB/rossi_dataset.csv"
output_file <- "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/oncodb_annotate.csv"
tmp_dir <- "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/oncodb_annotate"

# Read the dataset

# Save intermediate data
write_csv(rossi_data, file.path(tmp_dir, "raw_data.csv"))

# Create Cox PH model
# Use all columns except week and arrest as covariates
covariates <- setdiff(colnames(rossi_data), c("week", "arrest"))
formula <- as.formula(paste0("Surv(week, arrest) ~ ", paste(covariates, collapse = " + ")))
cox_model <- coxph(formula, data = rossi_data)

# Save model object
saveRDS(cox_model, file.path(tmp_dir, "cox_model.rds"))

# Extract summary results
summary_results <- summary(cox_model)

# Save full summary
capture.output(summary(cox_model), file = file.path(tmp_dir, "full_summary.txt"))

# Format results according to output example
formatted_results <- as_tibble(summary_results$coefficients, rownames = "covariate") %>%
  rename(
    coef = `coef`,
    `exp(coef)` = `exp(coef)`,
    `se(coef)` = `se(coef)`,
    z = `z`,
    p = `Pr(>|z|)`
  ) %>%
  mutate(
    `coef lower 95%` = coef - 1.96 * `se(coef)`,
    `coef upper 95%` = coef + 1.96 * `se(coef)`,
    `exp(coef) lower 95%` = exp(`coef lower 95%`),
    `exp(coef) upper 95%` = exp(`coef upper 95%`),
    `cmp to` = covariate,
    `-log2(p)` = -log2(p)
  ) %>%
  select(
    covariate, coef, `exp(coef)`, `se(coef)`,
    `coef lower 95%`, `coef upper 95%`,
    `exp(coef) lower 95%`, `exp(coef) upper 95%`,
    `cmp to`, z, p, `-log2(p)`
  ) %>%
  mutate(across(where(is.numeric), round, digits = 3))

# Save formatted results
write_csv(formatted_results, output_file)

# Save formatted intermediate results
write_csv(formatted_results, file.path(tmp_dir, "formatted_results.csv"))

# Print confirmation
