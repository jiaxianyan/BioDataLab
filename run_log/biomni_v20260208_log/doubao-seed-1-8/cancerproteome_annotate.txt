import pandas as pd
from scipy.stats import spearmanr

# Step 1: Read input files
protein_df = pd.read_csv('benchmark/dataset/CancerProteome/protein_abundance.csv')
transcript_df = pd.read_csv('benchmark/dataset/CancerProteome/transcript_levels.csv')

# Step 2: Merge on GeneID
merged_df = pd.merge(protein_df, transcript_df, on='GeneID', how='inner')

# Step 3: Save intermediate merged data
intermediate_path = '/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/cancerproteome_annotate/merged_data.csv'
merged_df.to_csv(intermediate_path, index=False)
print(f"Intermediate merged data saved to: {intermediate_path}")

# Step 4: Compute Spearman correlation
correlation, p_value = spearmanr(merged_df['Protein_Abundance'], merged_df['Transcript_Level'])
print(f"Spearman correlation coefficient: {correlation}")
print(f"P-value: {p_value}")

# Step 5: Save final correlation result
output_path = '/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/cancerproteome_annotate.txt'
with open(output_path, 'w') as f:
    f.write(f"{correlation:.4f}\n")
print(f"Final correlation saved to: {output_path}")
