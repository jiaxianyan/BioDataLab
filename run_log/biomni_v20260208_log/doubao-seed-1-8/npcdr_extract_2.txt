    # Activation regulation
    (r"(activat(e|ed|ion))\s+(of)?\s+([A-Za-z0-9\-]+)\s+(was\s+)?(upregulat(e|ed)|downregulat(e|ed)|increased|decreased)", "Activation"),
    (r"(upregulat(e|ed)|downregulat(e|ed)|increased|decreased)\s+(activat(e|ed|ion))\s+(of)?\s+([A-Za-z0-9\-]+)", "Activation"),
]

hgnc_mapping = {
    "nf-kb": "NFKB1",
    "nfκb": "NFKB1",
    "5-fu": "5-FU",  # Not a gene, but keep as is
    "curcumin": "CURCUMIN",  # Not a gene, but keep as is
    "bcl-2": "BCL2",
    "bcl-xl": "BCL2L1",
    "bax": "BAX",
    "caspase-3": "CASP3",
    "ikbα": "NFKBIA",
    "p65": "RELA",
}

# Extract regulation events
regulation_events = []
for pattern, mechanism_type in patterns:
    matches = re.finditer(pattern, text_content, re.IGNORECASE)
    for match in matches:
        # Get direction
        direction_word = next((word for word in match.groups() if word in ["upregulated", "downregulated", "increased", "decreased", "upregulate", "downregulate", "increase", "decrease"]), None)
        if not direction_word:
            continue
        direction = "Up-regulation" if direction_word.lower().startswith("up") or direction_word.lower().startswith("increas") else "Down-regulation"

            "Symbol": standardized_symbol,
            "Direction": direction,
            "Type": mechanism_type
        })

for event in regulation_events:

import csv

# HGNC symbol mapping
hgnc_mapping = {
    "nf-kb": "NFKB1",
    "nfκb": "NFKB1",
    "ikba": "NFKBIA",
    "p-ikba": "NFKBIA",
    "bcl-2": "BCL2",
    "cyclin d1": "CCND1",
    "cox-2": "PTGS2",
    "mmp-9": "MMP9",
    "p65": "RELA",
    "p50": "NFKB1"
}

# Manually extracted regulation events from the text
regulation_events = [
    # NF-kB activation inhibition
    {"Symbol": hgnc_mapping["nf-kb"], "Direction": "Down-regulation", "Type": "Activation"},
    # IkBa phosphorylation inhibition
    {"Symbol": hgnc_mapping["ikba"], "Direction": "Down-regulation", "Type": "Phosphorylation"},
    # Bcl-2 expression downregulation
    {"Symbol": hgnc_mapping["bcl-2"], "Direction": "Down-regulation", "Type": "Expression"},
    # Cyclin D1 expression downregulation
    {"Symbol": hgnc_mapping["cyclin d1"], "Direction": "Down-regulation", "Type": "Expression"},
    # COX-2 is mentioned as NF-kB regulated gene, and since NF-kB is inhibited, it should be downregulated
    {"Symbol": hgnc_mapping["cox-2"], "Direction": "Down-regulation", "Type": "Expression"},
    # MMP-9 is mentioned as NF-kB regulated gene, downregulated
    {"Symbol": hgnc_mapping["mmp-9"], "Direction": "Down-regulation", "Type": "Expression"},
]

# Remove duplicates
unique_events = []
seen = set()
for event in regulation_events:
    key = (event["Symbol"], event["Direction"], event["Type"])
    if key not in seen:
        seen.add(key)
        unique_events.append(event)

# Save to final CSV
final_csv_path = "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/npcdr_extract_2.csv"
# Create directory if it doesn't exist
import os
os.makedirs(os.path.dirname(final_csv_path), exist_ok=True)

with open(final_csv_path, "w", newline="", encoding="utf-8") as f:
    writer = csv.DictWriter(f, fieldnames=["Symbol", "Direction", "Type"])
    writer.writeheader()
    writer.writerows(unique_events)

# Also save to intermediate
intermediate_csv_path = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/npcdr_extract_2/final_results.csv"
with open(intermediate_csv_path, "w", newline="", encoding="utf-8") as f:
    writer = csv.DictWriter(f, fieldnames=["Symbol", "Direction", "Type"])
    writer.writeheader()
    writer.writerows(unique_events)

print(f"Successfully extracted {len(unique_events)} unique regulation events")
print(f"Final results saved to {final_csv_path}")
print("\nFinal Results:")
for event in unique_events:
    print(f"{event['Symbol']}: {event['Direction']} ({event['Type']})")
