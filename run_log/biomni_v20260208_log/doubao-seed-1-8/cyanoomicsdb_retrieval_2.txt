<execute>
from Bio import Entrez
import csv
import os

# Set email for Entrez (required by NCBI)
Entrez.email = "assistant@example.com"

# Create output directory if not exists
output_dir = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/cyanoomicsdb_retrieval_2"
os.makedirs(output_dir, exist_ok=True)

# Search PubMed for cyanobacteria papers published before 2026-01-01
search_query = "cyanobacteria AND ('2025/01/01'[Date - Publication] : '2025/12/31'[Date - Publication])"
handle = Entrez.esearch(db="pubmed", term=search_query, sort="pub date", retmax=5)
record = Entrez.read(handle)
handle.close()

# Get IDs of the papers
id_list = record["IdList"]
print(f"Found {len(id_list)} papers")

# Fetch details for each paper
handle = Entrez.efetch(db="pubmed", id=id_list, retmode="xml")
records = Entrez.read(handle)
handle.close()

# Prepare data for CSV
papers = []
for record in records["PubmedArticle"]:
    title = record["MedlineCitation"]["Article"]["ArticleTitle"]
    journal = record["MedlineCitation"]["Article"]["Journal"]["Title"]
    doi = None
    # Find DOI if available
    for id_info in record["PubmedData"]["ArticleIdList"]:
        if id_info.attributes["IdType"] == "doi":
            doi = id_info
            break
    papers.append({
        "Title": title,
        "Journal": journal,
        "DOI": doi if doi else ""
    })

# Save to CSV
output_file = os.path.join(output_dir, "cyanobacteria_papers.csv")
with open(output_file, "w", newline="", encoding="utf-8") as f:
    writer = csv.DictWriter(f, fieldnames=["Title", "Journal", "DOI"])
    writer.writeheader()
    writer.writerows(papers)

print(f"Results saved to {output_file}")

# Also save to the final required path
final_output = "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/cyanoomicsdb_retrieval_2.csv"
with open(final_output, "w", newline="", encoding="utf-8") as f:
    writer = csv.DictWriter(f, fieldnames=["Title", "Journal", "DOI"])
    writer.writeheader()
    writer.writerows(papers)

print(f"Final results saved to {final_output}")
</execute>
