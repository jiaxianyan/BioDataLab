import json
import requests
import os

# Create intermediate directory if not exists
tmp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/a3d_modb_retrieval"
os.makedirs(tmp_dir, exist_ok=True)

# UniProt API query
url = "https://rest.uniprot.org/uniprotkb/search"
params = {
    "query": "(organism_id:83333 AND reviewed:true AND protein_name:transmembrane)",
    "format": "json",
    "size": 500  # Increase if more results are expected
}

response = requests.get(url, params=params)
response.raise_for_status()  # Raise error if request fails

# Save raw results
raw_results_path = os.path.join(tmp_dir, "uniprot_results.json")
with open(raw_results_path, "w") as f:
    json.dump(response.json(), f, indent=2)

# Extract accession numbers
accessions = [protein["primaryAccession"] for protein in response.json()["results"]]

# Save intermediate accessions text file
accessions_txt_path = os.path.join(tmp_dir, "accessions.txt")
with open(accessions_txt_path, "w") as f:
    f.write("\n".join(accessions))

# Save final JSON output
output_path = "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/a3d_modb_retrieval.json"
os.makedirs(os.path.dirname(output_path), exist_ok=True)
with open(output_path, "w") as f:
    json.dump(accessions, f, indent=2)

# Print the result
print("Successfully retrieved and saved UniProt accessions:")
print(json.dumps(accessions, indent=2))
