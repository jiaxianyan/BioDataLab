from Bio import Entrez

Entrez.email = "your.email@example.com"

# PMIDs to retrieve

# Retrieve abstracts
abstracts = {}
for pmid in pmids:
    handle = Entrez.efetch(db="pubmed", id=pmid, rettype="abstract", retmode="text")
    abstract = handle.read()
