import os
from Bio import SeqIO

# Define paths
fasta_path = "benchmark/dataset/CompoDynamics/GCF_009428885.1_ASM942888v1_cds_from_genomic.fna"
temp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/compodynamics_annotate"
output_path = "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/compodynamics_annotate.txt"

# Ensure temp directory exists
os.makedirs(temp_dir, exist_ok=True)

# Initialize variables
total_gc = 0
total_length = 0

# Process each CDS sequence
for record in SeqIO.parse(fasta_path, "fasta"):
    # Check if sequence belongs to NZ_WHYS01000010.1
    if "NZ_WHYS01000010.1" in record.id:
        # Calculate GC content for this sequence
        seq = str(record.seq).upper()
        length = len(seq)
        gc_count = seq.count('G') + seq.count('C')
        gc_content = gc_count / length if length > 0 else 0
        
        # Accumulate weighted GC content
        total_gc += gc_content * length
        total_length += length
        
        # Save temp data
        temp_file = os.path.join(temp_dir, f"{record.id}_gc.txt")
        with open(temp_file, 'w') as f:
            f.write(f"Sequence ID: {record.id}\n")
            f.write(f"Length: {length}\n")
            f.write(f"GC count: {gc_count}\n")
            f.write(f"GC content: {gc_content:.4f}\n")

# Calculate weighted average GC content
if total_length > 0:
    weighted_gc = total_gc / total_length
else:
    weighted_gc = 0.0

# Save final result
with open(output_path, 'w') as f:
    f.write(f"{weighted_gc:.4f}\n")

print(f"Weighted average GC content: {weighted_gc:.4f}")
print(f"Result saved to: {output_path}")
print(f"Temp data saved to: {temp_dir}")
