# Use awk to filter reads containing either telomeric motif

BEGIN {OFS="\n"}
{
    if (NR%4 == 1) {header = $0}
    else if (NR%4 == 2) {seq = $0}
    else if (NR%4 == 3) {plus = $0}
    else if (NR%4 == 0) {
        qual = $0
        if (seq ~ /TTAGGGTTAGGGTTAGGGTTAGGG/ || seq ~ /CCCTAACCCTAACCCTAACCCTAA/) {
            print header, seq, plus, qual
        }
    }
}' | gzip -c > /root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/dda_refinement.fastq.gz
