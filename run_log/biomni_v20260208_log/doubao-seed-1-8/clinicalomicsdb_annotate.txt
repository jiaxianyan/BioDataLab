import pandas as pd
from scipy.stats import mannwhitneyu
from sklearn.metrics import roc_auc_score
import numpy as np

# Read the restructured data
sample_data = pd.read_csv('/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/clinicalomicsdb_annotate/restructured_data.csv')

# Separate groups
responder = sample_data[sample_data['group'] == 'responder']
non_responder = sample_data[sample_data['group'] == 'non-responder']

# Prepare results dataframe
results = pd.DataFrame(columns=['gene', 'p_value', 'auroc'])

# Get list of genes (all columns except sample and group)
genes = [col for col in sample_data.columns if col not in ['sample', 'group']]

# Process each gene
for gene in genes:
    # Get expression values
    resp_expr = responder[gene].values
    non_resp_expr = non_responder[gene].values
    
    # Wilcoxon rank-sum test (Mann-Whitney U)
    stat, p_value = mannwhitneyu(resp_expr, non_resp_expr, alternative='two-sided')
    
    # Prepare data for AUROC
    y_true = [1]*len(responder) + [0]*len(non_responder)
    y_score = np.concatenate([resp_expr, non_resp_expr])
    
    # Calculate AUROC
    try:
        auroc = roc_auc_score(y_true, y_score)
    except ValueError:
        # If all values are the same, AUROC is 0.5
        auroc = 0.5
    
    # Add to results
    results = results.append({
        'gene': gene,
        'p_value': p_value,
        'auroc': auroc
    }, ignore_index=True)

# Show first few results
print("First 5 results:")
print(results.head())
print("\nResults shape:", results.shape)

# Save intermediate results
results.to_csv('/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/clinicalomicsdb_annotate/intermediate_results.csv', index=False)
print("\nIntermediate results saved to /root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/clinicalomicsdb_annotate/intermediate_results.csv")

os.makedirs(os.path.dirname(final_output_path), exist_ok=True)
