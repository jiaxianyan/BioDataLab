import csv
from collections import defaultdict

# Define input and output paths
gff_file = "benchmark/dataset/CyanoOmicsDB/ncbi_dataset/data/GCF_000009725.1/demo_genomic.gff"
temp_gene_info = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/cyanoomicsdb_annotate_1/gene_info.tsv"

# Initialize dictionaries to store gene and CDS info
genes = {}
cds_dict = defaultdict(dict)

# First pass: read gene features
with open(gff_file, 'r') as f:
    for line in f:
        line = line.strip()
        if not line or line.startswith('#'):
            continue
        parts = line.split('\t')
        if len(parts) != 9:
            continue
        seqid, source, feature_type, start, end, score, strand, phase, attributes = parts
        if feature_type == 'gene':
            # Parse attributes
            attr_dict = {}
            for attr in attributes.split(';'):
                if '=' in attr:
                    key, value = attr.split('=', 1)
                    attr_dict[key] = value
            # Store gene info
            gene_id = attr_dict.get('ID', '')
            genes[gene_id] = {
                'locus_tag': attr_dict.get('locus_tag', '-'),
                'symbol': attr_dict.get('gene', '-'),
                'old_locus_tag': attr_dict.get('old_locus_tag', '-'),
                'location': f"{seqid}:{start}-{end}({strand})",
                'protein_id': '-',
                'product': '-'
            }

# Second pass: read CDS features and link to genes
with open(gff_file, 'r') as f:
    for line in f:
        line = line.strip()
        if not line or line.startswith('#'):
            continue
        parts = line.split('\t')
        if len(parts) != 9:
            continue
        seqid, source, feature_type, start, end, score, strand, phase, attributes = parts
        if feature_type == 'CDS':
            # Parse attributes
            attr_dict = {}
            for attr in attributes.split(';'):
                if '=' in attr:
                    key, value = attr.split('=', 1)
                    attr_dict[key] = value
            # Get parent gene ID
            parent_ids = attr_dict.get('Parent', '').split(',')
            for parent_id in parent_ids:
                if parent_id in genes:
                    genes[parent_id]['protein_id'] = attr_dict.get('protein_id', '-')
                    genes[parent_id]['product'] = attr_dict.get('product', '-')
                    break

# Write to TSV
with open(temp_gene_info, 'w', newline='') as tsvfile:
    writer = csv.writer(tsvfile, delimiter='\t')
    writer.writerow(["Locus_Tag", "Symbol", "Old_Locus_Tag", "Location", "Protein_ID", "Product"])
    for gene in genes.values():
        writer.writerow([
            gene['locus_tag'],
            gene['symbol'],
            gene['old_locus_tag'],
            gene['location'],
            gene['protein_id'],
            gene['product']
        ])

print(f"Gene information saved to {temp_gene_info}")
