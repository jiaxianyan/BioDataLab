import pandas as pd
import numpy as np
from scipy.stats import pearsonr

# Read the gene expression data
expr_data = pd.read_csv('benchmark/dataset/CancerSCEM/gene_expression.csv', index_col=0)

# Calculate Pearson correlation matrix
corr_matrix = expr_data.corr(method='pearson')

# Save correlation matrix to intermediate directory
corr_matrix.to_csv('/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/cancerscem_annotate_2/correlation_matrix.csv')

# Find gene pairs with correlation > 0.9, excluding self-correlations and duplicate pairs
high_corr_pairs = []
for i in range(len(corr_matrix.index)):
    for j in range(i+1, len(corr_matrix.columns)):
        corr = corr_matrix.iloc[i, j]
        if abs(corr) > 0.9:
            high_corr_pairs.append((corr_matrix.index[i], corr_matrix.columns[j], corr))

# Convert to DataFrame and sort by correlation magnitude
high_corr_df = pd.DataFrame(high_corr_pairs, columns=['gene_1', 'gene_2', 'correlation'])
high_corr_df = high_corr_df.sort_values(by='correlation', key=abs, ascending=False)

# Save high correlation pairs to intermediate directory
high_corr_df.to_csv('/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/cancerscem_annotate_2/high_correlation_pairs.csv', index=False)

# Prepare final output in required format (only gene1 and gene2 columns)
final_output = high_corr_df[['gene_1', 'gene_2']]
final_output.to_csv('/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/cancerscem_annotate_2.txt', index=False, header=True)

print(f"Found {len(high_corr_df)} highly correlated gene pairs (|correlation| > 0.9)")
print("Results saved to specified directories.")
