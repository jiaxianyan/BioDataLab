import json
from collections import OrderedDict

# Read the input traits
with open('benchmark/dataset/RAVAR/reported_trails.json', 'r') as f:
    original_traits = json.load(f)

# Remove duplicates while preserving order
unique_traits = list(OrderedDict.fromkeys(original_traits))

# Create standardized mapping
standardized_mapping = {}
for trait in unique_traits:
    # Remove publication-specific symbols like (30700)
    standardized = trait.replace(' (30700)', '')
    
    # Remove ambiguous prefixes like "Biochemistry "
    standardized = standardized.replace('Biochemistry ', '')
    
    # Unify similar terms: "White blood cell count (basophil)" to "Basophil count"
    if "White blood cell count (basophil)" in standardized:
        standardized = "Basophil count"
    
    standardized_mapping[trait] = standardized

# Save intermediate steps
intermediate_dir = '/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/ravar_refinement_1'
with open(f'{intermediate_dir}/unique_traits.json', 'w') as f:
    json.dump(unique_traits, f, indent=2)

with open(f'{intermediate_dir}/standardization_mapping.json', 'w') as f:
    json.dump(standardized_mapping, f, indent=2)

# Save final result
output_path = '/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/ravar_refinement_1.json'
with open(output_path, 'w') as f:
    json.dump(standardized_mapping, f, indent=2)

print("Standardization completed successfully!")
print(f"Final mapping saved to: {output_path}")
print("\nMapping details:")
for original, standardized in standardized_mapping.items():
    print(f"  '{original}' -> '{standardized}'")
