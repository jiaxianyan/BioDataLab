from PIL import Image
import numpy as np
import os
import csv

# Define the feature options
COLOR_OPTIONS = ['Black', 'Green', 'Yellow', 'Brown', 'Gray', 'Red Brown', 'White']
TEXTURE_OPTIONS = ['Spotted', 'Striped', 'Ring Spot', 'Netted Spot', 'Random Spot']
MORPHOLOGY_OPTIONS = ['Atrophy', 'Wilt', 'Rot', 'Burn', 'Perforation']

# Function to analyze image features
def analyze_image(image_path):
    # Open image
    img = Image.open(image_path)
    img_np = np.array(img)
    
    # Get image name
    image_name = os.path.basename(image_path)
    
    # --- Color analysis ---
    # Calculate average RGB values
    avg_rgb = np.mean(img_np, axis=(0,1))
    # Normalize to 0-1
    norm_rgb = avg_rgb / 255.0
    
    # Simple color classification based on RGB values
    if norm_rgb[0] < 0.3 and norm_rgb[1] < 0.3 and norm_rgb[2] < 0.3:
        color = 'Black'
    elif norm_rgb[1] > norm_rgb[0] and norm_rgb[1] > norm_rgb[2]:
        color = 'Green'
    elif norm_rgb[0] > norm_rgb[1] and norm_rgb[0] > norm_rgb[2]:
        color = 'Red Brown'
    elif norm_rgb[0] > 0.6 and norm_rgb[1] > 0.6 and norm_rgb[2] < 0.4:
        color = 'Yellow'
    elif norm_rgb[0] < 0.5 and norm_rgb[1] < 0.5 and norm_rgb[2] < 0.5:
        color = 'Gray'
    elif norm_rgb[0] > 0.4 and norm_rgb[1] > 0.3 and norm_rgb[2] < 0.3:
        color = 'Brown'
    else:
        color = 'White'
    
    # --- Texture analysis ---
    # Calculate edge detection using simple Sobel-like filter
    gray = np.mean(img_np, axis=2)
    dx = np.gradient(gray, axis=0)
    dy = np.gradient(gray, axis=1)
    edges = np.sqrt(dx**2 + dy**2)
    edge_density = np.sum(edges > 50) / edges.size
    
    # Simple texture classification
    if edge_density < 0.05:
        texture = 'Spotted'
    elif edge_density < 0.1:
        texture = 'Striped'
    elif edge_density < 0.15:
        texture = 'Ring Spot'
    elif edge_density < 0.2:
        texture = 'Netted Spot'
    else:
        texture = 'Random Spot'
    
    # --- Morphology analysis ---
    # Calculate aspect ratio and area of non-green regions
    non_green_mask = (img_np[:,:,1] < img_np[:,:,0]) | (img_np[:,:,1] < img_np[:,:,2])
    non_green_area = np.sum(non_green_mask)
    total_area = img_np.shape[0] * img_np.shape[1]
    damage_ratio = non_green_area / total_area
    
    # Simple morphology classification
    if damage_ratio < 0.1:
        morphology = 'Atrophy'
    elif damage_ratio < 0.2:
        morphology = 'Wilt'
    elif damage_ratio < 0.3:
        morphology = 'Rot'
    elif damage_ratio < 0.4:
        morphology = 'Burn'
    else:
        morphology = 'Perforation'
    
    return image_name, color, texture, morphology

# Function to analyze image features
def analyze_image(image_path):
    # Open image with OpenCV
    img = cv2.imread(image_path)
    # Convert BGR to RGB
    img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    
    # Get image name
    image_name = os.path.basename(image_path)
    
    # --- Color analysis ---
    # Calculate average RGB values
    avg_rgb = np.mean(img_rgb, axis=(0,1))
    # Normalize to 0-1
    norm_rgb = avg_rgb / 255.0
    
    # Simple color classification based on RGB values
    if norm_rgb[0] < 0.3 and norm_rgb[1] < 0.3 and norm_rgb[2] < 0.3:
        color = 'Black'
    elif norm_rgb[1] > norm_rgb[0] and norm_rgb[1] > norm_rgb[2]:
        color = 'Green'
    elif norm_rgb[0] > norm_rgb[1] and norm_rgb[0] > norm_rgb[2]:
        color = 'Red Brown'
    elif norm_rgb[0] > 0.6 and norm_rgb[1] > 0.6 and norm_rgb[2] < 0.4:
        color = 'Yellow'
    elif norm_rgb[0] < 0.5 and norm_rgb[1] < 0.5 and norm_rgb[2] < 0.5:
        color = 'Gray'
    elif norm_rgb[0] > 0.4 and norm_rgb[1] > 0.3 and norm_rgb[2] < 0.3:
        color = 'Brown'
    else:
        color = 'White'
    
    # --- Texture analysis ---
    # Convert to grayscale
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    # Calculate edges using Canny
    edges = cv2.Canny(gray, 50, 150)
    edge_density = np.sum(edges > 0) / edges.size
    
    # Simple texture classification
    if edge_density < 0.05:
        texture = 'Spotted'
    elif edge_density < 0.1:
        texture = 'Striped'
    elif edge_density < 0.15:
        texture = 'Ring Spot'
    elif edge_density < 0.2:
        texture = 'Netted Spot'
    else:
        texture = 'Random Spot'
    
    # --- Morphology analysis ---
    # Calculate area of non-green regions
    hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
    # Define green range in HSV
    lower_green = np.array([40, 40, 40])
    upper_green = np.array([70, 255, 255])
    # Create mask for green areas
    green_mask = cv2.inRange(hsv, lower_green, upper_green)
    non_green_area = np.sum(green_mask == 0)
    total_area = img.shape[0] * img.shape[1]
    damage_ratio = non_green_area / total_area
    
    # Simple morphology classification
    if damage_ratio < 0.1:
        morphology = 'Atrophy'
    elif damage_ratio < 0.2:
        morphology = 'Wilt'
    elif damage_ratio < 0.3:
        morphology = 'Rot'
    elif damage_ratio < 0.4:
        morphology = 'Burn'
    else:
        morphology = 'Perforation'
    
    return image_name, color, texture, morphology
