import xml.etree.ElementTree as ET
import json

# Parse the PubMed XML file
tree = ET.parse('/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/npcdr_retrieval/pubmed_reviews.xml')

# Extract PMIDs and abstracts
articles = []

    pmid = article.find('.//PMID').text
    abstract_elem = article.find('.//AbstractText')
    abstract = abstract_elem.text if abstract_elem is not None else ""
    articles.append({'PMID': pmid, 'Abstract': abstract})

import json

# Load the extracted articles
input_file = '/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/npcdr_retrieval/extracted_articles.json'
with open(input_file, 'r') as f:
    articles_list = json.load(f)

# Prepare the final result list
final_list = []

# Process each article one by one
for idx, art in enumerate(articles_list):
    pmid_str = art['PMID']
    abs_text = art['Abstract']
    
    # Convert to lowercase for easier matching
    abs_lower = abs_text.lower()
    
    # Check inclusion criteria
    is_accepted = False
    
    # First check if curcumin is mentioned
    if 'curcumin' in abs_lower:
        # Check if there's mention of drug combinations with curcumin
        combination_terms = [
            'combination', 'combinations', 'combined', 
            'synergistic', 'synergism', 'synergy',
            'co-administered', 'co-treatment', 'plus', 'and'
        ]
        
        # Check if any combination term appears near curcumin
        for term in combination_terms:
            if term in abs_lower:
                # Check proximity (within 50 words)
                words = abs_lower.split()
                try:
                    curcumin_idx = words.index('curcumin')
                    # Check surrounding words
                    start = max(0, curcumin_idx - 25)
                    end = min(len(words), curcumin_idx + 25)
                    surrounding = words[start:end]

                    continue
