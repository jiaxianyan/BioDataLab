# Core Task
Search the NCBI GEO database for Series published from 2019/10/01 to 2019/10/15 using the query: '(Cancer-related keywords) AND (Drug-related keywords) AND (Patient-related keywords) AND "Homo sapiens"' to collect paired pre- and post-treatment transcriptome data. The datasets must satisfy the following inclusion/exclusion criteria: (i) paired (i.e. from the same patient) pre- and posttreatment patient samples were required; (ii) for microarray data, only datasets from Affymetrix, Agilent and Illumina platforms were collected; (iii) bulk transcriptomes were collected; (iv) datasets, in which drug therapy is combined with other physical herapies, were excluded to ensure hat he perturbation is induced only by drug therapy.


# Output Requirements & File Path
- Output format: JSON list of objects. The list must include all Series obtained from the initial GEO search.
- Each object must contain exactly two fields:
  - `"Series"`: The GEO Series accession number (e.g., "GSE123456").
  - `"Accepted"`: Boolean value (`true` if the Series meets all inclusion/exclusion criteria, `false` otherwise).
- Save Path: Save the complete list to: `/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/cds_db_retrieval.json`
- Intermediate Storage Path: All intermediate code generated during execution, downloaded files, and temporary results must be stored in `/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/cds_db_retrieval/`

library(GEOquery)
library(tidyverse)

# Create output directory if not exists
dir.create("/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/", recursive = TRUE, showWarnings = FALSE)

# Read unique Series list
series_list <- readLines("/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/cds_db_retrieval/unique_series_list.txt")

# Initialize results list
results <- list()

# Define valid microarray platforms
valid_platforms <- c("Affymetrix", "Agilent", "Illumina")

# Process each Series
for (gse_id in series_list) {
  cat("\nProcessing", gse_id, "...\n")
  
  # Download Series data
  gse <- getGEO(gse_id, GSEMatrix = FALSE, destdir = "/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/cds_db_retrieval/")
  
  # Extract platform information
  platform <- gse@platforms[[1]]@title
  cat("Platform:", platform, "\n")
  
  # Check if it's bulk transcriptome (not single-cell)
  is_bulk <- !grepl("single.?cell|scRNA-seq|scRNAseq", gse@description, ignore.case = TRUE)
  cat("Is bulk:", is_bulk, "\n")
  
  # Check platform validity (if microarray)
  is_valid_platform <- any(grepl(paste(valid_platforms, collapse = "|"), platform, ignore.case = TRUE))
  # If it's RNA-seq, platform validity doesn't apply
  if (grepl("RNA-seq|RNAseq", platform, ignore.case = TRUE)) {
    is_valid_platform <- TRUE
  }
  cat("Valid platform:", is_valid_platform, "\n")
  
  # Check for paired pre/post samples
  sample_info <- pData(phenoData(gse))
  # Look for keywords indicating pre/post treatment
  has_paired <- any(grepl("pre.?treatment|baseline|post.?treatment|after.?treatment", sample_info$title, ignore.case = TRUE)) | 
                any(grepl("pre.?treatment|baseline|post.?treatment|after.?treatment", sample_info$characteristics_ch1, ignore.case = TRUE))
  cat("Has paired pre/post samples:", has_paired, "\n")
  
  # Check for combination therapy
  has_combination <- grepl("surgery|radiation|radiotherapy|chemoradiation", gse@description, ignore.case = TRUE) |
                     any(grepl("surgery|radiation|radiotherapy|chemoradiation", sample_info$characteristics_ch1, ignore.case = TRUE))
  cat("Has combination therapy:", has_combination, "\n")
  
  # Determine if accepted (all criteria met)
  accepted <- is_bulk && is_valid_platform && has_paired && !has_combination
  cat("Accepted:", accepted, "\n")
  
  # Add to results
  results[[gse_id]] <- list(Series = gse_id, Accepted = accepted)
}

# Convert to JSON
results_json <- toJSON(results, auto_unbox = TRUE)
cat("\nFinal results JSON:\n", results_json, "\n")

# Save to file
write(results_json, "/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/cds_db_retrieval.json")
cat("\nResults saved to output file.\n")
