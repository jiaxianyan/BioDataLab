with open(xml_path, 'r', encoding='utf-8') as f:

    # Get PMID
    pmid = pubmed_article.find('PMID').text
    
    # Get abstract text
    abstract_text = ""
    abstract = pubmed_article.find('Abstract')
    if abstract:
        abstract_text = " ".join([p.text for p in abstract.find_all('AbstractText')]).lower()
    
    # Check if article discusses BCR-ABL1 fusion neoantigens
    accepted = False
    if abstract_text:
        # Check for BCR-ABL1 related terms
        bcr_abl_terms = ["bcr-abl1", "bcr-abl", "bcr/abl"]
        has_bcr_abl = any(term in abstract_text for term in bcr_abl_terms)

            for ft in fusion_terms:
                if ft in abstract_text:
                    # Check if fusion is linked to neoantigen

                        accepted = True
                        break
    
    articles.append({
        "PMID": pmid,
        "Accepted": accepted
    })

# Save results

import xml.etree.ElementTree as ET
import json
import os
import glob

def process_article(xml_file):
    try:
        # Extract PMID from filename
        pmid = os.path.basename(xml_file).replace("article_", "").replace(".xml", "")
        print(f"\nProcessing PMID: {pmid}")
        
        # Parse XML
        tree = ET.parse(xml_file)
        root = tree.getroot()
        
        # Get abstract text
        abstract_text = ""
        abstract = root.find('.//Abstract')
        if abstract is not None:
            abstract_parts = []
            for t in abstract.findall('.//AbstractText'):
                if t.text is not None:
                    abstract_parts.append(t.text)
            abstract_text = " ".join(abstract_parts).lower()
            print(f"Abstract snippet: {abstract_text[:200]}...")
        
        # Check if article discusses BCR-ABL1 fusion neoantigens
        accepted = False
        if abstract_text:
            # Check for BCR-ABL1 related terms
            has_bcr_abl = any(term in abstract_text for term in ["bcr-abl1", "bcr-abl", "bcr/abl"])
            
            # Check for neoantigen related terms
            has_neoantigen = any(term in abstract_text for term in ["neoantigen", "neo-antigen", "neoepitope", "tumor specific antigen"])
            
            # Check for fusion-specific neoantigen discussion
            if has_bcr_abl and has_neoantigen and "fusion" in abstract_text:
                accepted = True
                print("ACCEPTED: This article discusses BCR-ABL1 fusion neoantigens")
        
        return {
            "PMID": pmid,
            "Accepted": accepted
        }
    except Exception as e:
        print(f"Error processing article {xml_file}: {e}")
        import traceback
        traceback.print_exc()
        return None
