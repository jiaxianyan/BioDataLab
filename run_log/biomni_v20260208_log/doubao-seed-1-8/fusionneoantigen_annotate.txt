Given a full-length fusion protein sequence in `benchmark/dataset/FusionNeoAntigen/full_fusion_protein.fasta` and a breakpoint coordinate in `benchmark/dataset/FusionNeoAntigen/breakpoint.txt`, extract a 27-amino-acid peptide centered on the breakpoint. 

The peptide should consist of 13 amino acids upstream of the breakpoint, the breakpoint amino acid itself, and 13 amino acids downstream.

All intermediate files generated during the extraction process should be stored in `/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/fusionneoantigen_annotate/`.

# Read breakpoint
with open('benchmark/dataset/FusionNeoAntigen/breakpoint.txt', 'r') as f:
    breakpoint_pos = int(f.read().strip())

# Read FASTA sequence
with open('benchmark/dataset/FusionNeoAntigen/full_fusion_protein.fasta', 'r') as f:
    lines = f.readlines()
    # Skip header line
    sequence = ''.join([line.strip() for line in lines[1:]])

# Calculate indices (convert to 0-based for Python)
start_idx = breakpoint_pos - 1 - 13  # 13 upstream of breakpoint
end_idx = breakpoint_pos - 1 + 1 + 13  # breakpoint + 13 downstream

# Extract peptide
peptide = sequence[start_idx:end_idx]

# Verify length
print(f"Extracted peptide length: {len(peptide)} (expected: 27)")
print(f"Extracted peptide: {peptide}")

# Save intermediate file
intermediate_path = '/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/fusionneoantigen_annotate/extracted_peptide.txt'
with open(intermediate_path, 'w') as f:
    f.write(peptide)
print(f"Intermediate file saved to: {intermediate_path}")

# Save final FASTA
output_path = '/root/autodl-tmp/biomni_v20260208_pred_results/doubao-seed-1-8/fusionneoantigen_annotate.fasta'
with open(output_path, 'w') as f:
    f.write('>fusion_junction_peptide\n')
    f.write(peptide + '\n')
print(f"Final FASTA saved to: {output_path}")
