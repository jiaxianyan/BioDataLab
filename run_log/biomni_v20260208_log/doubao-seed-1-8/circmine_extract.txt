library(tidyverse)

setwd("/root/autodl-tmp/biomni_v20260208_tmp/doubao-seed-1-8/circmine_extract/")

# Load the metadata
metadata <- readRDS("metadata_raw.rds")

# Create clean patient data with case identifiers
patient_data <- metadata %>%
  # Extract relevant columns including case identifier
  select(
    case = characteristics_ch1,
    disease_state = characteristics_ch1.1,
    age = characteristics_ch1.2,
    gender = characteristics_ch1.4,
    stage = characteristics_ch1.6
  ) %>%
  # Clean up the columns
  mutate(
    case = str_remove(case, "case: "),
    disease_state = str_remove(disease_state, "disease state: "),
    age = str_remove(age, "age: "),
    age = str_remove(age, " years"),
    gender = str_remove(gender, "gender: "),
    stage = str_remove(stage, "Stage: ")
  ) %>%
  # Convert age to numeric
  mutate(age = as.numeric(age)) %>%
  # Add series and country columns
  mutate(
    series = "GSE101684",
    country = "N/A"
  ) %>%
  # Handle missing values
  mutate(across(everything(), ~ ifelse(is.na(.) | . == "", "N/A", .))) %>%
  # Reorder columns

  # Deduplicate by case identifier to get unique patients
  distinct(case, .keep_all = TRUE) %>%
  # Remove case column from final output
