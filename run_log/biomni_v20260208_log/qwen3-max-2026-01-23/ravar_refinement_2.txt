# Define classification logic
def classify_trait(row):
    reported_trait = str(row["Reported Trait"]).lower()
    trait_label = str(row["Trait Label"]).lower()
    
    # Keywords for material entity (physical things)
    material_entity_keywords = ["abnormality", "tissue", "system", "cholecystitis"]
    # Keywords for material property (measurements, properties)
    material_property_keywords = ["measurement"]
    # Information entity is unlikely here, but included for completeness
    
    if any(kw in reported_trait or kw in trait_label for kw in material_property_keywords):
        return "material property"
    elif any(kw in reported_trait or kw in trait_label for kw in material_entity_keywords):
        return "material entity"
    else:
        # Default to material entity for diseases/abnormalities
        return "material entity"

# Apply classification
