import os
from collections import defaultdict

# Create output directories

# Function to read BED file
def read_bed_file(filename):
    regions = defaultdict(list)
    with open(filename, 'r') as f:
        for line in f:
            if line.strip():
                parts = line.strip().split('\t')
                chrom = parts[0]
                start = int(parts[1])
                end = int(parts[2])
                regions[chrom].append((start, end, line.strip()))
    return regions

# Function to check if two intervals overlap
def intervals_overlap(start1, end1, start2, end2):
    return start1 < end2 and start2 < end1

# Read both files
rbp_regions = read_bed_file("benchmark/dataset/scAPAatlas/rbp_binding_sites.bed")
apa_regions = read_bed_file("benchmark/dataset/scAPAatlas/apa_regions.bed")

# Find intersections
intersected_lines = []

for chrom in rbp_regions:
    if chrom in apa_regions:
        # Sort regions for efficiency (though datasets seem small)
        rbp_list = sorted(rbp_regions[chrom])
        apa_list = sorted(apa_regions[chrom])
        
        # Check all pairs for overlap
        for rbp_start, rbp_end, rbp_line in rbp_list:
            for apa_start, apa_end, apa_line in apa_list:
                if intervals_overlap(rbp_start, rbp_end, apa_start, apa_end):
                    intersected_lines.append(rbp_line)
                    break  # Found overlap, no need to check other APA regions for this RBP

# Write results
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/scapaatlas_annotate.bed"
with open(output_file, 'w') as f:
    for line in intersected_lines:
        f.write(line + '\n')

# Copy to intermediate directory
import shutil
shutil.copy(output_file, "/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/scapaatlas_annotate/")

# Print results
print(f"Intersected results saved to {output_file}")
print(f"Total number of intersected regions: {len(intersected_lines)}")
if intersected_lines:
    print("Sample intersected regions:")

        print(line)
