import scanpy as sc
import pandas as pd
import os

# Define paths
input_file = "benchmark/dataset/DRMef/pbmc3k_cd4_fixed.h5ad"
output_dir = "/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/drmref_annotate"
os.makedirs(output_dir, exist_ok=True)
output_csv = "/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/drmref_annotate.csv"

# Step 1: Read the dataset
adata = sc.read_h5ad(input_file)
print("Dataset loaded successfully.")
print(f"Observation columns: {adata.obs.columns.tolist()}")
print(f"Group labels: {adata.obs['group'].unique()}")

# Step 2: Perform differential expression analysis
sc.tl.rank_genes_groups(adata, groupby='group', method='wilcoxon', reference='sensitive')
print("Differential expression analysis completed.")

# Step 3: Extract results for 'resistant' group
result = sc.get.rank_genes_groups_df(adata, group='resistant')
result.to_csv(os.path.join(output_dir, "diff_expr_results.csv"), index=False)
print("Differential expression results saved.")

# Step 4: Filter significant genes
significant_genes = result[
    (result['pvals_adj'] < 0.05) & (result['logfoldchanges'] > 1.0)
]['names'].tolist()
print(f"Number of significant genes: {len(significant_genes)}")

# Step 5: Save gene names to CSV
gene_df = pd.DataFrame({'gene': significant_genes})
os.makedirs(os.path.dirname(output_csv), exist_ok=True)
gene_df.to_csv(output_csv, index=False)
print(f"Significant gene names saved to {output_csv}.")
