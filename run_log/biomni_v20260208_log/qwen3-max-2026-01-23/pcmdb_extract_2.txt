import os
import json

# Define paths
paper_dir = "benchmark/dataset/PCMDB"
intermediate_dir = "/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23pcmdb_extract_2"
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/pcmdb_extract_2.json"

# Ensure output directory exists
os.makedirs(os.path.dirname(output_file), exist_ok=True)

# List of paper files
paper_files = ["example_1.md", "example_2.md", "example_3.md"]

# Keywords to search for
keywords = ["GFP reporter", "GUS gene reporter", "western blotting"]

# List to store kept paper names
kept_papers = []

# Process each paper
for paper_file in paper_files:
    file_path = os.path.join(paper_dir, paper_file)
    try:
        with open(file_path, 'r') as f:
            content = f.read()
        
        # Check if any keyword is present (case-insensitive)
        contains_keyword = False
        content_lower = content.lower()
        for keyword in keywords:
            if keyword.lower() in content_lower:
                contains_keyword = True
                break
        
        if not contains_keyword:
            # Keep paper name without suffix
            paper_name = os.path.splitext(paper_file)[0]
            kept_papers.append(paper_name)
        
        # Save intermediate result
        intermediate_file = os.path.join(intermediate_dir, f"{paper_file}.checked")
        with open(intermediate_file, 'w') as f:
            f.write("Contains marker methods: " + str(contains_keyword))
            
    except Exception as e:
        print(f"Error processing {file_path}: {e}")

# Save final result
with open(output_file, 'w') as f:
    json.dump(kept_papers, f, indent=2)

print(f"Kept papers: {kept_papers}")
print(f"Results saved to {output_file}")
