from Bio import SeqIO

# Define input and output paths
input_fasta = "benchmark/dataset/CompoDynamics/GCF_009428885.1_ASM942888v1_cds_from_genomic.fna"
output_fasta = "/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/compodynamics_integration.fna"

# Define standard stop codons
stop_codons = {"TAA", "TAG", "TGA"}

# Open output file for writing valid sequences
with open(output_fasta, "w") as out_handle:
    # Parse the input FASTA file
    for record in SeqIO.parse(input_fasta, "fasta"):
        seq = str(record.seq)
        seq_len = len(seq)

        # Check if sequence length is a multiple of 3
        if seq_len % 3 != 0:
            continue

        # Check if sequence starts with "ATG"
        if not seq.startswith("ATG"):
            continue

        # Check if sequence ends with a standard stop codon
        if seq[-3:] not in stop_codons:
            continue

        # Check for internal stop codons
        has_internal_stop = False
        for i in range(0, seq_len - 3, 3):  # Exclude the last codon
            codon = seq[i:i+3]
            if codon in stop_codons:
                has_internal_stop = True
                break

        if has_internal_stop:
            continue

        # If all checks pass, write the sequence to the output file
        SeqIO.write(record, out_handle, "fasta")

print(f"Valid sequences saved to {output_fasta}")
