from Bio import Entrez
import json

# Set email for Entrez
Entrez.email = "example@example.com"

# Load PMIDs from file
with open("/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/adcdb_extract/pmids.txt", "r") as f:
    pmid_list = [line.strip() for line in f if line.strip()]

print(f"Fetching details for {len(pmid_list)} PMIDs...")

# Fetch article details (title and abstract)
articles = []
for pmid in pmid_list:
    try:
        handle = Entrez.efetch(db="pubmed", id=pmid, retmode="xml")
        records = Entrez.read(handle)
        handle.close()
        
        # Extract title and abstract
        article = records["PubmedArticle"][0]["MedlineCitation"]["Article"]
        title = article.get("ArticleTitle", "")
        abstract = ""
        if "Abstract" in article:
            abstract_text = article["Abstract"].get("AbstractText", [])
            if abstract_text:
                abstract = " ".join(str(part) for part in abstract_text)
        
        articles.append({
            "PMID": pmid,
            "Title": title,
            "Abstract": abstract
        })
    except Exception as e:
        print(f"Error fetching PMID {pmid}: {e}")
        articles.append({
            "PMID": pmid,
            "Title": "",
            "Abstract": ""
        })

# Save articles to a JSON file
with open("/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/adcdb_extract/articles.json", "w") as f:
    json.dump(articles, f, indent=2)

print("Article details fetched and saved.")

with open("/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/adcdb_extract/articles.json", "r") as f:
    articles = json.load(f)

activity_keywords = [
    "pharmacology", "mechanism of action", "efficacy", "antibody-drug conjugate", "ADC",
    "biological activity", "in vitro", "in vivo", "cytotoxicity", "target binding",
    "receptor binding", "pharmacokinetics", "pharmacodynamics", "therapeutic effect"
]

    results.append({
