import scanpy as sc
import pandas as pd
import os
import anndata

# Define file paths
mtx_file = "benchmark/dataset/DISCO/refinement/matrix.mtx"
features_file = "benchmark/dataset/DISCO/refinement/features.tsv"
barcodes_file = "benchmark/dataset/DISCO/refinement/barcodes.tsv"
temp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/disco_refinement"
output_rds = "/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/disco_refinement.rds"

# Read the sparse matrix
mtx = sc.read_mtx(mtx_file).X.T  # Transpose to get genes as rows, cells as columns

# Load features (genes) and barcodes (cells)
features = pd.read_csv(features_file, sep="\t", header=None)[0].tolist()
barcodes = pd.read_csv(barcodes_file, sep="\t", header=None)[0].tolist()

# Create AnnData object
adata = anndata.AnnData(X=mtx, obs=pd.DataFrame(index=barcodes), var=pd.DataFrame(index=features))

# Save intermediate AnnData object in the temp directory
temp_h5ad = os.path.join(temp_dir, "disco_refinement.h5ad")
adata.write(temp_h5ad)
print(f"Intermediate AnnData saved to {temp_h5ad}")
