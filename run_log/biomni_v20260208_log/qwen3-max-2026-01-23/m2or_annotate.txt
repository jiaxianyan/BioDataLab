# Step 4: Compare sequences to find the single point mutation
wild_type_seq = open("/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/m2or_annotate/wild_type_seq.txt").read().strip()
mutant_seq = open("/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/m2or_annotate/mutant_seq.txt").read().strip()

mutation = None
if len(wild_type_seq) == len(mutant_seq):
    for i, (wt_aa, mut_aa) in enumerate(zip(wild_type_seq, mutant_seq)):
        if wt_aa != mut_aa:
            mutation = f"{wt_aa}{i+1}{mut_aa}"
            break
else:
    # If lengths differ, use pairwise alignment (though problem states single point mutation)
    from Bio import pairwise2
    alignments = pairwise2.align.globalxx(wild_type_seq, mutant_seq)
    best_alignment = alignments[0]
    aligned_wt, aligned_mut = best_alignment.seqA, best_alignment.seqB
    for i, (wt_aa, mut_aa) in enumerate(zip(aligned_wt, aligned_mut)):
        if wt_aa != mut_aa and wt_aa != "-" and mut_aa != "-":
            # Find original position in wild-type
            original_pos = 0
            for j in range(i+1):
                if aligned_wt[j] != "-":
                    original_pos += 1
            mutation = f"{wt_aa}{original_pos}{mut_aa}"
            break

print("Identified mutation:", mutation)

# Save intermediate mutation result
with open("/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/m2or_annotate/mutation.txt", "w") as f:
    f.write(mutation if mutation else "No mutation found")
