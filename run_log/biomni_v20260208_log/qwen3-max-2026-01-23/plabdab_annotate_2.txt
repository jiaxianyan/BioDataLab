import re
import pandas as pd

def extract_antigen_from_title(title):
    """
    Extract antigen target from antibody-related title.
    Returns standardized antigen name or "None" if not found.
    """
    # Convert to lowercase for case-insensitive matching, but keep original for extraction
    title_lower = title.lower()
    
    # Common patterns for antibody-antigen relationships
    anti_patterns = [
        r'anti[ -]([^,\.;:!?]+)',  # anti-XXX or anti XXX
        r'antibod.* against ([^,\.;:!?]+)',  # antibody against XXX
        r'antibod.* to ([^,\.;:!?]+)',  # antibody to XXX
        r'antibod.* targeting ([^,\.;:!?]+)',  # antibody targeting XXX
        r'anti[ -]bod.* to ([^,\.;:!?]+)',  # antibod to XXX
    ]
    
    # Try each pattern
    for pattern in anti_patterns:
        match = re.search(pattern, title, re.IGNORECASE)
        if match:
            antigen_candidate = match.group(1).strip()
            # Clean up the antigen name
            antigen_candidate = re.sub(r'[\'\"()]', '', antigen_candidate)
            antigen_candidate = antigen_candidate.strip()
            
            # Handle common cases and standardize
            if antigen_candidate.lower() == 'ige':
                return 'IgE'
            elif antigen_candidate.lower() == 'il-8' or 'interleukin-8' in antigen_candidate.lower():
                return 'IL-8'
            elif antigen_candidate.lower() == 'vegf' or 'vascular endothelial growth factor' in antigen_candidate.lower():
                return 'VEGF'
            elif antigen_candidate.lower() == 'pd-1' or 'programmed death-1' in antigen_candidate.lower():
                return 'PD-1'
            elif antigen_candidate.lower() == 'tnf-alpha' or 'tumor necrosis factor alpha' in antigen_candidate.lower():
                return 'TNF-alpha'
            elif antigen_candidate.lower() == 'her2' or 'human epidermal growth factor receptor 2' in antigen_candidate.lower():
                return 'HER2'
            else:
                # Return the extracted term as-is, but capitalize appropriately
                # If it's all caps, keep it; otherwise, title case
                if antigen_candidate.isupper():
                    return antigen_candidate
                else:
                    return antigen_candidate.title()
    
    # Check for other specific patterns
    if 'interleukin' in title_lower or 'il-' in title_lower:
        # Look for IL-X patterns
        il_match = re.search(r'il[-\s]*(\d+)', title, re.IGNORECASE)
        if il_match:
            return f"IL-{il_match.group(1)}"
    
    # If no specific antigen is mentioned (just methods, screening, etc.)
    method_keywords = ['screening', 'method', 'construction', 'expression', 'generation', 'identification', 'phage display', 'library']
    if any(keyword in title_lower for keyword in method_keywords) and not re.search(r'anti[ -]', title, re.IGNORECASE):
        return "None"
    
    # Default case - if we can't identify a specific antigen
    return "None"

# Test the function on the sample titles
test_titles = [

    "Screening and characterization of human phage display libraries",
    "Construction and expression of single-chain Fv antibodies",
    "Generation of high-affinity fully human anti-interleukin-8 antibodies",
    "Screening and identification of a humanized single-chain Fv antibody"
]

print("Testing antigen extraction:")
for i, title in enumerate(test_titles):
    result = extract_antigen_from_title(title)
    print(f"{i}: '{title}' → '{result}'")
