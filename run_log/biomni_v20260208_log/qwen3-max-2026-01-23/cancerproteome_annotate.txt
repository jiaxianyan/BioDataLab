import pandas as pd
from scipy.stats import spearmanr
import os

# Define paths
protein_file = "benchmark/dataset/CancerProteome/protein_abundance.csv"
transcript_file = "benchmark/dataset/CancerProteome/transcript_levels.csv"
output_dir = "/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/cancerproteome_annotate"
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/cancerproteome_annotate.txt"

# Create output directory if it doesn't exist
os.makedirs(output_dir, exist_ok=True)

# Load data
protein_df = pd.read_csv(protein_file)
transcript_df = pd.read_csv(transcript_file)

# Merge datasets on GeneID
merged_df = pd.merge(protein_df, transcript_df, on="GeneID")

# Compute Spearman correlation
correlation, _ = spearmanr(merged_df["Protein_Abundance"], merged_df["Transcript_Level"])

# Save intermediate results
merged_df.to_csv(os.path.join(output_dir, "merged_data.csv"), index=False)

# Save correlation coefficient
with open(output_file, "w") as f:
    f.write(f"{correlation:.2f}")

print(f"Spearman correlation coefficient: {correlation:.2f}")
