# Clean age values by removing " years" suffix and converting to numeric
clean_age <- function(age_str) {
  if (is.na(age_str) || age_str == "") return("N/A")
  # Remove " years" and any extra spaces
  cleaned <- gsub(" years", "", age_str)
  cleaned <- gsub(" ", "", cleaned)
  # Return as numeric or "N/A" if conversion fails
  if (grepl("^[0-9]+$", cleaned)) {
    return(as.numeric(cleaned))
  } else {
    return("N/A")
  }
}
