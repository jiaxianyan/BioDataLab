import requests
import json
import os
import time

# Define search parameters
base_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/"
search_term = '(("circular rna"[All Fields] OR "circrna"[All Fields]) AND "Homo sapiens"[Organism]) AND ("2021/01/01"[Publication Date] : "2021/01/15"[Publication Date])'
db = "gds"

# Perform the search
try:
    # First, get the count and IDs
    search_url = f"{base_url}esearch.fcgi?db={db}&term={search_term}&retmax=100&usehistory=y"
    response = requests.get(search_url)
    
    if response.status_code == 200:
        # Parse the XML response to get the WebEnv and QueryKey
        from xml.etree import ElementTree as ET
        root = ET.fromstring(response.content)
        
        web_env = root.find('WebEnv').text if root.find('WebEnv') is not None else None
        query_key = root.find('QueryKey').text if root.find('QueryKey') is not None else None
        id_list = [id_elem.text for id_elem in root.findall('IdList/Id')]
        
        print(f"Found {len(id_list)} GEO Series entries")
        
        # Get GSE accessions
        gse_accessions = []
        if web_env and query_key:
            # Fetch summaries in batches
            retstart = 0
            retmax = 20
            while retstart < len(id_list):
                summary_url = f"{base_url}esummary.fcgi?db={db}&query_key={query_key}&WebEnv={web_env}&retstart={retstart}&retmax={retmax}"
                summary_response = requests.get(summary_url)
                
                if summary_response.status_code == 200:
                    summary_root = ET.fromstring(summary_response.content)
                    for docsum in summary_root.findall('DocumentSummary'):
                        accession = docsum.get('accession', '')
                        if accession.startswith('GSE'):
                            gse_accessions.append(accession)
                
                retstart += retmax
                time.sleep(0.3)  # Be respectful to NCBI servers
        
        print(f"GSE accessions: {gse_accessions}")
        
        # Save the list of accessions
        with open("/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/circmine_retrieval/gse_accessions.txt", "w") as f:
            for acc in gse_accessions:
                f.write(acc + "\n")
                
    else:
        print(f"Error: HTTP {response.status_code}")
        gse_accessions = []
        
except Exception as e:
    print(f"Error: {e}")
    gse_accessions = []

import requests
import json
import os
import time
from xml.etree import ElementTree as ET

# Read the GSE accessions
with open("/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/circmine_retrieval/gse_accessions.txt", "r") as f:
    gse_accessions = [line.strip() for line in f.readlines()]

print(f"Analyzing {len(gse_accessions)} GSE entries...")

# Function to check if a GSE provides accessible human circRNA expression data
def check_gse_circrna_data(gse_id):
    # Extract the numeric part of the GSE ID
    gds_id = gse_id.replace("GSE", "")
    
    # Fetch the GSE record
    base_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/"
    summary_url = f"{base_url}esummary.fcgi?db=gds&id={gds_id}"
    
    try:
        response = requests.get(summary_url)
        if response.status_code == 200:
            root = ET.fromstring(response.content)
            
            # Check if there's any document summary
            docsum = root.find('DocumentSummary')
            if docsum is not None:
                # Get the title and description
                title = docsum.find('title').text if docsum.find('title') is not None else ""
                description = docsum.find('summary').text if docsum.find('summary') is not None else ""
                
                # Check if the study mentions circRNA and human samples
                has_circrna = any(term in (title + " " + description).lower() for term in ["circular rna", "circrna", "circ-rna"])
                has_human = "homo sapiens" in (title + " " + description).lower() or "human" in (title + " " + description).lower()
                
                # Check if there's accessible data (has samples and platform info)
                has_samples = docsum.find('Samples') is not None and len(docsum.find('Samples')) > 0
                
                # If it has circRNA, human samples, and accessible data, accept it
                return has_circrna and has_human and has_samples
                
    except Exception as e:
        print(f"Error analyzing {gse_id}: {e}")
    
    return False

# Analyze each GSE
results = []
for gse_id in gse_accessions:
    print(f"Checking {gse_id}...")
    accepted = check_gse_circrna_data(gse_id)
    results.append({"Series": gse_id, "Accepted": accepted})
    time.sleep(0.5)  # Be respectful to NCBI servers

# Save results to JSON
output_path = "/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/circmine_retrieval.json"
with open(output_path, "w") as f:
    json.dump(results, f, indent=2)

print(f"Results saved to {output_path}")
print("Results:")
for result in results:
    print(f"{result['Series']}: {'Accepted' if result['Accepted'] else 'Rejected'}")

# Function to check if a GSE provides accessible human circRNA expression data
def check_gse_circrna_data(gse_id):
    # Extract the numeric part of the GSE ID
    gds_id = gse_id.replace("GSE", "")
    
    # Fetch the GSE record
    base_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/"
    summary_url = f"{base_url}esummary.fcgi?db=gds&id={gds_id}"
    
    try:
        with urllib.request.urlopen(summary_url) as response:
            content = response.read()
            root = ET.fromstring(content)
            
            # Check if there's any document summary
            docsum = root.find('DocumentSummary')
            if docsum is not None:
                # Get the title and description
                title = docsum.find('title').text if docsum.find('title') is not None else ""
                description = docsum.find('summary').text if docsum.find('summary') is not None else ""
                
                # Check if the study mentions circRNA and human samples
                has_circrna = any(term in (title + " " + description).lower() for term in ["circular rna", "circrna", "circ-rna"])
                has_human = "homo sapiens" in (title + " " + description).lower() or "human" in (title + " " + description).lower()
                
                # Check if there's accessible data (has samples and platform info)
                has_samples = docsum.find('Samples') is not None and len(docsum.find('Samples')) > 0
                
                # If it has circRNA, human samples, and accessible data, accept it
                return has_circrna and has_human and has_samples
                
    except Exception as e:
        print(f"Error analyzing {gse_id}: {e}")
    
    return False
