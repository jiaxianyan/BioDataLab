# Core Task
You are a specialized Bioinformatics NLP Agent acting as an expert in Synthetic Biology and Genetic Code Expansion (GCE). Your core competency is the high-fidelity extraction of experimental data from scientific literature regarding the site-specific incorporation of non-canonical amino acids (ncAAs). You will process specific PDF publications to curate a structured dataset of orthogonal translation systems (aaRS/tRNA pairs), their origins, and their applications in protein engineering.

# Workflow Instructions
- `Literature Processing`: Parse the provided PDF files located at the specified path.
- `Data Extraction & Formatting`:
  1) ncAA Identifiers: Extract abbreviations and full names. Use "not available" if missing.
  2) aaRS ID Construction: Follow the syntax [Org Abbr]-[Natural Substrate]RS ([Mutations/Wild Type]). (e.g., Ec-MetRS (wild type) or Mb-PylRS (Y271C, N311Q, Y349F, V366C)).
  3) tRNA ID Construction: Follow the syntax [Org Abbr]-tRNA [Natural AA] [Anticodon]. (e.g., Bs-tRNA Tyr CUA). Each tRNA has a name composed of three words: (a) abbreviation of the organism from which it was derived; (b) tRNA; (c) AA naturally transported by the tRNA; (d) anticodon. For example, 'Bs-tRNA Tyr CUA' indicates a tRNA naturally found in Bacillus subtilis, which transports tyrosine, whose anticodon is CUA. 
  4) Organism Mapping: Identify full scientific names for both aaRS and tRNA origins.
  5) Protein Application: Identify the target protein. Critical: If multiple proteins are tested in one paper, create a unique row for each protein.
  6) Site Mapping: If multiple independent sites are tested, separate with ; and sort by residue number (ascending). If multiple sites are incorporated simultaneously (jointly), use +.
  7) Host System: Identify the test organism or specify "in vitro".

    return text

def parse_ncAA_info(text):
    # Look for patterns like "p-azido-L-phenylalanine (pAzF)" or similar
    # This is a simplified approach; real parsing would need more robust regex/NLP
    ncAA_abbr = "not available"
    ncAA_name = "not available"

    patterns = [
        r"([A-Za-z0-9\-\(\)]+)\s*\(([^)]+?)\)",  # abbr (name)
        r"([^)]+?)\s*\(([A-Za-z0-9\-\(\)]+)\)"   # name (abbr)
    ]
    
    for pattern in patterns:
        matches = re.findall(pattern, text[:1000])  # Look in first 1000 chars
        if matches:
            for match in matches:
                if len(match[0]) > 2 and len(match[1]) > 2:  # Basic sanity check

                        ncAA_abbr = match[0]
                        ncAA_name = match[1]

                        ncAA_name = match[0]
                    break
            if ncAA_abbr != "not available":
                break
    
    return ncAA_abbr, ncAA_name

def parse_aaRS_info(text):
    # Look for patterns like "Ec-TyrRS" or "Methanocaldococcus jannaschii tyrosyl-tRNA synthetase"
    aaRS_id = "not available"
    aaRS_origin = "not available"
    
    # Common organism abbreviations
    org_map = {
        "Ec": "Escherichia coli",
        "Mj": "Methanocaldococcus jannaschii", 
        "Mb": "Methanosarcina barkeri",
        "Bs": "Bacillus subtilis",
        "Sc": "Saccharomyces cerevisiae"
    }
    
    # Look for aaRS patterns
    aaRS_pattern = r"([A-Z][a-z]?)-([A-Za-z]+)RS"
    matches = re.findall(aaRS_pattern, text)
    
    if matches:
        org_abbr, substrate = matches[0]
        aaRS_id = f"{org_abbr}-{substrate}RS"
        
        # Check for mutations
        mutation_pattern = r"\(([A-Z]\d+[A-Z],?\s*)+\)"
        mutation_matches = re.findall(mutation_pattern, text)
        if mutation_matches:
            aaRS_id += f" ({mutation_matches[0]})"
        else:
            aaRS_id += " (wild type)"
            
        if org_abbr in org_map:
            aaRS_origin = org_map[org_abbr]
    
    return aaRS_id, aaRS_origin

def parse_tRNA_info(text):

    # Look for tRNA patterns
    tRNA_pattern = r"([A-Z][a-z]?)-tRNA\s*[\^]?([A-Za-z]+)\s*([AUCG]{3})"
    matches = re.findall(tRNA_pattern, text)

def parse_protein_info(text):
    # This is highly paper-specific; looking for common patterns
    protein = "not available"
    position = "not available"
    host = "not available"
    
    # Look for protein names
    protein_patterns = [
        r"(green fluorescent protein|GFP)",
        r"(superoxide dismutase|SOD)",
        r"(dihydrofolate reductase|DHFR)",
        r"(ubiquitin)",
        r"(calmodulin)"
    ]
    
    for pattern in protein_patterns:
        if re.search(pattern, text, re.IGNORECASE):
            protein = re.search(pattern, text, re.IGNORECASE).group(1)
            break
    
    # Look for positions
    pos_pattern = r"position\s+(\d+)|site\s+(\d+)|(\d+)\s+residue"
    pos_matches = re.findall(pos_pattern, text, re.IGNORECASE)
    if pos_matches:
        position = pos_matches[0][0] or pos_matches[0][1] or pos_matches[0][2]
    
    # Look for host organisms
    host_patterns = [
        ("Escherichia coli", ["E\. coli", "Escherichia coli"]),
        ("Saccharomyces cerevisiae", ["S\. cerevisiae", "yeast"]),
        ("in vitro", ["in vitro"])

        for pattern in patterns:
            if re.search(pattern, text, re.IGNORECASE):
                host = host_name
                break
        if host != "not available":
            break
    
    return protein, position, host

    """Extract text from first few pages of PDF"""

def extract_text_pdftotext(pdf_path, max_pages=5):
    """Extract text using pdftotext command-line tool"""
    temp_txt = pdf_path.replace('.pdf', '_temp.txt')
    cmd = f"pdftotext -f 1 -l {max_pages} -layout '{pdf_path}' '{temp_txt}'"
    subprocess.run(cmd, shell=True, capture_output=True)
    
    if os.path.exists(temp_txt):
        with open(temp_txt, 'r', encoding='utf-8', errors='ignore') as f:
            text = f.read()
        os.remove(temp_txt)
        return text
    else:
        return ""

def parse_ncAA_info(text):

        r"([^)]+?)\s*\(([A-Za-z0-9\-\(\)]{2,})\)"
    ]
    
    search_text = text[:2000]
    
    for pattern in patterns:
        matches = re.findall(pattern, search_text)
        if matches:
            for match in matches:
                part1, part2 = match[0], match[1]
                if (len(part1) <= 10 and any(c.isupper() for c in part1) and
