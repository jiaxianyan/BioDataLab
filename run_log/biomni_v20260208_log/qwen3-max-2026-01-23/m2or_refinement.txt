import os
from rdkit import Chem
from rdkit.Chem import inchi

# Define paths
input_file = "benchmark/dataset/M2OR/1a1e_ligand.mol2"
tmp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/m2or_refinement/"
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/m2or_refinement.txt"

# Create temporary directory if it doesn't exist
os.makedirs(tmp_dir, exist_ok=True)
print(f"Created temporary directory: {tmp_dir}")

# Create output directory if it doesn't exist
output_dir = os.path.dirname(output_file)
os.makedirs(output_dir, exist_ok=True)
print(f"Created output directory: {output_dir}")

# Read the MOL2 file using RDKit
print("Reading MOL2 file with RDKit...")
mol = Chem.MolFromMol2File(input_file, removeHs=False)
if mol is None:
    print("Failed to read molecule from MOL2 file!")
    # Try alternative approach with sanitize=False
    mol = Chem.MolFromMol2File(input_file, sanitize=False, removeHs=False)
    if mol is None:
        print("Still failed with sanitize=False!")
    else:
        print("Successfully read with sanitize=False")
else:
    print("Successfully read MOL2 file!")

if mol is not None:
    # Generate InChIKey
    print("Generating InChIKey...")
    inchi_key = inchi.MolToInchiKey(mol)
    print(f"InChIKey: {inchi_key}")
    
    # Save to output file
    with open(output_file, 'w') as f:
        f.write(inchi_key)
    print(f"InChIKey saved to: {output_file}")
    
    # Also save to tmp directory for reference
    tmp_output = os.path.join(tmp_dir, "inchikey_result.txt")
    with open(tmp_output, 'w') as f:
        f.write(inchi_key)
    print(f"InChIKey also saved to tmp directory: {tmp_output}")
else:
    print("ERROR: Could not generate molecule object from MOL2 file")
