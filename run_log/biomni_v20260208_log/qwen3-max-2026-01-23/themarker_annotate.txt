import json
import os

# Define keyword mappings for each category
keyword_map = {
    'PDY': ['response to treatment', 'pharmacodynamic', 'drug response', 'therapeutic effect', 'biological response'],
    'SAF': ['toxicity', 'adverse', 'safety', 'side effect', 'hepatotoxicity', 'nephrotoxicity', 'cardiotoxicity'],
    'MOI': ['monitor', 'track', 'follow-up', 'disease progression', 'treatment progress', 'longitudinal'],
    'PRD': ['predict', 'prognostic', 'likely to respond', 'therapy selection', 'treatment selection', 'responder'],
    'SUR': ['surrogate', 'clinical endpoint', 'substitute for', 'replacement for']
}

# Normalize keywords to lowercase for matching
for k in keyword_map:
    keyword_map[k] = [kw.lower() for kw in keyword_map[k]]

def classify_biomarker(function_text):
    function_lower = function_text.lower()
    scores = {cat: 0 for cat in keyword_map}
    
    for cat, keywords in keyword_map.items():
        for kw in keywords:
            if kw in function_lower:
                scores[cat] += 1
                
    # Return category with highest score; default to MOI if tie or no match
    best_cat = max(scores, key=scores.get)
    if scores[best_cat] == 0:
        return 'MOI'  # Default assumption for monitoring if no clear signal
    return best_cat

# Read and parse the file
with open('benchmark/dataset/TheMarker/biomarker.txt', 'r') as f:

functions_list = []

for line in lines[1:]:
    parts = line.strip().split('\t')
    if len(parts) > function_col_index:
        func = parts[function_col_index]
        functions_list.append(func)
        label = classify_biomarker(func)
        labels.append(label)

        # In case of malformed line, default to MOI

import json
import os

# Create directories
os.makedirs('/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/themarker_annotate', exist_ok=True)
os.makedirs('/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/', exist_ok=True)

# Read input file
with open('benchmark/dataset/TheMarker/biomarker.txt', 'r') as f:
    lines = f.readlines()

# Parse header to find Function column
header = lines[0].strip().split('\t')
function_idx = header.index('Function')

# Classification keywords
pdy_words = ['response to treatment', 'pharmacodynamic', 'drug response', 'therapeutic effect', 'biological response', 'treatment effect']
saf_words = ['toxicity', 'adverse', 'safety', 'side effect', 'hepatotoxicity', 'nephrotoxicity', 'cardiotoxicity', 'damage', 'injury']
moi_words = ['monitor', 'track', 'follow-up', 'disease progression', 'treatment progress', 'longitudinal', 'measure', 'assess']
prd_words = ['predict', 'prognostic', 'likely to respond', 'therapy selection', 'treatment selection', 'responder', 'forecast', 'indicate response']
sur_words = ['surrogate', 'clinical endpoint', 'substitute for', 'replacement for', 'correlate with outcome']

all_words = {
    'PDY': pdy_words,
    'SAF': saf_words,
    'MOI': moi_words,
    'PRD': prd_words,
    'SUR': sur_words
}

# Convert to lowercase
for cat in all_words:
    all_words[cat] = [w.lower() for w in all_words[cat]]

labels = []
functions = []

for line in lines[1:]:
    parts = line.strip().split('\t')
    if len(parts) <= function_idx:
        labels.append('MOI')
        functions.append('')
        continue
        
    func = parts[function_idx]
    functions.append(func)
    func_lower = func.lower()
    
    # Score each category
    scores = {'PDY': 0, 'SAF': 0, 'MOI': 0, 'PRD': 0, 'SUR': 0}
    for cat, words in all_words.items():
        for word in words:
            if word in func_lower:
                scores[cat] += 1
                
    # Select best category
    best_cat = max(scores, key=scores.get)
    if scores[best_cat] == 0:
        labels.append('MOI')
    else:
        labels.append(best_cat)

# Save intermediate
inter_dir = '/root/autodl-tmp/biomni_v20260208_tmp/qwen3-max-2026-01-23/themarker_annotate'
with open(f'{inter_dir}/functions.json', 'w') as f:
    json.dump(functions, f)
with open(f'{inter_dir}/labels.json', 'w') as f:
    json.dump(labels, f)

# Save final
out_file = '/root/autodl-tmp/biomni_v20260208_pred_results/qwen3-max-2026-01-23/themarker_annotate.json'
with open(out_file, 'w') as f:
    json.dump(labels, f)

print(f"Processed {len(labels)} biomarkers")
print(f"Output saved to {out_file}")
