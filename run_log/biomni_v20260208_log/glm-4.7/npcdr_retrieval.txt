# Load the abstracts data
intermediate_file = os.path.join(tmp_dir, "step2_abstracts_retrieved.json")
with open(intermediate_file, 'r') as f:
    articles_data = json.load(f)

# Keywords and phrases that indicate drug combinations with curcumin
combination_keywords = [
    'curcumin combination',
    'curcumin with',
    'curcumin plus',
    'curcumin and',
    'curcumin combined',
    'curcumin co-administration',
    'curcumin together with',
    'curcumin synerg',
    'curcumin potentiat',
    'curcumin enhance',
    'curcumin adjunct',
    'curcumin therapy combination',
    'curcumin-based combination',
    'curcumin-containing',
    'curcumin formulation with',
    'curcumin therapy with'
]

# Analyze each article
results = []

for article in articles_data:
    pmid = article["PMID"]
    title = article["Title"]
    abstract = article["Abstract"]
    
    print(f"\n--- PMID: {pmid} ---")
    print(f"Title: {title[:100]}...")
    
    # Convert to lowercase for analysis
    abstract_lower = abstract.lower()
    title_lower = title.lower()
    
    # Check for mentions of curcumin
    has_curcumin = 'curcumin' in abstract_lower or 'curcumin' in title_lower
    
    # Check for drug combination mentions
    has_combination = False
    combination_found = []
    
    for keyword in combination_keywords:
        if keyword in abstract_lower or keyword in title_lower:
            has_combination = True
            combination_found.append(keyword)
    
    # Also check for specific drug names with curcumin
    # Common drugs that might be combined with curcumin
    drug_names = ['chemotherapy', 'paclitaxel', 'doxorubicin', 'cisplatin', '5-fu', 
                  '5-fluorouracil', 'gemcitabine', 'docetaxel', 'vincristine', 
                  'methotrexate', 'sorafenib', 'erlotinib', 'gefitinib', 'imatinib']
    
    for drug in drug_names:
        if drug in abstract_lower:
            # Check if curcumin is mentioned in the same context
            if 'curcumin' in abstract_lower:
                # Check proximity (within 100 words)
                words = abstract_lower.split()
                curcumin_indices = [i for i, w in enumerate(words) if 'curcumin' in w]
                drug_indices = [i for i, w in enumerate(words) if drug in w]
                
                for ci in curcumin_indices:
                    for di in drug_indices:
                        if abs(ci - di) < 50:  # Within 50 words
                            has_combination = True
                            combination_found.append(f"curcumin near {drug}")
                            break
    
    # Determine if article is accepted
    accepted = has_curcumin and has_combination
    
    result = {
        "PMID": pmid,
        "Accepted": accepted
    }
    
    results.append(result)
    
    print(f"Has curcumin: {has_curcumin}")
    print(f"Has combination: {has_combination}")
    if combination_found:
        print(f"Combination evidence: {combination_found[:3]}")
    print(f"ACCEPTED: {accepted}")

# Save intermediate analysis
intermediate_file = os.path.join(tmp_dir, "step3_analysis_results.json")
with open(intermediate_file, 'w') as f:
    json.dump({
        "analysis": results,
        "details": articles_data

print(f"\nSaved analysis results to: {intermediate_file}")
print(f"\nTotal articles analyzed: {len(results)}")
print(f"Accepted articles: {sum(1 for r in results if r['Accepted'])}")
