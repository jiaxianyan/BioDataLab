import os
import re

print("Systematically extracting marker information...")

# Define paths
tmp_dir = '/root/autodl-tmp/biomni_v20260208_tmp/glm-4.7/pcmdb_extract/'
text_file = os.path.join(tmp_dir, 'extracted_text.txt')

with open(text_file, 'r', encoding='utf-8') as f:
    full_text = f.read()

# Extract all AT gene IDs with their context
at_gene_pattern = r'AT\d+G\d+'
at_matches = re.finditer(at_gene_pattern, full_text)

markers = []

for match in at_matches:
    # Get context around the AT gene ID
    start = max(0, match.start() - 200)
    end = min(len(full_text), match.end() + 200)
    context = full_text[start:end]
    
    gene_id = match.group()
    
    # Try to extract gene name from context (usually before the AT ID in parentheses)
    gene_name_pattern = r'([A-Z][A-Za-z0-9\-]+)\s*\(' + re.escape(gene_id) + r'\)'
    gene_name_match = re.search(gene_name_pattern, context)
    
    if gene_name_match:
        gene_name = gene_name_match.group(1)
    else:
        gene_name = ""
    
    # Try to extract cell type from context
    cell_type = ""
    
    # Look for cell type keywords
    cell_type_patterns = [
        r'stele-specific',
        r'cortex\s+cells?',
        r'nonhair\s+cells?',
        r'root\s+hair\s+cells?',
        r'lateral\s+root\s+cap\s+cells?',
        r'endodermis\s+cells?',
        r'xylem\s+cells?',
        r'phloem\s+cells?',
        r'columella',
        r'cluster\s+\d+',
        r'enriched\s+in\s+([A-Za-z\s]+?)(?:[,.\s]|cell)',
        r'expressed\s+primarily\s+in\s+([A-Za-z\s]+?)(?:[,.\s]|cell)',
        r'most\s+highly\s+expressed\s+in\s+([A-Za-z\s]+?)(?:[,.\s]|cell)',
        r'specific\s+to\s+([A-Za-z\s]+?)(?:[,.\s]|cell)'
    ]
    
    for pattern in cell_type_patterns:
        match = re.search(pattern, context, re.IGNORECASE)
        if match:
            if 'cluster' in pattern.lower():
                cell_type = match.group(0)
            else:
                cell_type = match.group(1) if match.groups() else match.group(0)
            cell_type = cell_type.strip()
            if cell_type:
                break
    
    if cell_type:
        markers.append({
            'gene_id': gene_id,
            'gene_name': gene_name,
            'cell_type': cell_type,
            'context': context
        })

print(f"\nExtracted {len(markers)} potential marker entries")

# Save detailed extraction
detailed_file = os.path.join(tmp_dir, 'detailed_markers.txt')
with open(detailed_file, 'w', encoding='utf-8') as f:
    for i, marker in enumerate(markers):
        f.write(f"=== MARKER {i+1} ===\n")
        f.write(f"Gene ID: {marker['gene_id']}\n")
        f.write(f"Gene Name: {marker['gene_name']}\n")
        f.write(f"Cell Type: {marker['cell_type']}\n")
        f.write(f"Context: {marker['context']}\n")
        f.write("\n")

print(f"✓ Detailed markers saved to: {detailed_file}")

# Print summary
print("\nExtracted markers summary:")
for i, marker in enumerate(markers):
