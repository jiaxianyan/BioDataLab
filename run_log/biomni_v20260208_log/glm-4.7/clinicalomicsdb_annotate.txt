# Create binary labels for AUROC (1 for responder, 0 for non-responder)

for i, gene in enumerate(gene_names):

    # Split into responder and non-responder groups
    responder_expr = gene_expression[responder_mask]
    non_responder_expr = gene_expression[non_responder_mask]
    
    # Perform Wilcoxon rank-sum test (Mann-Whitney U test)
    try:
        stat, p_value = mannwhitneyu(responder_expr, non_responder_expr, alternative='two-sided')

        # Handle edge cases (e.g., all values are the same)
        p_value = 1.0
    
    # Calculate AUROC
