# Function to search PubChem for drug information
def search_pubchem(drug_name):
    """
    Search PubChem for drug information including generic and brand names
    """
    try:
        # Search for the compound in PubChem
        search_url = f"https://pubchem.ncbi.nlm.nih.gov/rest/pug/compound/name/{drug_name}/cids/JSON"
        response = requests.get(search_url, timeout=10)
        
        if response.status_code == 200:
            data = response.json()
            if 'IdentifierList' in data and 'CID' in data['IdentifierList']:
                cids = data['IdentifierList']['CID']
                if cids:
                    cid = cids[0]  # Get the first CID
                    
                    # Get synonyms for this CID
                    synonym_url = f"https://pubchem.ncbi.nlm.nih.gov/rest/pug/compound/cid/{cid}/synonyms/TXT"
                    response2 = requests.get(synonym_url, timeout=10)
                    
                    if response2.status_code == 200:
                        synonyms = response2.text.strip().split('\n')
                        
                        # Try to identify generic name and brand names
                        # Generic name is often the IUPAC name or the first synonym
                        # Brand names are typically capitalized and distinctive
                        generic_name = drug_name
                        brand_names = []
                        
                        for synonym in synonyms:
                            synonym = synonym.strip()
                            if synonym and len(synonym) > 2:
                                # Heuristics to identify brand names
                                if synonym[0].isupper() and ' ' not in synonym and synonym != drug_name:
                                    if not any(c.isdigit() for c in synonym):
                                        brand_names.append(synonym)
                        
                        return {
                            'query_name': drug_name,
                            'generic_name': generic_name,
                            'brand_name': '; '.join(brand_names[:3]) if brand_names else 'N/A'
                        }
        return {
            'query_name': drug_name,
            'generic_name': drug_name,
            'brand_name': 'N/A'
        }
    except Exception as e:
        print(f"Error searching for {drug_name}: {e}")
        return {
            'query_name': drug_name,
            'generic_name': drug_name,
            'brand_name': 'N/A'
        }
