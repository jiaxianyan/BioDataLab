# Find differences
differences = []
mutation = None

for i in range(len(wild_type_sequence)):
    if i < len(mutant_sequence):
        if wild_type_sequence[i] != mutant_sequence[i]:
            wt_aa = wild_type_sequence[i]
            mut_aa = mutant_sequence[i]
            position = i + 1  # 1-based indexing
            mutation = f"{wt_aa}{position}{mut_aa}"
            differences.append((position, wt_aa, mut_aa))

print(f"\nTotal differences found: {len(differences)}")

if differences:
    print(f"Mutation: {mutation}")
else:
    print("No mutations found - sequences are identical")

# Save comparison results
comparison_result = {

with open(tmp_dir / "comparison_result.json", "w") as f:
