print("\n" + "="*80)
print("ANALYZING DATA AVAILABILITY STATEMENTS")
print("="*80)

records = medline_records.split("\n\n")
availability_info = {}

for record in records:
    pmid = None
    availability = ""
    abstract = ""
    
    lines = record.split("\n")
    for line in lines:
        if line.startswith("PMID- "):
            pmid = line[6:].strip()
        elif line.startswith("AB  - ") or line.startswith("AB- "):
            abstract += line[6:] + " "
        elif line.startswith("OT  - ") or line.startswith("OT- "):
            # Check for data availability in other tags
            if "availability" in line.lower() or "GEO" in line or "GSE" in line:
                availability += line[6:] + " "
    
    if pmid:
        # Also check abstract for availability
        if "availability" in abstract.lower() or "GEO" in abstract or "GSE" in abstract or "deposited" in abstract.lower():
            availability = abstract
        
        availability_info[pmid] = {
            'PMID': pmid,
            'Abstract': abstract.strip(),
            'Availability': availability.strip()
        }
        
        print(f"\nPMID: {pmid}")
        print(f"Availability info: {availability[:200]}..." if len(availability) > 200 else f"Availability info: {availability}")

# Save availability info
