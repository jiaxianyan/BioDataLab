from Bio import SeqIO
import os

input_file = "benchmark/dataset/CompoDynamics/GCF_009428885.1_ASM942888v1_cds_from_genomic.fna"
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-pro-preview/compodynamics_integration.fna"

# Define standard stop codons
stop_codons = {"TAA", "TAG", "TGA"}

valid_sequences = []
total_seqs = 0
kept_seqs = 0

print(f"Processing file: {input_file}")

with open(output_file, "w") as out_handle:
    for record in SeqIO.parse(input_file, "fasta"):
        total_seqs += 1
        seq = str(record.seq).upper()
        
        # Criterion 1: Length is a multiple of 3
        if len(seq) % 3 != 0:
            continue
            
        # Criterion 2: Starts with "ATG"
        if not seq.startswith("ATG"):
            continue
            
        # Criterion 3: Ends with a standard stop codon
        if seq[-3:] not in stop_codons:
            continue
            
        # Criterion 4: Contains an internal stop codon
        # We check codons from index 0 up to len(seq)-3 (excluding the final stop codon)
        # range(start, stop, step)
        has_internal_stop = False
        for i in range(0, len(seq) - 3, 3):
            codon = seq[i:i+3]
            if codon in stop_codons:
                has_internal_stop = True
                break
        
        if has_internal_stop:
            continue
            
        # If all criteria are met, keep the sequence
        valid_sequences.append(record)
        SeqIO.write(record, out_handle, "fasta")
        kept_seqs += 1

print(f"Total sequences processed: {total_seqs}")
print(f"Valid sequences kept: {kept_seqs}")
print(f"Output saved to: {output_file}")</execute>
