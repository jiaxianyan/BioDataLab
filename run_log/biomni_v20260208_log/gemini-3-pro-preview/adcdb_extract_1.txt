import os

pdf_dir = "benchmark/dataset/ADCdb"
pdf_files = ["ref2.pdf", "ref4.pdf", "ref5.pdf"]
output_dir = "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-pro-preview/adcdb_extract/"

extracted_texts = {}

for pdf_file in pdf_files:
    pdf_path = os.path.join(pdf_dir, pdf_file)
    print(f"Extracting text from {pdf_path}...")
    try:
        with pdfplumber.open(pdf_path) as pdf:
            text = ""
            for page in pdf.pages:

            extracted_texts[pdf_file] = text
            # Save raw text for inspection
            with open(os.path.join(output_dir, f"{pdf_file}.txt"), "w") as f:
                f.write(text)
    except Exception as e:
        print(f"Error extracting {pdf_file}: {e}")

# Print the first 1000 characters of each extracted text to get an idea of the content
