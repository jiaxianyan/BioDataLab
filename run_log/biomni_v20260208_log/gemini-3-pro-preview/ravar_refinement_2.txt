    trait_name = item['Reported Trait']
    trait_label = item.get('Trait Label', '')
    ontology_id = item.get('Trait Ontology ID', '')
    
    # Classification Logic

    elif 'score' in lower_name or 'score' in lower_label:
        category = "information entity"
    elif 'assessment' in lower_name or 'assessment' in lower_label:
        category = "information entity"

    # HP terms are usually phenotypes (Material Property)
