import pandas as pd
from scipy.stats import spearmanr
import os

# Define file paths
protein_file = 'benchmark/dataset/CancerProteome/protein_abundance.csv'
transcript_file = 'benchmark/dataset/CancerProteome/transcript_levels.csv'
output_dir = '/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-pro-preview/cancerproteome_annotate'
final_output_file = '/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-pro-preview/cancerproteome_annotate.txt'

# Load data
print("Loading data...")
df_protein = pd.read_csv(protein_file)
df_transcript = pd.read_csv(transcript_file)

print(f"Protein data shape: {df_protein.shape}")
print(f"Transcript data shape: {df_transcript.shape}")

# Merge data on GeneID
print("Merging data...")
merged_df = pd.merge(df_protein, df_transcript, on='GeneID')
print(f"Merged data shape: {merged_df.shape}")

# Check for missing values
if merged_df[['Protein_Abundance', 'Transcript_Level']].isnull().any().any():
    print("Warning: Missing values detected. Dropping rows with missing values.")
    merged_df = merged_df.dropna(subset=['Protein_Abundance', 'Transcript_Level'])
    print(f"Shape after dropping missing values: {merged_df.shape}")

# Save merged data for verification (optional but good practice)
merged_df.to_csv(os.path.join(output_dir, 'merged_data.csv'), index=False)

# Calculate Spearman correlation
print("Calculating Spearman correlation...")
correlation, p_value = spearmanr(merged_df['Protein_Abundance'], merged_df['Transcript_Level'])
print(f"Spearman correlation: {correlation}")
print(f"P-value: {p_value}")

# Save the result
with open(final_output_file, 'w') as f:
    f.write(f"{correlation}\n")

print(f"Result saved to {final_output_file}")</execute>
