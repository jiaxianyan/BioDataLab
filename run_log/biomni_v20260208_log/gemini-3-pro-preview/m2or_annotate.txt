from Bio import SeqIO
import json

# Paths
mutant_path = "benchmark/dataset/M2OR/mutant_M2OR.fasta"
wt_path = "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-pro-preview/m2or_annotate/P08172.fasta"
output_path = "/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-pro-preview/m2or_annotate.json"

# Load sequences
mutant_record = next(SeqIO.parse(mutant_path, "fasta"))
wt_record = next(SeqIO.parse(wt_path, "fasta"))

mutant_seq = str(mutant_record.seq)
wt_seq = str(wt_record.seq)

# Compare sequences
print(f"Wild-type length: {len(wt_seq)}")
print(f"Mutant length: {len(mutant_seq)}")

if len(wt_seq) != len(mutant_seq):
    print("Warning: Sequence lengths differ!")

mutations = []
for i, (wt_aa, mut_aa) in enumerate(zip(wt_seq, mutant_seq)):
    if wt_aa != mut_aa:
        mutation_str = f"{wt_aa}{i+1}{mut_aa}"
        mutations.append(mutation_str)
        print(f"Found mutation: {mutation_str}")

if len(mutations) == 0:
    print("No mutations found.")
    final_mutation = "None"
elif len(mutations) > 1:
    print(f"Multiple mutations found: {mutations}. Selecting the first one or checking if this is expected.")
    # Based on the prompt "single point mutation", we ideally expect one.
    final_mutation = mutations[0]
else:
    final_mutation = mutations[0]

# Prepare result
result = {
    "wild_type_sequence": wt_seq,
    "mutation": final_mutation
}

# Save result
with open(output_path, 'w') as f:
    json.dump(result, f, indent=2)

print(f"Result saved to {output_path}")</execute>
