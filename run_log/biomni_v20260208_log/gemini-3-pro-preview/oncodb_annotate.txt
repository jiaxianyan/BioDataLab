# Create directories if they don't exist (R can't easily do `mkdir -p` without system calls, but I did it in bash previously. Just to be safe)
dir.create("/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-pro-preview/", recursive = TRUE, showWarnings = FALSE)

# Load library
library(survival)

# Load data
data <- read.csv("benchmark/dataset/OncoDB/rossi_dataset.csv")

# Perform Cox Regression
# week is duration, arrest is event
# All other columns are covariates
cox_model <- coxph(Surv(week, arrest) ~ ., data = data)

# Summary of the model
model_summary <- summary(cox_model)

# Extract coefficients matrix
# Columns usually: coef, exp(coef), se(coef), z, p
coefs <- model_summary$coefficients

# Extract confidence intervals
# Columns usually: exp(coef), exp(-coef), lower .95, upper .95
conf_int <- model_summary$conf.int

# We need confidence intervals for coef (not exp(coef)) as well.
# CI for coef = coef +/- 1.96 * se
z_score_95 <- qnorm(0.975)
coef_lower <- coefs[, "coef"] - z_score_95 * coefs[, "se(coef)"]
coef_upper <- coefs[, "coef"] + z_score_95 * coefs[, "se(coef)"]

# Prepare the output data frame
results <- data.frame(
  covariate = rownames(coefs),
  coef = coefs[, "coef"],
  `exp(coef)` = coefs[, "exp(coef)"],
  `se(coef)` = coefs[, "se(coef)"],
  `coef lower 95%` = coef_lower,
  `coef upper 95%` = coef_upper,
  `exp(coef) lower 95%` = conf_int[, "lower .95"],
  `exp(coef) upper 95%` = conf_int[, "upper .95"],
  `cmp to` = rownames(coefs), # As per example, just repeating the variable name
  z = coefs[, "z"],
  p = coefs[, "Pr(>|z|)"],
  check.names = FALSE
)

# Calculate -log2(p)
results$ `-log2(p)` <- -log2(results$p)

# Round numeric columns for cleaner output (optional but good for matching example style, though example implies high precision is fine too, it showed 2 decimals. # Write to CSV
