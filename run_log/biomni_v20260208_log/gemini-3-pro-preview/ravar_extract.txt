def parse_line_regex(line, trait):
    # Regex for scientific notation like 5.44×10 −6 or simple float like 0.064
    # The minus sign might be hyphen or unicode minus
    # Pattern: number, optionally followed by x10 and exponent
    
    # We first try to split the line into parts, but the space in scientific notation breaks simple split.
    # Let's clean the line first. 
    # "5.44×10 −6" -> "5.44e-6"
    
    clean_line = line.replace("×10 −", "e-").replace("×10 -", "e-").replace("−", "-")
    # Also sometimes it might be "×10 " followed by positive number? Unlikely for p-values but possible.
    
    parts = clean_line.split()
    
    if len(parts) < 3: return None
    
    # Identify Gene. 
    # If starts with Tissue (Aorta/Tibialartery), gene is second.
    # Else gene is first.
    gene_idx = 0
    p_idx = 1 # Default for gene rows
    
    if "Tibialartery" in parts[0] or "Aorta" in parts[0]:
        gene_idx = 1
        p_idx = 2
    
    if gene_idx >= len(parts) or p_idx >= len(parts): return None
    
    gene = parts[gene_idx]
    if len(gene) < 2 and not any(c.isalpha() for c in gene): return None # Skip junk
    if gene in ["Tissue", "Gene", "p.sbp.dsvm", "p.dbp.dsvm"]: return None

    p_str = parts[p_idx]
    
    try:
        p_val = float(p_str)
        return {"Gene": gene, "Trait": trait, "P-value": p_val}
    except ValueError:
        return None
