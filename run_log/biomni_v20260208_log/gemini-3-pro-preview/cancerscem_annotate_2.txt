import pandas as pd
import numpy as np

# Define file paths
input_file = "benchmark/dataset/CancerSCEM/gene_expression.csv"
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-pro-preview/cancerscem_annotate_2.txt"

print("Loading data...")
# Read the CSV, setting the first column ('Gene') as the index
df = pd.read_csv(input_file, index_col=0)
print(f"Data shape: {df.shape}")

# Transpose the dataframe so genes are columns (variables) and samples are rows (observations)
# We want correlation between genes.
df_t = df.T
print(f"Transposed shape (samples x genes): {df_t.shape}")

print("Calculating Pearson correlation matrix...")
# Calculate correlation matrix
corr_matrix = df_t.corr(method='pearson')

print("Filtering for highly correlated pairs (> 0.9)...")
# Get upper triangle of correlation matrix to avoid duplicates and self-correlation
# np.triu returns upper triangle, k=1 excludes diagonal
upper_tri = corr_matrix.where(np.triu(np.ones(corr_matrix.shape), k=1).astype(bool))

# Stack to get pairs and filter
pairs = upper_tri.stack()
high_corr_pairs = pairs[pairs > 0.9]

print(f"Found {len(high_corr_pairs)} pairs.")

# Sort pairs for consistency (optional but good practice)
# The index of high_corr_pairs is a MultiIndex (Gene_A, Gene_B)

# Save to file
print(f"Saving to {output_file}...")
with open(output_file, 'w') as f:
    for (gene1, gene2), val in high_corr_pairs.items():
        f.write(f"{gene1}\n{gene2}\n")

print("Done.")</execute>
