import os

print(f"Processing {gff_file}...")

data = []

with open(gff_file, 'r') as f:
    for line in f:
        if line.startswith("#"):

        if len(parts) < 9:
            continue
            
        seqid = parts[0]
        feature_type = parts[2]
        start = parts[3]
        end = parts[4]
        strand = parts[6]
        attributes_str = parts[8]
        
        if feature_type != 'CDS':

        # Parse attributes
        attributes = {}
        for attr in attributes_str.split(';'):
            if '=' in attr:
                key, value = attr.split('=', 1)
                attributes[key] = value
        
        locus_tag = attributes.get('locus_tag', '-')
        symbol = attributes.get('gene', '-')
        old_locus_tag = attributes.get('old_locus_tag', '-')
        protein_id = attributes.get('protein_id', '-')
        product = attributes.get('product', '-')
        
        # Format Location
        location = f"{seqid}:{start}-{end}({strand})"
        
        # Pfam (Placeholder as annotation tool/DB is missing)
        pfam = "-"
        
        data.append({
            'Locus_Tag': locus_tag,
            'Symbol': symbol,
            'Old_Locus_Tag': old_locus_tag,
            'Location': location,
            'Protein_ID': protein_id,
            'Product': product,
            'Pfam': pfam
        })

print(f"Extracted {len(data)} CDS entries.")

# Write to TSV
header = ['Locus_Tag', 'Symbol', 'Old_Locus_Tag', 'Location', 'Protein_ID', 'Product', 'Pfam']
