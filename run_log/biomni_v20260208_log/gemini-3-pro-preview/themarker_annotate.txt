<execute> import pandas as pd
import json
import os

# Paths
input_file = 'benchmark/dataset/TheMarker/biomarker.txt'
output_dir = '/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-pro-preview'
intermediate_dir = '/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-pro-preview/themarker_annotate'
output_file = os.path.join(output_dir, 'themarker_annotate.json')

# Ensure directories exist
os.makedirs(output_dir, exist_ok=True)
os.makedirs(intermediate_dir, exist_ok=True)

# Read Data
df = pd.read_csv(input_file, sep='\t')

# Logic for classification
# Based on analysis of the biomarkers:
# B07XCZ: LAMA3. Methylation of LAMA3 is often a Predictive biomarker (e.g. for chemotherapy response).
# B36BFV: IL2 mRNA. IL-2 is a key cytokine. Changes in IL-2 are classic Pharmacodynamic markers of immune modulation.
# B12RGS: Clusterin. Clusterin is a well-established Safety biomarker (Kidney Safety Biomarker).
# B1W4DT: uPAR. Soluble uPAR is used for Monitoring disease severity/prognosis (e.g. in inflammation, cancer).

# Map
# PDY: Pharmacodynamic
# SAF: Safety
# MOI: Monitoring
# PRD: Predictive
# SUR: Surrogate

# Assigning labels
id_to_label = {
    'B07XCZ': 'PRD',  # LAMA3
    'B36BFV': 'PDY',  # IL2
    'B12RGS': 'SAF',  # Clusterin
    'B1W4DT': 'MOI'   # uPAR
}

labels = []
reasoning = []

for index, row in df.iterrows():
    bid = row['Biomarker ID']
    label = id_to_label.get(bid, 'MOI') # Default to MOI if unknown
    labels.append(label)
    
    # Save reasoning
    reasoning.append({
        'Biomarker ID': bid,
        'Name': row['Biomarker Name'],
        'Assigned Label': label,
        'Reasoning': f"Classified based on known biological role. {row['Biomarker Name']} fits {label}."
    })

# Save output
with open(output_file, 'w') as f:
    json.dump(labels, f)

# Save intermediate
intermediate_file = os.path.join(intermediate_dir, 'reasoning.json')
with open(intermediate_file, 'w') as f:
    json.dump(reasoning, f, indent=2)

print(f"Labels saved to {output_file}")
print(f"Reasoning saved to {intermediate_file}")
print("Generated Labels:", labels)</execute>
