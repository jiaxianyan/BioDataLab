import requests
import json
import os

# Define the output path
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-pro-preview/a3d_modb_retrieval.json"
intermediate_dir = "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-pro-preview/a3d_modb_retrieval"

# UniProt API endpoint
base_url = "https://rest.uniprot.org/uniprotkb/search"

# Construct the query
# taxon_id:83333 -> E. coli K12
# reviewed:true -> Swiss-Prot
# name:transmembrane -> Protein name includes "transmembrane"
# Note: "protein_name" is the specific field, but "name" usually works for general text search in names. 
# Let's use specific field logic: (organism_id:83333) AND (reviewed:true) AND (protein_name:transmembrane)

query = '(organism_id:83333) AND (reviewed:true) AND (protein_name:transmembrane)'
params = {
    "query": query,
    "format": "json",
    "fields": "accession",
    "size": 500  # Page size
}

print(f"Querying UniProt with: {query}")

results = []
url = base_url
while url:
    try:
        response = requests.get(url, params=params)
        response.raise_for_status()
        
        data = response.json()
        
        # Extract accessions from the current page
        if 'results' in data:
            for entry in data['results']:
                results.append(entry['primaryAccession'])
        
        # Check for pagination (Link header or next link in response body depending on API version, 
        # but modern UniProt REST API often handles pagination via 'Link' headers for cursor-based pagination)
        # Actually, for the new API, we need to check headers for 'Link' with rel="next"
        
        # The requests library 'link' property parses the Link header
        if 'next' in response.links:
            url = response.links['next']['url']
            params = {} # Params are part of the next link usually
            print(f"Fetching next page... Current count: {len(results)}")
        else:
            url = None
            
    except Exception as e:
        print(f"Error occurred: {e}")
        break

print(f"Total proteins found: {len(results)}")

# Save intermediate raw result just in case
intermediate_file = os.path.join(intermediate_dir, "raw_accessions.json")
with open(intermediate_file, 'w') as f:
    json.dump(results, f, indent=2)

# Save final result
with open(output_file, 'w') as f:
    json.dump(results, f, indent=2)

print(f"Saved results to {output_file}")</execute>
