from Bio import Entrez
import json
import os

# Set email for Entrez
Entrez.email = "assistant@example.com"

# Define the query and parameters
query = '("Curcumin"[Mesh] OR Curcumin[Title/Abstract]) AND ("Drug Therapy, Combination"[Mesh] OR "Drug Synergism"[Mesh] OR combin*[Title/Abstract] OR synerg*[Title/Abstract])'
mindate = "2025/11/01"
maxdate = "2025/11/05"

print(f"Searching PubMed with query: {query}")
print(f"Date range: {mindate} to {maxdate}")

# Perform the search
try:
    handle = Entrez.esearch(db="pubmed", term=query, mindate=mindate, maxdate=maxdate, ptyp="Review", retmax=1000)
    record = Entrez.read(handle)
    handle.close()
    
    pmids = record["IdList"]
    print(f"Found {len(pmids)} articles.")
    print(f"PMIDs: {pmids}")
    
    # Save PMIDs to a temporary file
    tmp_path = "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-pro-preview/npcdr_retrieval/pmids.json"
    with open(tmp_path, 'w') as f:
        json.dump(pmids, f)
        
except Exception as e:
    print(f"Error during search: {e}")</execute>

    full_text = (title + " " + abstract).lower()
    
    # Logic for acceptance:
    # 1. Must contain "curcumin"
    # 2. Must contain combination-related keywords
    
    has_curcumin = "curcumin" in full_text
    
    # Keywords indicating combination/synergy
    combo_keywords = [
        "combin", "synerg", "co-admin", "co-treat", "plus", 
        "sensitiz", "adjuvant", "concomitant", "drug delivery system", 
        "nanoformulation", "encapsul", "load", "nanoparticle"
    ]
    # Note: The prompt asks for "drug combinations or synergistic effects". 
    # Sometimes "combination" in search hits refers to "combinatorial chemistry" or "combined analysis".
    # Also need to be careful with "nano" keywords - sometimes "curcumin-loaded nanoparticles" 
    # are considered a "formulation" rather than a "drug combination" in the strict sense of two active drugs.
    # However, often reviews on curcumin formulations are about improving its delivery *for* therapy, 
    # sometimes *in combination*.
