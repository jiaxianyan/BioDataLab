from Bio import Entrez
import json
import os

# Set email for Entrez (required)
Entrez.email = "assistant@example.com"

# Define the search query
# Combining drug names with biological activity terms in Title/Abstract
drug_terms = '(Zilovertamab vedotin[Title/Abstract] OR "MK-2140"[Title/Abstract] OR "VLS-101"[Title/Abstract])'
bio_terms = '(pharmacology[Title/Abstract] OR "mechanism of action"[Title/Abstract] OR efficacy[Title/Abstract] OR ADC[Title/Abstract] OR "biological activity"[Title/Abstract] OR "antitumor activity"[Title/Abstract])'
query = f'{drug_terms} AND {bio_terms} AND English[Language]'

print(f"Search Query: {query}")

# Search PubMed
try:
    handle = Entrez.esearch(db="pubmed", term=query, retmax=1000)
    record = Entrez.read(handle)
    handle.close()
    
    id_list = record["IdList"]
    print(f"Found {len(id_list)} articles.")

    if not id_list:
        print("No articles found.")
        # Create an empty list if no results
        articles_data = []
    else:
        # Fetch details for the found IDs
        handle = Entrez.efetch(db="pubmed", id=id_list, retmode="xml")
        papers = Entrez.read(handle)
        handle.close()
        
        articles_data = []
        pubmed_articles = papers['PubmedArticle']
        
        for article in pubmed_articles:
            pmid = str(article['MedlineCitation']['PMID'])
            article_data = article['MedlineCitation']['Article']
            title = article_data.get('ArticleTitle', '')
            abstract_list = article_data.get('Abstract', {}).get('AbstractText', [])
            abstract = " ".join(abstract_list) if abstract_list else ""
            
            articles_data.append({
                "PMID": pmid,
                "Title": title,
                "Abstract": abstract
            })
            
    # Save intermediate data
    intermediate_path = "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-pro-preview/adcdb_extract/articles_raw.json"
    with open(intermediate_path, 'w') as f:
        json.dump(articles_data, f, indent=2)
        
    print(f"Saved raw article data to {intermediate_path}")
    print(f"First 3 PMIDs found: {[a['PMID'] for a in articles_data[:3]]}")

except Exception as e:
    print(f"An error occurred: {e}")</execute>

import json
import re

# Load the raw data
intermediate_path = "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-pro-preview/adcdb_extract/articles_raw.json"
with open(intermediate_path, 'r') as f:
    articles = json.load(f)

final_results = []

for article in articles:
    pmid = article['PMID']
    title = article['Title'].lower()
    abstract = article['Abstract'].lower()
    text = title + " " + abstract
    
    # Drug keywords (already filtered by search, but good to re-verify or for scoring)
    drug_keywords = ["zilovertamab vedotin", "mk-2140", "vls-101"]
    
    # Negative indicators for "Original Research"
    review_keywords = [
        "review article", "systematic review", "comprehensive review", 
        "current opinion", "overview of", "discusses the current", 
        "summarizes the", "literature review", "new directions for", # Specific for one of the titles seen
        "advances in", "update on"
    ]
    
    # Positive indicators for "Original Research" investigating activity
    research_keywords = [
        "we investigated", "this study", "participants were enrolled", 
        "phase 1", "phase 2", "phase 3", "xenograft", "mice", "in vivo", "in vitro",
        "pharmacokinetics", "safety and efficacy", "results showed", 
        "methods", "objective response", "dose-escalation", "cohort"
    ]
    
    is_review = False
    for kw in review_keywords:
        if kw in title: # Title is a strong indicator for reviews
            is_review = True
            break
        # Sometimes abstracts say "This review..."
        if "this review" in abstract:
            is_review = True
            break

    # Logic: 
    # 1. Default False.
    # 2. If it has research keywords AND is not a review -> True.
    # 3. Special handling: If abstract is empty, assume False (or check Title closely).
    
    accepted = False
    
    if not is_review:
        # Check for evidence of original research
        score = 0
        for kw in research_keywords:
            if kw in text:
                score += 1
        
        # Threshold: Needs to sound like a study
        if score >= 1:
            accepted = True
            
    # Manual override/check for specific ambiguous cases based on observation
    # PMID 35561299: "New Directions for Mantle Cell Lymphoma in 2022" -> Likely review.
    # Logic above catches "new directions for" in review_keywords.
    
    print(f"PMID: {pmid} | Accepted: {accepted} | Review Detected: {is_review}")
    
    final_results.append({
        "PMID": pmid,
        "Accepted": accepted
    })

# Save final result
output_path = "/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-pro-preview/adcdb_extract_2.json"
with open(output_path, 'w') as f:
    json.dump(final_results, f, indent=2)

print(f"\nSaved final results to {output_path}")</execute>
