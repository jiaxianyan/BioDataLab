# Create a script to fetch detailed SOFT files for each Series
script_content = '''
from Bio import Entrez
import json
import time
import re

Entrez.email = "user@example.com"

gse_accessions = ['GSE164803', 'GSE163076', 'GSE137053', 'GSE164546', 'GSE157293']

def fetch_geo_soft(gse_id):
    """Fetch SOFT format data for a GEO Series"""
    try:
        print(f"\\nFetching SOFT data for {gse_id}...")
        handle = Entrez.efetch(db="gds", id=f"{gse_id}", rettype="summary", retmode="text")
        soft_data = handle.read()
        handle.close()
        return soft_data
    except Exception as e:
        print(f"Error fetching SOFT for {gse_id}: {e}")
        return None

# Also try to get more details via web scraping
def analyze_series(gse_id, soft_data):
    """Analyze if Series provides accessible circRNA expression data"""
    analysis = {
        "gse": gse_id,
        "has_circRNA_focus": False,
        "is_human": False,
        "has_expression_data": False,
        "data_accessible": False,
        "platform_info": "",
        "sample_count": 0,
        "notes": []
    }
    
    if soft_data:
        soft_lower = soft_data.lower()
        
        # Check for circRNA focus
        if "circular rna" in soft_lower or "circrna" in soft_lower or "circ rna" in soft_lower:
            analysis["has_circRNA_focus"] = True
            analysis["notes"].append("Contains circRNA keywords")
        
        # Check for human
        if "homo sapiens" in soft_lower or "human" in soft_lower:
            analysis["is_human"] = True
            analysis["notes"].append("Human organism confirmed")
        
        # Check for expression/profiling data
        if any(term in soft_lower for term in ["expression", "profiling", "sequencing", "microarray", "rna-seq"]):
            analysis["has_expression_data"] = True
            analysis["notes"].append("Expression data indicated")
        
        # Check data accessibility indicators
        if any(term in soft_lower for term in ["supplementary", "matrix", "raw data", "processed data", "series_matrix"]):
            analysis["data_accessible"] = True
            analysis["notes"].append("Data files mentioned")
    
    return analysis

# Analyze all Series
results = {}
for gse in gse_accessions:
    soft_data = fetch_geo_soft(gse)
    analysis = analyze_series(gse, soft_data)
    results[gse] = analysis
    
    print(f"  Analysis for {gse}:")
    print(f"    circRNA focus: {analysis['has_circRNA_focus']}")
    print(f"    Human: {analysis['is_human']}")
