search_query = (
    '("BCR-ABL1"[All Fields] OR "BCR-ABL"[All Fields] OR "BCR/ABL"[All Fields]) '
    'AND ("neoantigen"[All Fields] OR "neo-antigen"[All Fields] OR '

print(search_query)
print("\n" + "="*80 + "\n")

# Search PubMed
try:
    handle = Entrez.esearch(
        db="pubmed",
        term=search_query,
        retmax=10000,  # Maximum number of results
        sort="relevance"
    )
    search_results = Entrez.read(handle)
    handle.close()
    
    pmid_list = search_results["IdList"]

    print(f"✓ PubMed search completed successfully")

    # Save PMIDs to intermediate file

    print(f"  - PMIDs saved to: {pmid_file}")
    
    if len(pmid_list) > 0:
        print(f"\nFirst 10 PMIDs: {pmid_list[:10]}")
    
except Exception as e:
    print(f"✗ Error during PubMed search: {str(e)}")
    pmid_list = []

# Function to analyze if article discusses BCR-ABL1 fusion neoantigens
def analyze_article(pmid, title, abstract):
    """
    Analyze if the article discusses BCR-ABL1 fusion neoantigens.
    Returns True if the article extensively discusses neoantigens derived from BCR-ABL1 fusion.
    """
    text = (title + " " + abstract).lower()
    
    # Check for BCR-ABL terms
    bcr_abl_terms = ['bcr-abl', 'bcr/abl', 'bcr abl']
    has_bcr_abl = any(term in text for term in bcr_abl_terms)
    
    # Check for neoantigen/epitope terms
    neoantigen_terms = ['neoantigen', 'neo-antigen', 'neoepitope', 'neo-epitope']
    has_neoantigen = any(term in text for term in neoantigen_terms)
    
    # Check for fusion-related terms
    fusion_terms = ['fusion', 'junction', 'breakpoint']
    has_fusion = any(term in text for term in fusion_terms)
    
    # Check for T cell/immune response terms (indicating neoantigen research)
    immune_terms = ['t cell', 't-cell', 'tcr', 'cd4', 'cd8', 'epitope', 'immunotherapy', 'immune response']
    has_immune = any(term in text for term in immune_terms)
    
    # Check for specific indicators of neoantigen discussion
    specific_indicators = [
        'mutation' in text and 'antigen' in text,
        'peptide' in text and ('mhc' in text or 'hla' in text),
        'epitope' in text and 'specific' in text,
        'neoantigen' in text,
        'tumor specific antigen' in text
    ]
    has_specific = any(specific_indicators)
    
    # Decision logic
    # Must have BCR-ABL AND (neoantigen terms OR strong immune/epitope focus with fusion)
    if has_bcr_abl:
        if has_neoantigen:
            return True
        elif has_fusion and has_immune and has_specific:
            return True
    
    return False
