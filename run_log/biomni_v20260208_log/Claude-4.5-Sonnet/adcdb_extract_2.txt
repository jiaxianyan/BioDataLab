# Create analysis prompt template
def analyze_abstract(pmid, title, abstract):
    """
    Analyze if an article investigates the biological activity of Zilovertamab vedotin
    """
    prompt = f"""You are a biomedical research expert. Analyze the following article to determine if it investigates the biological activity of Zilovertamab vedotin (also known as MK-2140 or VLS-101).

Biological activity investigation includes:
- Pharmacological studies
- Mechanism of action research
- Efficacy studies
- ADC (antibody-drug conjugate) characterization
- Antitumor activity
- Cytotoxicity studies
- Therapeutic effects
- Clinical trials evaluating the drug's effects

Article Details:
PMID: {pmid}
Title: {title}
Abstract: {abstract}

Question: Does this article investigate the biological activity of Zilovertamab vedotin (MK-2140/VLS-101)?

Answer with ONLY "YES" or "NO" followed by a brief explanation (max 2 sentences).
"""
    
    try:
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": "You are a biomedical research expert analyzing scientific literature."},
                {"role": "user", "content": prompt}
            ],
            temperature=0.1,
            max_tokens=150
        )
        
        result = response.choices[0].message.content.strip()
        return result
    except Exception as e:
        print(f"  Error analyzing PMID {pmid}: {e}")
        return "ERROR"
