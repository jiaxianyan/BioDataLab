def contains_telomeric_repeats(sequence, g_motif, c_motif):
    """
    Check if a sequence contains at least 4 consecutive telomeric repeats.
    Returns True if either G-rich or C-rich motif is found.
    """
    return g_motif in sequence or c_motif in sequence
