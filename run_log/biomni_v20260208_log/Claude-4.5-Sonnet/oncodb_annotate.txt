coef_table <- coef_summary$coefficients
conf_int <- coef_summary$conf.int

# Create a data frame with all required information
results_df <- data.frame(
  covariate = rownames(coef_table),
  coef = coef_table[, "coef"],
  `exp(coef)` = coef_table[, "exp(coef)"],
  `se(coef)` = coef_table[, "se(coef)"],
  `coef lower 95%` = log(conf_int[, "lower .95"]),
  `coef upper 95%` = log(conf_int[, "upper .95"]),
  `exp(coef) lower 95%` = conf_int[, "lower .95"],
  `exp(coef) upper 95%` = conf_int[, "upper .95"],
  `cmp to` = rownames(coef_table),
  z = coef_table[, "z"],
  p = coef_table[, "Pr(>|z|)"],
  check.names = FALSE
)

# Calculate -log2(p)
results_df$`-log2(p)` <- -log2(results_df$p)

# Round to appropriate decimal places
results_df$coef <- round(results_df$coef, 2)
results_df$`exp(coef)` <- round(results_df$`exp(coef)`, 2)
results_df$`se(coef)` <- round(results_df$`se(coef)`, 2)

results_df$z <- round(results_df$z, 2)
