handle.close()

# Fetching the GSE accessions for these IDs
gse_accessions = []
for id in ids:
    handle = Entrez.esummary(db="gds", id=id)
    summary = Entrez.read(handle)
    handle.close()
    accession = summary[0]['Accession']
    if accession.startswith("GSE"):
        gse_accessions.append(accession)

print(f"GSE Accessions: {gse_accessions}")

# Save the list of accessions to a temporary file
with open(os.path.join(tmp_dir, "gse_list.txt"), "w") as f:
    for gse in gse_accessions:
        f.write(gse + "\n")</execute>

#!R
library(GEOquery)

gse_ids <- c('GSE164803', 'GSE163076', 'GSE137053', 'GSE164546', 'GSE157293')
results <- list()

for (gse_id in gse_ids) {
  cat(paste("Processing", gse_id, "...\n"))

  })

    results[[gse_id]] <- FALSE
    next
  }
  
  # Extract metadata from the first platform
  metadata <- Meta(gse[[1]])
  title <- metadata$title
  summary <- metadata$summary
  supp_files <- metadata$supplementary_file
  
  cat(paste("Title:", title, "\n"))
  # cat(paste("Summary:", summary, "\n"))
  
  # Heuristic for circRNA expression data:
  # 1. Mentions "circRNA" or "circular RNA" in title or summary.
  # 2. Check if it's from human (already filtered by search, but good to verify).
  # 3. Check for expression data (count matrices, processed files).
  
  text_to_check <- tolower(paste(title, summary, collapse = " "))
  has_circ_keyword <- grepl("circrna|circular rna|circ-", text_to_check)
  
  # Check supplementary files for potential expression matrices
  # Often circRNA data is in supplementary files as .txt, .xlsx, .csv, .gz
  has_supp_files <- !is.null(supp_files) && length(supp_files) > 0
  
  # Determine acceptance
  # Must be human (Homo sapiens) and have circRNA expression data.
  accepted <- has_circ_keyword && has_supp_files
  
  # More specific check for expression data
  is_expression <- grepl("expression|profiling|rna-seq|sequencing|microarray|count|matrix", text_to_check)
  
  final_decision <- accepted && is_expression
  
  results[[gse_id]] <- final_decision
  cat(paste("Accepted:", final_decision, "\n\n"))

  list(Series = id, Accepted = results[[id]])

#!R
library(GEOquery)
library(jsonlite)

gse_ids <- c('GSE164803', 'GSE163076', 'GSE137053', 'GSE164546', 'GSE157293')
results_list <- list()

tmp_dir <- "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-flash-preview/circmine_retrieval/"

for (gse_id in gse_ids) {
  cat(paste("Processing", gse_id, "...\n"))
  
  # Get GEO object - getGPL=FALSE to speed up and minimize memory
  gse_list <- tryCatch({
    getGEO(gse_id, destdir = tmp_dir, getGPL = FALSE)
  }, error = function(e) {
    cat(paste("Error downloading", gse_id, ":", e$message, "\n"))
    return(NULL)
  })
  
  if (is.null(gse_list) || length(gse_list) == 0) {
    results_list[[length(results_list) + 1]] <- list(Series = gse_id, Accepted = FALSE)
    next
  }
  
  # For series info, we often look at the first ExpressionSet
  eset <- gse_list[[1]]
  
  # Metadata for the series is stored in the experimentData or at the list level
  # In GEOquery, getGEO returns a list of ExpressionSets
  # The series-level metadata is available via the 'annotation' or metadata slots
  
  title <- eset@experimentData@title
  summary <- eset@experimentData@abstract
  
  # Access supplementary file info from the platform/sample level if needed, 
  # but GEO series usually have a 'supplementary_file' field in the metadata
  # Let's try getting the series metadata directly using getGEO with different parameters if needed
  # or use the internal metadata.
  
  # Better way to get series metadata:
  meta <- eset@experimentData@other
  
  # Let's also check the characteristics of samples to ensure human
  sample_meta <- pData(eset)
  organism <- unique(sample_meta$organism_ch1)
  
  cat(paste("Title:", title, "\n"))
  cat(paste("Organism:", paste(organism, collapse=", "), "\n"))
  
  text_to_check <- paste(title, summary, paste(unlist(meta), collapse=" "), sep=" ")
  text_to_check <- tolower(text_to_check)
  
  has_circ <- grepl("circrna|circular rna|circ-", text_to_check)
  
  # Check for keywords indicating expression data availability
  has_data <- grepl("expression|count|matrix|profiling|sequencing|rna-seq|microarray", text_to_check)
  
  # Verify human source
  is_human <- any(grepl("Homo sapiens", organism, ignore.case = TRUE))
  
  # Final check: Does it have supplementary files (typical for GEO expression data)
  # We can check the 'supplementary_file' in the metadata
  # Usually in eset@experimentData@other$supplementary_file
  supp_files <- eset@experimentData@other$supplementary_file
  has_supp <- !is.null(supp_files) && any(nchar(supp_files) > 0)
  
  cat(paste("Keywords found:", has_circ, "Is Human:", is_human, "Has Data:", has_data, "Has Supp:", has_supp, "\n"))
  
  accepted <- has_circ && is_human && has_data && has_supp
  
  results_list[[length(results_list) + 1]] <- list(Series = gse_id, Accepted = as.logical(accepted))
  cat(paste("Final Decision for", gse_id, ":", accepted, "\n\n"))
}

# Save results
write_json(results_list, "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-flash-preview/circmine_retrieval/analysis_results.json", auto_unbox = TRUE, pretty = TRUE)</execute>
