import pandas as pd
import numpy as np

# Load the raw metadata
df = pd.read_csv("/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-flash-preview/circmine_extract/metadata_raw.csv", index_index=0)

# Define the columns to extract based on the R output
# Case ID is in 'case:ch1' or 'characteristics_ch1'
# Age is in 'age:ch1'
# Gender is in 'gender:ch1'
# Disease state is in 'disease state:ch1'
# Stage is in 'Stage:ch1'
# Country is in 'contact_country'

def clean_val(val):
    if pd.isna(val) or str(val).lower() == 'nan' or str(val).strip() == '':
        return "N/A"
    return str(val).strip()

# Create a list to store patient-level data
patient_data = []

# Group by 'case:ch1' to consolidate samples from the same patient
# Note: Sample names like "case2 lung adenocarcinoma tissue" and "case2 adjacent normal lung tissue" 
# confirm that 'case:ch1' is the patient identifier.

for patient_id, group in df.groupby('case:ch1'):
    # Extract values, taking the first non-NA entry for each field
    age_raw = group['age:ch1'].iloc[0]
    # Age might be "66 years", need numeric
    age = "".join(filter(str.isdigit, str(age_raw))) if pd.notna(age_raw) else "N/A"
    
    gender = clean_val(group['gender:ch1'].iloc[0])
    disease_state = clean_val(group['disease state:ch1'].iloc[0])
    stage = clean_val(group['Stage:ch1'].iloc[0])
    country = clean_val(group['contact_country'].iloc[0])
    
    patient_data.append({
        'series': 'GSE101684',
        'age': age,
        'gender': gender,
        'disease state': disease_state,
        'stage': stage,
        'country': country
    })

# Convert to DataFrame
final_df = pd.DataFrame(patient_data)

# Ensure age is numeric or N/A
final_df['age'] = pd.to_numeric(final_df['age'], errors='coerce').fillna("N/A")

# Save to CSV
