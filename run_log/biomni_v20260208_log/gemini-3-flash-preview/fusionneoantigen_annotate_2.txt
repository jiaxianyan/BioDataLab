import os
import subprocess

# Define paths
tmp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-flash-preview/fusionneoantigen_annotate_2"
protein_pdb = "benchmark/dataset/FusionNeoAntigen/dock/1a1e_protein.pdb"
smiles_file = "benchmark/dataset/FusionNeoAntigen/dock/smiles.txt"
pocket_file = "benchmark/dataset/FusionNeoAntigen/dock/pocket_config.txt"

# 1. Parse pocket config
config = {}
with open(pocket_file, 'r') as f:
    for line in f:
        if '=' in line:
            key, val = line.split('=')
            config[key.strip()] = val.strip()

print(f"Docking Config: {config}")

# 2. Prepare protein (using obabel if available, or just check for ADFR suite/mgltools)
# We'll try to use 'mk_prepare_receptor.py' or 'obabel'
protein_pdbqt = os.path.join(tmp_dir, "protein.pdbqt")

# Prepare protein with obabel (adds hydrogens and converts)
# Note: Real docking usually requires more careful prep, but obabel is a common tool.
print("Preparing protein...")
subprocess.run(["obabel", "-ipdb", protein_pdb, "-opdbqt", "-O", protein_pdbqt, "-xr"], check=True)

# 3. Process SMILES and prepare ligands
with open(smiles_file, 'r') as f:
    smiles_list = [line.strip() for line in f if line.strip()]

ligand_pdbqts = []
for i, smiles in enumerate(smiles_list):
    out_pdbqt = os.path.join(tmp_dir, f"ligand_{i}.pdbqt")
    print(f"Preparing ligand {i}: {smiles}")
    # Use obabel to convert SMILES to 3D PDBQT
    # --gen3d generates 3D coordinates
    # -p 7.4 adds hydrogens at pH 7.4 (optional)
    subprocess.run(["obabel", f"-:{smiles}", "--gen3d", "-opdbqt", "-O", out_pdbqt, "-p", "7.4"], check=True)
    ligand_pdbqts.append((smiles, out_pdbqt))

print(f"Prepared {len(ligand_pdbqts)} ligands.")

# 4. Docking with Vina
scores = []
for smiles, lig_pdbqt in ligand_pdbqts:
    out_log = lig_pdbqt.replace(".pdbqt", "_log.txt")
    out_dock = lig_pdbqt.replace(".pdbqt", "_dock.pdbqt")
    
    cmd = [
        "vina",
        "--receptor", protein_pdbqt,
        "--ligand", lig_pdbqt,
        "--center_x", config['center_x'],
        "--center_y", config['center_y'],
        "--center_z", config['center_z'],
        "--size_x", config['size_x'],
        "--size_y", config['size_y'],
        "--size_z", config['size_z'],
        "--exhaustiveness", config.get('exhaustiveness', '8'),
        "--num_modes", config.get('num_modes', '9'),
        "--out", out_dock,
        "--log", out_log
