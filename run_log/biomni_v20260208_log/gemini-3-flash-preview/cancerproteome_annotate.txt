import pandas as pd
from scipy.stats import spearmanr

# Define file paths
protein_file = 'benchmark/dataset/CancerProteome/protein_abundance.csv'
transcript_file = 'benchmark/dataset/CancerProteome/transcript_levels.csv'
output_dir = '/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-flash-preview/cancerproteome_annotate'
final_output = '/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-flash-preview/cancerproteome_annotate.txt'

# Load the data
protein_df = pd.read_csv(protein_file)
transcript_df = pd.read_csv(transcript_file)

print("Protein abundance head:")
print(protein_df.head())
print("Transcript level head:")
print(transcript_df.head())

# Merge datasets
merged_df = pd.merge(protein_df, transcript_df, on='GeneID')

print("Merged head:")
print(merged_df.head())

# Save merged data as an intermediate result
merged_df.to_csv(f"{output_dir}/merged_data.csv", index=False)

# Calculate Spearman correlation

print(f"Spearman Correlation: {correlation}")

with open(final_output, 'w') as f:
    f.write(f"{correlation}\n")</execute>
