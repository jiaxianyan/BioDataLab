def clean_p(p_str):
    if not p_str: return 1.0
    # Handle scientific notation like '5.44×10 −6'
    p_str = p_str.replace('×10', 'E').replace('−', '-').replace(' ', '').replace('−', '-')
    try:
        return float(p_str)
    except:
        return 1.0

results = []

# Traits identified: SBP (systolic blood pressure) and DBP (diastolic blood pressure)
# Thresholds: gene-level (SKAT) P < 1E-4; SNP-level P < 1E-6

        results.append({'EFO Trait Label': 'diastolic blood pressure', 'Accepted': accepted})

    if tables3:
        for row in tables3[0][2:]: # Skip multi-row header

            # DBP
