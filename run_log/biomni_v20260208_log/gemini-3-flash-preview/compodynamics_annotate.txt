import sys
from Bio import SeqIO

def calculate_gc(sequence):
    g = sequence.count('G') + sequence.count('g')
    c = sequence.count('C') + sequence.count('c')
    return (g + c) / len(sequence)

fasta_file = 'benchmark/dataset/CompoDynamics/GCF_009428885.1_ASM942888v1_cds_from_genomic.fna'
target_accession = "NZ_WHYS01000010.1"

total_gc_weighted = 0
total_length = 0

# Iterate through sequences in the CDS FASTA file
# The header format is: >lcl|NZ_WHYS01000010.1_cds_WP_152943596.1_1 ...
for record in SeqIO.parse(fasta_file, "fasta"):
    if target_accession in record.id:
        seq_len = len(record.seq)
        gc_content = calculate_gc(str(record.seq))
        
        total_gc_weighted += gc_content * seq_len
        total_length += seq_len

if total_length > 0:
    weighted_average_gc = total_gc_weighted / total_length
    print(f"Weighted Average GC: {weighted_average_gc}")
