import requests
import pandas as pd
import json

def get_all_studies(query):
    # Use the specific search endpoint for the GWAS Catalog
    base_url = "https://www.ebi.ac.uk/gwas/rest/api/studies/search/findByText"
    params = {'text': query, 'size': 100}
    headers = {'Accept': 'application/json'}
    
    all_results = []
    
    response = requests.get(base_url, params=params, headers=headers)
    if response.status_code != 200:
        print(f"Error fetching data: {response.status_code}")
        return []
    
    data = response.json()
    
    def extract_2025_studies(data_obj):
        count = 0
        if '_embedded' in data_obj:
            studies = data_obj['_embedded']['studies']
            for study in studies:
                pub_date = study.get('publicationDate', '')
                # Filter for year 2025
                if pub_date and pub_date.startswith('2025'):
                    pmid = study.get('pubmedId', 'N/A')
                    accession = study.get('accessionId', 'N/A')
                    all_results.append({'PMID': pmid, 'Accession': accession})
                    count += 1
        return count

    extract_2025_studies(data)
    
    # Check for more pages
    page_count = 1
    while 'next' in data.get('_links', {}):
        next_url = data['_links']['next']['href']
        response = requests.get(next_url, headers=headers)
        if response.status_code != 200:
            break
        data = response.json()
        extract_2025_studies(data)
        page_count += 1
        if page_count > 50: # Safety break
            break
            
    return all_results

# Define variables

TMP_JSON="/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-flash-preview/pharmgwas_extract/data.json"

# Fetch data using curl (first page)
curl -s -G "https://www.ebi.ac.uk/gwas/rest/api/studies/search/findByText" \
    --data-urlencode "text=Atrial Fibrillation" \
    --data-urlencode "size=500" \
    -H "Accept: application/json" > "$TMP_JSON"

# Check if the file was created and contains data
if [ ! -s "$TMP_JSON" ]; then
    echo "Error: Failed to fetch data from API."
    exit 1
fi

# Extract PMID and Accession for 2025 publications using jq
# Output headers first
echo "PMID,Accession" > "$OUTPUT_PATH"

# Parse JSON and filter
# publicationDate is usually in format "2025-01-01"
# We extract pubmedId and accessionId
cat "$TMP_JSON" | jq -r '._embedded.studies[] | select(.publicationDate != null and (.publicationDate | startswith("2025"))) | [.pubmedId, .accessionId] | @csv' | tr -d '"' >> "$OUTPUT_PATH"
