import pandas as pd
import os
import re

tmp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-flash-preview/dda_extract/"
df1 = pd.read_csv(os.path.join(tmp_dir, "gse116315_pheno_1.csv"))
df2 = pd.read_csv(os.path.join(tmp_dir, "gse116315_pheno_2.csv"))

# Merge the dataframes
df = pd.concat([df1, df2], ignore_index=True)

results = []

for _, row in df.iterrows():
    # 1. Sample Name: GSM accession and title
    gsm = str(row['geo_accession'])
    title = str(row['title'])
    sample_name = f"{gsm}-{title}"
    
    # 2. Source Project ID
    project_id = "GSE116315"
    
    # 3. Cell Line/Tissue
    # Try 'cell type:ch1', then 'source_name_ch1'
    cell_line = row.get('cell type:ch1', row.get('source_name_ch1', "N/A"))
    if pd.isna(cell_line): cell_line = "N/A"
    
    # 4. Organism
    organism = row.get('organism_ch1', "N/A")
    if pd.isna(organism): organism = "N/A"
    
    # 5. Treatment
    # The title contains treatment info like "HU_28h_10mM" or "NT"
    # HU is Hydroxyurea. NT is No Treatment.
    treatment = "N/A"
    title_upper = title.upper()
    if "HU" in title_upper:
        treatment = "Hydroxyurea"
    elif "NT" in title_upper:
        treatment = "None"
    
    # 6. Genotype
    genotype = row.get('genotype:ch1', "N/A")
    if pd.isna(genotype): genotype = "N/A"
    
    # 7. Technique
    # library_strategy is 'OTHER' in some records, but data_processing says 'END-seq'
    # Checking library_strategy and data_processing
    tech = "END-seq" # Based on the study description and data_processing fields
    
    results.append({
        "Sample Name": sample_name,
        "Source Project ID": project_id,
        "Cell Line/Tissue": cell_line,
        "Organism": organism,
        "Treatment": treatment,
        "Genotype": genotype,
        "Technique": tech
    })

output_df = pd.DataFrame(results)
print(output_df.head(20))

# Save the result
save_path = "/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-flash-preview/dda_extract.csv"
output_df.to_csv(save_path, index=False)
