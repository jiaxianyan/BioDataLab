### Step 3, 4, and 5: Load, Assemble, and Save in R

The previews show that `barcodes.tsv` and `features.tsv` contain single columns of names without headers. I will use the `Matrix` package in R to read the MatrixMarket file and then set the row and column names.

<execute>
#!R
library(Matrix)

# Paths
mtx_file <- "benchmark/dataset/DISCO/refinement/matrix.mtx"
features_file <- "benchmark/dataset/DISCO/refinement/features.tsv"
barcodes_file <- "benchmark/dataset/DISCO/refinement/barcodes.tsv"
output_file <- "/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-flash-preview/disco_refinement.rds"

# Load data
cat("Loading matrix...\n")
mat <- readMM(mtx_file)

cat("Loading features and barcodes...\n")
features <- read.table(features_file, header = FALSE, stringsAsFactors = FALSE)$V1
barcodes <- read.table(barcodes_file, header = FALSE, stringsAsFactors = FALSE)$V1

# Check dimensions
cat(sprintf("Matrix dimensions: %d x %d\n", nrow(mat), ncol(mat)))
cat(sprintf("Features count: %d\n", length(features)))
cat(sprintf("Barcodes count: %d\n", length(barcodes)))

# Assign names
# Usually, rows are features and columns are cells in these formats
if (length(features) == nrow(mat) && length(barcodes) == ncol(mat)) {
    rownames(mat) <- features
    colnames(mat) <- barcodes
} else {
    stop("Dimensions mismatch between matrix and metadata files.")
}

# Save as RDS
cat("Saving RDS file...\n")
saveRDS(mat, file = output_file)
cat("Done.\n")</execute>
