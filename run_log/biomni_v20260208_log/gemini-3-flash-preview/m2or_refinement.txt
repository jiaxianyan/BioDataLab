from rdkit import Chem

# Define the file paths
mol2_file_path = 'benchmark/dataset/M2OR/1a1e_ligand.mol2'
output_file_path = '/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-flash-preview/m2or_refinement.txt'

try:
    # Read the molecule from the MOL2 file
    # Note: RDKit's MolFromMol2Block or MolFromMol2File might be picky. 
    # MolFromMol2File is available in some versions, but sometimes we use Chem.MolFromMol2File
    mol = Chem.MolFromMol2File(mol2_file_path)
    
    if mol is None:
        print(f"Failed to read molecule from {mol2_file_path}")
    else:
        # Generate InChIKey
        inchikey = Chem.MolToInchiKey(mol)
        print(f"Generated InChIKey: {inchikey}")
        
        # Save to the output file
        with open(output_file_path, 'w') as f:
            f.write(inchikey)
        print(f"InChIKey saved to {output_file_path}")

except Exception as e:
    print(f"An error occurred: {e}")</execute>
