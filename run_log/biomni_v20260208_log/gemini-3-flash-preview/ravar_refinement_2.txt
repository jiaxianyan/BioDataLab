import pandas as pd
import json
import os

# Create the output directory if it doesn't exist
output_dir = "/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-flash-preview"
os.makedirs(output_dir, exist_ok=True)
tmp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/gemini-3-flash-preview/ravar_refinement_2"
os.makedirs(tmp_dir, exist_ok=True)

# Path to the data file
file_path = 'benchmark/dataset/RAVAR/trails_data.csv'

# Read the CSV file (using tab separator based on the 'head' output)
df = pd.read_csv(file_path, sep='\t')

# Display the traits for categorization
print("Traits to categorize:")
print(df)

# Logic for categorization:
# HP terms (Human Phenotype Ontology) are generally phenotypes/abnormalities -> material property
# Measurements (EFO:0800068 - acisoga measurement) -> information entity
# Diseases (EFO:1001289 - Cholecystitis, Acute) are usually classified as properties/dispositions in BFO-based ontologies -> material property

results = []
for index, row in df.iterrows():
    trait_label = row['Trait Label']
    ontology_id = row['Trait Ontology ID']
    
    if "measurement" in trait_label.lower():
        category = "information entity"
    elif "abnormality" in trait_label.lower():
        category = "material property"
    elif "cholecystitis" in trait_label.lower():
        category = "material property"
    else:
        # Defaulting based on common ontology mapping
        category = "material property"
    
    results.append(category)
    print(f"Trait: {trait_label} ({ontology_id}) -> Category: {category}")

# Save the classification result as a JSON list
output_file = os.path.join(output_dir, "ravar_refinement_2.json")
with open(output_file, 'w') as f:
    json.dump(results, f)

print(f"\nSaved results to {output_file}")
print(f"Content: {results}")</execute>
