import pandas as pd

# Load the data
asm_sites = pd.read_csv('benchmark/dataset/ASMdb/mock/mock_asm_sites.csv')
genes = pd.read_csv('benchmark/dataset/ASMdb/mock/mock_genes.csv')

# Define gene regions including 3kb upstream
# For + strand: start = start - 3000, end = end
# For - strand: start = start, end = end + 3000
# Note: biological coordinates often start at 1 or 0. Assuming coordinates in file are inclusive or 0-based.
# The problem says 3kb upstream.

def get_extended_range(row):
    if row['strand'] == '+':
        return max(0, row['start'] - 3000), row['end']
    else:
        return row['start'], row['end'] + 3000

genes[['ext_start', 'ext_end']] = genes.apply(lambda r: pd.Series(get_extended_range(r)), axis=1)

# Group ASM sites by chromosome for faster lookup
asm_by_chrom = {chrom: group for chrom, group in asm_sites.groupby('chrom')}

gene_sample_counts = {}

for idx, gene in genes.iterrows():
    chrom = gene['chrom']
    start = gene['ext_start']
    end = gene['ext_end']
    gene_id = gene['gene_id']
    
    if chrom in asm_by_chrom:
        chrom_asm = asm_by_chrom[chrom]
        # Find ASM sites within the gene region
        mask = (chrom_asm['pos'] >= start) & (chrom_asm['pos'] <= end)
        matching_asm = chrom_asm[mask]
        
        # Count unique samples for this gene
        if not matching_asm.empty:
            unique_samples = matching_asm['sample_id'].nunique()
            gene_sample_counts[gene_id] = unique_samples
        else:
            gene_sample_counts[gene_id] = 0
    else:
        gene_sample_counts[gene_id] = 0

# Convert results to DataFrame
results_df = pd.DataFrame(list(gene_sample_counts.items()), columns=['gene_id', 'frequency'])

# Sort by frequency descending, then by gene_id (to ensure stability if needed)
results_df = results_df.sort_values(by=['frequency', 'gene_id'], ascending=[False, True])

# Take top 100
top_100 = results_df.head(100)

# Save to the specified output file
output_path = '/root/autodl-tmp/biomni_v20260208_pred_results/gemini-3-flash-preview/asmdb_annotate.csv'
top_100.to_csv(output_path, index=False)

print(f"Top 5 genes for verification:\n{top_100.head()}")
print(f"Results saved to {output_path}")</execute>
