# Categories:
# 1 = enhancing drug efficacy by augmenting sensitivity and achieving synergy
# 2 = decreasing adverse drug reactions
# 3 = reversing drug resistance

# We will use keyword-based heuristics on the abstract/full text.

def classify_text(txt: str) -> int:
    t = txt.lower()
    # Signals for toxicity/adverse reactions -> Type 2
    tox_keywords = [
        "reduce toxicity", "reduced toxicity", "attenuate toxicity", "decrease toxicity",
        "less toxic", "nephrotoxicity", "cardiotoxicity", "hepatotoxicity",
        "bone marrow suppression", "myelosuppression", "side effect", "adverse effect",
        "gastrointestinal toxicity", "neurotoxicity", "ototoxicity", "protective effect",
        "organ protection", "protect against toxicity"
    ]
    # Signals for resistance/reversal -> Type 3
    resist_keywords = [
        "drug resistance", "chemoresistance", "multidrug resistance", "mdr",
        "reverse resistance", "reversal of resistance", "sensitize resistant",
        "overcome resistance", "re-sensitize", "resensitize", "resensitise"
    ]
    # Signals for synergy/sensitization (non-resistance context) -> Type 1
    synergy_keywords = [
        "synergistic", "synergy", "additive effect", "enhance efficacy",
        "enhanced efficacy", "enhanced antitumor", "chemosensitizer",
        "chemo-sensitizer", "radio-sensitizer", "radiosensitizer",
        "increase sensitivity", "sensitize to", "sensitize cancer cells",
        "combination therapy improved", "potentiate", "potentiated the effect"
    ]

    # Count matches
    tox_score = sum(1 for k in tox_keywords if k in t)
    resist_score = sum(1 for k in resist_keywords if k in t)
    synergy_score = sum(1 for k in synergy_keywords if k in t)

    # Simple priority rules:
    # If strong resistance context -> 3
    if resist_score > 0 and resist_score >= tox_score and resist_score >= synergy_score:
        return 3
    # If strong toxicity/ADR context -> 2
    if tox_score > 0 and tox_score >= resist_score and tox_score >= synergy_score:
        return 2
    # Otherwise default to synergy/efficacy if any signal -> 1
    if synergy_score > 0:
        return 1
    # Fallback: if nothing clear, default to 1 (most common for combo studies)
    return 1
