import os, json, datetime, textwrap, math, sys
import requests

tmp_dir = "/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.1/covpdb_retrieval"

# Load PDB IDs from prior step
step3_file = os.path.join(tmp_dir, "step3_pdb_ids_by_date_resolution.json")
with open(step3_file) as f:
    step3_data = json.load(f)
pdb_ids = step3_data["pdb_ids"]

print(f"Loaded {len(pdb_ids)} PDB IDs from step 3.")

# RCSB Structure API for neighbors/contacts or full structure; we will use mmCIF files via RCSB REST
# We'll inspect bonds and coordinates to detect covalent protein-ligand bonds within ≤ 2.0 Å.
# To avoid heavy parsing, we will primarily rely on RCSB polymer/nonpolymer bonding data if available.

def fetch_assembly_info(pdb_id):
    """Fetch assembly and bond info using RCSB entry/assembly APIs."""
    url = f"https://data.rcsb.org/rest/v1/core/entry/{pdb_id}"
    r = requests.get(url, timeout=60)
    r.raise_for_status()
    return r.json()

def fetch_structure(pdb_id):
    """Fetch simplified structure representation from structure API (bonds, entities)."""
    url = f"https://data.rcsb.org/rest/v1/core/assembly/{pdb_id}/1"
    r = requests.get(url, timeout=60)
    if r.status_code != 200:
        # assembly 1 may not exist; fall back to entry-level data or other assembly
        return None
    return r.json()

# Helper to detect protein vs ligand from RCSB entity types:
PROTEIN_TYPES = {"Protein", "Peptide", "Polypeptide(L)", "polypeptide(L)", "polymer"}
LIGAND_TYPES = {"Non-polymer", "non-polymer", "Bound", "ligand"}

covalent_candidates = []

for idx, pdb_id in enumerate(pdb_ids, 1):
    print(f"\n[{idx}/{len(pdb_ids)}] Processing {pdb_id} ...")
    entry_url = f"https://data.rcsb.org/rest/v1/core/entry/{pdb_id}"
    try:
        er = requests.get(entry_url, timeout=60)
        er.raise_for_status()
        entry = er.json()
    except Exception as e:
        print(f"  Failed to fetch entry for {pdb_id}: {e}")
        continue

    # Retrieve entities (polymer vs non-polymer) from entry
    entities = entry.get("rcsb_entry_container_identifiers", {})
    polymer_entity_ids = set(entities.get("polymer_entity_ids", []))
    nonpolymer_entity_ids = set(entities.get("nonpolymer_entity_ids", []))

    if not polymer_entity_ids or not nonpolymer_entity_ids:
        print("  No polymer or non-polymer entities; skipping as no protein-ligand complex.")
        continue

    # Now query interatomic bonds (using neighbors API)
    # We'll use residue-level neighbors within 2.0 Å between polymer and non-polymer
    neighbors_url = "https://data.rcsb.org/rest/v1/core/entry/neighbor_interactions"
    # This API is not standard; instead, we will use the "interactions" service.
    # Use RCSB GraphQL interactions: but simpler is to use "interaction" reports.
    interaction_url = f"https://data.rcsb.org/rest/v1/core/entry/{pdb_id}/interactions"
    try:
        ir = requests.get(interaction_url, timeout=60)
        if ir.status_code != 200:
            print(f"  No interaction data for {pdb_id}; skipping.")
            continue
        inter = ir.json()
    except Exception as e:
        print(f"  Failed to fetch interactions for {pdb_id}: {e}")
        continue

    # Structure of interaction data can vary; we search for covalent bond annotations
    # Most RCSB interaction reports label 'covalent' interactions.
    found_covalent = False
    ligands_in_covalent = set()

    def is_polymer_partner(partner):
        et = partner.get("entity_type", "").lower()
        return "polymer" in et or "protein" in et

    def is_ligand_partner(partner):
        et = partner.get("entity_type", "").lower()
        return "non-polymer" in et or "ligand" in et

    # Inspect interaction data
    for grp in inter.get("interaction_groups", []):
        itype = grp.get("type", "").lower()
        if "covalent" not in itype:
            continue
        # For each interaction in this group, inspect partners
        for it in grp.get("interactions", []):
            partners = it.get("partners", [])
            if len(partners) != 2:
                continue
            p1, p2 = partners
            if (is_polymer_partner(p1) and is_ligand_partner(p2)) or (is_polymer_partner(p2) and is_ligand_partner(p1)):
                # Check distance if reported
                dist = it.get("distance")
                if dist is not None and dist > 2.0:
                    continue
                # Extract ligand identifier (chem_comp id)
                lig_comp_ids = []
                for p in partners:
                    if is_ligand_partner(p):
                        lig_comp_ids.append(p.get("chem_comp_id"))
                for l in lig_comp_ids:
                    if l:
                        ligands_in_covalent.add(l)
                found_covalent = True

    if found_covalent and ligands_in_covalent:
        print(f"  Found covalent protein–ligand interactions; ligands: {sorted(ligands_in_covalent)}")
        covalent_candidates.append(
            {
                "pdb_id": pdb_id,
                "ligand_ids": sorted(ligands_in_covalent),
            }
        )
    else:
        print("  No covalent protein–ligand interactions found within criteria.")

print(f"\nTotal covalent protein–ligand candidate entries: {len(covalent_candidates)}")

step4_file = os.path.join(tmp_dir, "step4_covalent_candidates_raw.json")
with open(step4_file, "w") as f:
    json.dump(
        {
            "covalent_candidates": covalent_candidates,
            "timestamp": datetime.datetime.utcnow().isoformat(),
        },
        f,
        indent=2,
    )
print(f"Saved raw covalent candidate list to: {step4_file}")

import os, json, datetime, textwrap

tmp_dir = "/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.1/covpdb_retrieval"
biolip_path_root = "benchmark/dataset/CovPDB/ligand_list"

# Collect artifact ligand IDs from all files in the ligand_list directory
artifact_ligands = set()
for root, dirs, files in os.walk(biolip_path_root):
    for fn in files:
        full = os.path.join(root, fn)
        with open(full) as f:
            for line in f:
                line = line.strip()
                if not line or line.startswith("#"):
                    continue
                # Assume ligand IDs are whitespace or comma separated; normalize to 3-letter IDs uppercased
                parts = [p.strip() for p in line.replace(",", " ").split()]
                for p in parts:
                    if 1 <= len(p) <= 4:  # typical ligand ID length
                        artifact_ligands.add(p.upper())

print(f"Total unique artifact ligand IDs loaded from BioLiP files: {len(artifact_ligands)}")
print("Sample artifact ligands:", sorted(list(artifact_ligands))[:20])

artifact_log = os.path.join(tmp_dir, "step5_artifact_ligands.json")
with open(artifact_log, "w") as f:
    json.dump(
        {
            "artifact_ligands": sorted(list(artifact_ligands)),
            "timestamp": datetime.datetime.utcnow().isoformat(),
        },
        f,
        indent=2,
    )
print(f"Saved artifact ligand list to: {artifact_log}")

# Load covalent candidates
step4_file = os.path.join(tmp_dir, "step4_covalent_candidates_raw.json")
with open(step4_file) as f:
    step4_data = json.load(f)
covalent_candidates = step4_data["covalent_candidates"]

filtered_entries = []
for entry in covalent_candidates:
    pdb_id = entry["pdb_id"]
    ligs = entry["ligand_ids"]
    non_artifact_ligs = [l for l in ligs if l.upper() not in artifact_ligands]
    if non_artifact_ligs:
        filtered_entries.append(
            {
                "pdb_id": pdb_id,
                "ligand_ids": sorted(set(non_artifact_ligs)),
            }
        )

print(f"Number of entries after removing artifact ligands: {len(filtered_entries)}")

step5_file = os.path.join(tmp_dir, "step5_covalent_candidates_filtered.json")
with open(step5_file, "w") as f:
    json.dump(
        {
            "filtered_entries": filtered_entries,
            "timestamp": datetime.datetime.utcnow().isoformat(),
        },
        f,
        indent=2,
    )
print(f"Saved filtered covalent candidate list to: {step5_file}")
