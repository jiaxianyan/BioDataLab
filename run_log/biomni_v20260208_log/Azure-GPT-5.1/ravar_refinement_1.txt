# Abbreviation / wording normalization dictionary.
# This is generic and may be refined if needed.
NORMALIZATION_MAP = {
    r"\bU\.?S\.?A\.?\b": "USA",
    r"\bU\.?K\.?\b": "UK",
    r"\bSNPs?\b": "single nucleotide polymorphism",
    r"\bHR\b": "hazard ratio",

    r"\bCI\b": "confidence interval",
    r"\bBMI\b": "body mass index",
    r"\bBP\b": "blood pressure",
    r"\bHbA1c\b": "hemoglobin A1c",
    r"\bGWAS\b": "genome-wide association study",
    r"\bP\s*value\b": "p-value",
    r"\bp value\b": "p-value",
}

def normalize_language(text: str) -> str:
    """Apply regex-based normalization for common abbreviations and phrases."""
    normalized = text
    for pattern, replacement in NORMALIZATION_MAP.items():
        normalized = re.sub(pattern, replacement, normalized, flags=re.IGNORECASE)
    return normalized

def standardize_trait(trait: str) -> str:
    """Standardize a trait string: strip noise, normalize language."""
    if trait is None:
        return ""
    s = str(trait)

    # 1) Remove enclosing quotes or unmatched quotation marks commonly used in text
    s = s.strip()

    # 2) Remove leading/trailing punctuation like ':', ';', ',', '.', '*'
    s = re.sub(r'^[\s\-\:\;\,\.\*\[\]\(\)]+', '', s)
    s = re.sub(r'[\s\-\:\;\,\.\*\[\]\(\)]+$', '', s)

    # 3) Remove publication-specific artifacts: figure/table refs, supplementary refs
    # e.g., "(Fig. 1)", "[Table 2]", "Figure 3", "Table S1"
    s = re.sub(r'\b(Fig(?:ure)?|Table|Supplementary Figure|Supplementary Table)\s*[S\dIVXivx]+\b', '', s)
    s = re.sub(r'\((Fig(?:ure)?|Table)\s*[S\dIVXivx]+\)', '', s)
    s = re.sub(r'\[Fig(?:ure)?\s*[S\dIVXivx]+\]', '', s)
    s = re.sub(r'\[Table\s*[S\dIVXivx]+\]', '', s)

    # 4) Remove trailing reference markers like [1], [23], (1), {1}
    s = re.sub(r'\s*\[[0-9,\-\s]+\]\s*$', '', s)
    s = re.sub(r'\s*\([0-9,\-\s]+\)\s*$', '', s)
    s = re.sub(r'\s*\{[0-9,\-\s]+\}\s*$', '', s)

    # 5) Replace multiple whitespaces with single and normalize dashes
    s = s.replace("–", "-").replace("—", "-")
    s = re.sub(r'\s+', ' ', s)

    # 6) Normalize common ambiguous wording: remove "the", "a", "an" at ends, generic phrases
    # but keep inner 'the' etc.
    s = s.strip(" -")
    # Lowercase for further linguistic normalization
    lower = s.lower()

    # 7) Remove leading ambiguous phrases like "association of", "effect of", "impact of"
    ambiguous_starts = [
        "association of ", "associations of ", "effect of ", "effects of ",
        "impact of ", "impacts of ", "role of ", "roles of ",
        "relationship between ", "relationships between ",
        "correlation between ", "study of ", "analysis of ",
    ]
    for prefix in ambiguous_starts:
        if lower.startswith(prefix):
            s = s[len(prefix):]
            lower = s.lower()
            break

    # 8) Remove trailing vague qualifiers that don't define the trait itself
    ambiguous_ends = [
        " in humans", " in human", " in patients", " in subjects",
        " in mice", " in mouse", " in rats", " in controls",
        " in cases", " in individuals", " in participants",
        " levels", " level", " measurement", " measurements",
        " value", " values", " rate", " rates",
    ]
    for suffix in ambiguous_ends:
        if lower.endswith(suffix):
            s = s[: -len(suffix)]
            lower = s.lower()
            break

    # 9) Apply language normalization (abbreviations, etc.)
    s = normalize_language(s)

    # 10) Final trim and collapse spaces
    s = s.strip()
    s = re.sub(r'\s+', ' ', s)

    # If it becomes empty, fallback to original trimmed
    if not s:
        s = trait.strip() if isinstance(trait, str) else str(trait)
    return s

def normalize_language(text: str) -> str:
    normalized = text
    for pattern, replacement in NORMALIZATION_MAP.items():
        normalized = re.sub(pattern, replacement, normalized, flags=re.IGNORECASE)
    return normalized

def standardize_trait(trait: str) -> str:
    if trait is None:
        return ""
    s = str(trait)

    # Basic trim
    s = s.strip()

    # Remove leading/trailing punctuation
    s = re.sub(r'^[\s\-\:\;\,\.\*\[\]\(\)]+', '', s)
    s = re.sub(r'[\s\-\:\;\,\.\*\[\]\(\)]+$', '', s)

    # Remove figure/table/supplement refs
    s = re.sub(r'\b(Supplementary Figure|Supplementary Table|Fig(?:ure)?|Table)\s*[S\dIVXivx]+\b', '', s)
    s = re.sub(r'\((Fig(?:ure)?|Table)\s*[S\dIVXivx]+\)', '', s)
    s = re.sub(r'\[(Fig(?:ure)?|Table)\s*[S\dIVXivx]+\]', '', s)

    # Remove trailing reference markers
    s = re.sub(r'\s*\[[0-9,\-\s]+\]\s*$', '', s)
    s = re.sub(r'\s*\([0-9,\-\s]+\)\s*$', '', s)
    s = re.sub(r'\s*\{[0-9,\-\s]+\}\s*$', '', s)

    # Normalize dashes and whitespace
    s = s.replace("–", "-").replace("—", "-")
    s = re.sub(r'\s+', ' ', s)

    s = s.strip(" -")
    lower = s.lower()

    ambiguous_starts = [
        "association of ", "associations of ", "effect of ", "effects of ",
        "impact of ", "impacts of ", "role of ", "roles of ",
        "relationship between ", "relationships between ",
        "correlation between ", "study of ", "analysis of ",
    ]
    for prefix in ambiguous_starts:
        if lower.startswith(prefix):
            s = s[len(prefix):]
            lower = s.lower()
            break

    ambiguous_ends = [
        " in humans", " in human", " in patients", " in subjects",

        " in cases", " in individuals", " in participants",
        " levels", " level", " measurement", " measurements",

            break
