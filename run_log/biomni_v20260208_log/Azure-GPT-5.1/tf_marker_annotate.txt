    """
    Very simple heuristic classifier using description fields.
    The json structure is assumed from context; we will try common keys.
    Categories:
      0 -> TF
      1 -> T Marker
      2 -> I Marker
      3 -> TFMarker
      4 -> TF Pmarker
    (We return the string labels; will map to list order when saving final.)
    """

    label = "TF"

    # Keyword heuristics
    if "potential marker" in text or "putative marker" in text or "candidate marker" in text:

            label = "TF Pmarker"
        else:
            label = "TF Pmarker"

    # TFMarker: explicit TF that is a well-known marker for cell/tissue
    if any(kw in text for kw in ["cell marker", "tissue marker", "definitive marker", "specific marker", "marker of breast cancer", "marker of", "used as a marker"]):
        if "transcription factor" in text or "tf " in text or "tf," in text or "tf." in text:
            label = "TFMarker"

    # T Marker: regulated by TFs
    if any(kw in text for kw in ["regulated by", "downstream of", "target gene of", "transcriptional target", "regulated as a marker"]):
        # Clamp to T Marker if it sounds like a downstream marker
        if "marker gene" in text or "marker genes" in text or "cell surface marker" in text:
            label = "T Marker"

    # I Marker: markers influencing TF activity
    if any(kw in text for kw in ["activate", "inhibit", "modulate", "enhance", "suppress", "repress"]) and "transcription factor" in text:
        if "binds to" in text or "interacts with" in text or "activates" in text or "induces" in text:
            label = "I Marker"

    # Explicit TF hint, if nothing else
    if label == "TF":
        if "transcription factor" in text or "tf " in text or "tf," in text or "tf." in text:
            label = "TF"

    return label
