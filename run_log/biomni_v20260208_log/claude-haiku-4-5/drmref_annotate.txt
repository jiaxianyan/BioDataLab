import os
import pandas as pd
import numpy as np
import scanpy as sc
from scipy import stats
from statsmodels.stats.multitest import multipletests
import warnings
warnings.filterwarnings('ignore')

# Read the dataset
dataset_path = './benchmark/dataset/DRMef/pbmc3k_cd4_fixed.h5ad'
adata = sc.read_h5ad(dataset_path)

print("✓ Step 4: Performing Wilcoxon rank-sum test")

# Get gene expression data
X = adata.X.toarray() if hasattr(adata.X, 'toarray') else np.array(adata.X)
gene_names = adata.var_names.values
groups = adata.obs['group'].values

# Separate data by group
resistant_idx = np.where(groups == 'resistant')[0]
sensitive_idx = np.where(groups == 'sensitive')[0]

resistant_expr = X[resistant_idx, :]
sensitive_expr = X[sensitive_idx, :]

print(f"Resistant cells: {len(resistant_idx)}")
print(f"Sensitive cells: {len(sensitive_idx)}")
print(f"Number of genes to test: {len(gene_names)}")

# Perform Wilcoxon rank-sum test for each gene
p_values = []
log2_fold_changes = []

print("\nCalculating p-values and fold changes...")
for i in range(len(gene_names)):
    if i % 200 == 0:
        print(f"  Processed {i}/{len(gene_names)} genes...")
    
    resistant_vals = resistant_expr[:, i]
    sensitive_vals = sensitive_expr[:, i]
    
    # Wilcoxon rank-sum test
    stat, p_val = stats.ranksums(resistant_vals, sensitive_vals)
    p_values.append(p_val)
    
    # Calculate log2 fold change
    resistant_mean = np.mean(resistant_vals)
    sensitive_mean = np.mean(sensitive_vals)
    
    # Avoid log(0) by adding pseudocount
    log2_fc = np.log2((resistant_mean + 1e-6) / (sensitive_mean + 1e-6))
    log2_fold_changes.append(log2_fc)

p_values = np.array(p_values)
log2_fold_changes = np.array(log2_fold_changes)

print(f"\n✓ Wilcoxon test completed for all {len(gene_names)} genes")
