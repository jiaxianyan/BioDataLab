def calculate_gc_content(sequence):
    """Calculate GC content as a fraction (0-1)"""
    seq = sequence.upper()
    g_count = seq.count('G')
    c_count = seq.count('C')
    total_count = len(seq)
    
    if total_count == 0:
        return 0
    
    return (g_count + c_count) / total_count
