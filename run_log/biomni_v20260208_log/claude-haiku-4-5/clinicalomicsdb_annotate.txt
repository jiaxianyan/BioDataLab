import pandas as pd
import numpy as np
from scipy.stats import ranksums
from sklearn.metrics import roc_auc_score
import os

# Load the expression data
data_path = "benchmark/dataset/ClinicalOmicsDB/expression_data.csv"
df = pd.read_csv(data_path)

# Extract and structure the data
group_row = df.iloc[0, 1:].values
genes = df.iloc[1:, 0].values.tolist()
expression_data = df.iloc[1:, 1:].copy()
expression_data.columns = df.columns[1:]

# Convert all expression values to numeric
for col in expression_data.columns:
    expression_data[col] = pd.to_numeric(expression_data[col], errors='coerce')

expression_data = expression_data.reset_index(drop=True)

# Create binary labels
binary_labels = np.array([1 if g == 'responder' else 0 for g in group_row])

# Separate indices by group
responder_indices = np.where(binary_labels == 1)[0]
non_responder_indices = np.where(binary_labels == 0)[0]

# Step 3: Perform statistical tests for each gene
print("Step 3: Performing statistical tests for each gene...")

results = []

for i, gene in enumerate(genes):
    # Get expression values for responders and non-responders
    responder_expr = expression_data.iloc[i, responder_indices].values
    non_responder_expr = expression_data.iloc[i, non_responder_indices].values
    
    # Remove NaN values
    responder_expr = responder_expr[~np.isnan(responder_expr)]
    non_responder_expr = non_responder_expr[~np.isnan(non_responder_expr)]
    
    # Perform Wilcoxon rank-sum test
    statistic, p_value = ranksums(responder_expr, non_responder_expr)
    
    # Calculate AUROC score
    # Create labels: 1 for responder, 0 for non-responder
    y_true = np.concatenate([np.ones(len(responder_expr)), np.zeros(len(non_responder_expr))])
    y_scores = np.concatenate([responder_expr, non_responder_expr])
    
    auroc = roc_auc_score(y_true, y_scores)
    
    results.append({
        'gene': gene,
        'p_value': p_value,
        'auroc': auroc
    })
    
    if (i + 1) % 10 == 0:
        print(f"Processed {i + 1}/{len(genes)} genes")

print(f"\nCompleted processing all {len(genes)} genes")
