import pandas as pd
import json
import os

# Reload data
file_path = "benchmark/dataset/RAVAR/trails_data.csv"
df = pd.read_csv(file_path, sep='\t')

# Step 2 & 3: Analyze and classify traits
# Based on ontological principles:
# - Material property: Physical/chemical characteristics of materials/substances
# - Information entity: Data, measurements, observations, phenotypes
# - Material entity: Physical objects, biological structures, organisms, tissues

classifications = []
analysis_log = []

print("="*80)
print("TRAIT CLASSIFICATION ANALYSIS")
print("="*80)

for idx, row in df.iterrows():
    trait_label = row['Trait Label']
    trait_id = row['Trait Ontology ID']
    description = row['EFO description'] if pd.notna(row['EFO description']) else "No description"
    
    print(f"\n--- Trait {idx}: {trait_label} ---")
    print(f"ID: {trait_id}")
    print(f"Description: {description}")
    
    # Classification logic
    if idx == 0:  # "Abnormality of connective tissue"
        # This describes a structural/material abnormality of biological tissue
        category = "material entity"
        reasoning = "Refers to connective tissue (tendons, ligaments, fascia) - physical biological structures"
        
    elif idx == 1:  # "Abnormality of the nervous system"
        # This describes a structural/material abnormality of the nervous system
        category = "material entity"
        reasoning = "Refers to the nervous system - a physical biological structure/organ system"
        
    elif idx == 2:  # "Acisoga measurement"
        # This is a measurement/observation (information)
        category = "information entity"
        reasoning = "Measurement is data/information collected from observations"
        
    elif idx == 3:  # "Acute cholecystitis"
        # This is a disease/pathological condition (material entity - diseased tissue/organ)
        category = "material entity"
        reasoning = "Refers to inflammation of gallbladder - a pathological condition of a material structure"
    
    classifications.append(category)
    
    log_entry = {
        "trait_index": idx,
        "trait_label": trait_label,
        "trait_id": trait_id,
        "category": category,
        "reasoning": reasoning
    }
    analysis_log.append(log_entry)
    
    print(f"Category: {category}")
    print(f"Reasoning: {reasoning}")

print("\n" + "="*80)
print("CLASSIFICATION SUMMARY")
print("="*80)
print(f"Classifications: {classifications}")
