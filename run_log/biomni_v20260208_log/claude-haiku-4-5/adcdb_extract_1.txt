import os

pdf_dir = "./benchmark/dataset/ADCdb"

extracted_texts = {}

print("=" * 80)
print("STEP 1: EXTRACTING TEXT FROM PDF FILES")
print("=" * 80)

for pdf_name in pdf_files:
    pdf_path = os.path.join(pdf_dir, pdf_name)
    print(f"\nüìÑ Processing: {pdf_name}")
    
    try:
        with pdfplumber.open(pdf_path) as pdf:
            num_pages = len(pdf.pages)
            print(f"   Pages: {num_pages}")
            
            full_text = ""
            for i, page in enumerate(pdf.pages):
                text = page.extract_text()
                if text:
                    full_text += f"\n--- PAGE {i+1} ---\n{text}"
            
            extracted_texts[pdf_name] = full_text
            
            # Save to temporary file
            temp_file = os.path.join(tmp_dir, f"{pdf_name.replace('.pdf', '')}_extracted.txt")
            with open(temp_file, 'w', encoding='utf-8') as f:
                f.write(full_text)
            
            print(f"   ‚úì Extracted {len(full_text)} characters")
            print(f"   ‚úì Saved to: {temp_file}")
            
    except Exception as e:
        print(f"   ‚úó Error: {str(e)}")

print("\n" + "=" * 80)
print("Extraction Summary:")
print("=" * 80)
for pdf_name, text in extracted_texts.items():
    print(f"{pdf_name}: {len(text)} characters extracted")
