# Step 3: Extract metadata from both datasets

# Function to extract metadata
extract_metadata <- function(pheno_data, dataset_name) {
  metadata_list <- list()
  
  for (i in 1:nrow(pheno_data)) {
    sample_info <- list()
    
    # Get sample accession
    sample_info$sample <- pheno_data$geo_accession[i]
    
    # Extract age
    if (dataset_name == "GSE153428") {
      age_col <- "age:ch1"
      age_val <- pheno_data[[age_col]][i]
      sample_info$age <- ifelse(is.na(age_val) || age_val == "", "N/A", as.character(age_val))
    } else if (dataset_name == "GSE154769") {
      age_col <- "age:ch1"
      age_val <- pheno_data[[age_col]][i]
      if (!is.na(age_val) && age_val != "") {
        sample_info$age <- paste(as.character(age_val), "Years")
      } else {
        sample_info$age <- "N/A"
      }
    }
    
    # Extract gender/sex
    if (dataset_name == "GSE153428") {
      sample_info$gender <- "N/A"
    } else if (dataset_name == "GSE154769") {
      sex_val <- pheno_data$Sex.ch1[i]
      sample_info$gender <- ifelse(is.na(sex_val) || sex_val == "", "N/A", tolower(as.character(sex_val)))
    }
    
    # Extract clinical phenotypes
    if (dataset_name == "GSE153428") {
      # Extract from title and characteristics
      title <- pheno_data$title[i]
      tissue <- pheno_data$"tissue:ch1"[i]
      sample_info$`clinical phenotypes` <- ifelse(is.na(title) || title == "", "N/A", as.character(title))
    } else if (dataset_name == "GSE154769") {
      # Extract from characteristics and title
      days_positive <- pheno_data$"days_since_positive_test:ch1"[i]
      if (!is.na(days_positive) && days_positive != "") {
        sample_info$`clinical phenotypes` <- paste("COVID-19 positive,", as.character(days_positive), "days since positive test")
      } else {
        sample_info$`clinical phenotypes` <- "COVID-19 positive"
      }
    }
    
    # Extract drug information
    if (dataset_name == "GSE153428") {
      # Check treatment protocol for drug information
      treatment <- pheno_data$treatment_protocol_ch1[i]
      if (!is.na(treatment) && grepl("IFN", treatment, ignore.case = TRUE)) {
        sample_info$drug <- "IFN-lambda"
      } else {
        sample_info$drug <- "N/A"
      }
    } else if (dataset_name == "GSE154769") {
      sample_info$drug <- "N/A"
    }
    
    metadata_list[[i]] <- sample_info
  }
  
  return(metadata_list)
}
