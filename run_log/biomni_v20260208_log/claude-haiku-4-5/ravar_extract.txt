# Open the PDF and extract content
with pdfplumber.open(pdf_path) as pdf:
    print(f"Total pages: {len(pdf.pages)}")
    
    # Extract text from all pages
    full_text = ""
    for i, page in enumerate(pdf.pages):
        text = page.extract_text()
        full_text += f"\n--- Page {i+1} ---\n"
        full_text += text
    
    # Also try to extract tables
    tables_data = []
    for i, page in enumerate(pdf.pages):
        tables = page.extract_tables()
        if tables:
            for table in tables:
                tables_data.append({
                    "page": i+1,
                    "table": table
                })

print(f"✓ Extracted text from {len(pdf.pages)} pages")
print(f"✓ Found {len(tables_data)} tables")

# Save extracted text for inspection
text_output_path = os.path.join(tmp_dir, "extracted_text.txt")
with open(text_output_path, "w", encoding="utf-8") as f:

print("     - Gene-level tests: P < 1E-4")

print("   Tables analyzed:")
print("     - Table 1: Aorta and tibial arteries SBP SKAT and MetaXcan results")

with open(output_path, "r") as f:
    data = json.load(f)

total = len(data)
accepted = sum(1 for r in data if r["Accepted"])
rejected = total - accepted

print("   Total associations identified: " + str(total))
print("   Associations accepted: " + str(accepted))
print("   Associations rejected: " + str(rejected))
print("   Acceptance rate: " + str(round(100*accepted/total, 1)) + "%")

print("\n5. BREAKDOWN BY TRAIT:")
traits = {}
for record in data:
    trait = record["EFO Trait Label"]
    if trait not in traits:
        traits[trait] = {"total": 0, "accepted": 0}
    traits[trait]["total"] += 1
    if record["Accepted"]:
        traits[trait]["accepted"] += 1

for trait in sorted(traits.keys()):
    counts = traits[trait]
    rate = round(100*counts["accepted"]/counts["total"], 1)
    print("   - " + trait + ": " + str(counts["accepted"]) + "/" + str(counts["total"]) + " accepted (" + str(rate) + "%)")

print("\n6. EFO TRAIT LABELS IDENTIFIED:")
unique_traits = sorted(set(r["EFO Trait Label"] for r in data))
for i, trait in enumerate(unique_traits, 1):
    print("   " + str(i) + ". " + trait)

print("\n7. OUTPUT FILES:")
print("   Main output: " + output_path)
print("   File size: " + str(os.path.getsize(output_path)) + " bytes")

print("   Records: " + str(len(data)))

    fsize = os.path.getsize(fpath)
