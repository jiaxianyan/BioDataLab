import pdfplumber
import os
import re

dataset_path = "./benchmark/dataset/NPCDR/"
pdf_files = [
    "27980427.pdf",
    "25653233.pdf",
    "23429291.pdf",
    "21088503.pdf",
    "32872376.pdf",
    "29626307.pdf"
]

# Classification keywords
keywords_category1 = ["synergy", "synergistic", "enhance", "enhancing", "sensitize", "sensitivity", "efficacy", "augment", "potentiate"]
keywords_category2 = ["adverse", "toxicity", "side effect", "toxin", "reduce toxicity", "mitigate adverse", "decrease adverse"]
keywords_category3 = ["resistance", "resistant", "overcome resistance", "reverse resistance", "reversal", "overcome"]

analysis_results = {}

print("Analyzing each PDF...\n")

for pdf_file in pdf_files:
    full_path = os.path.join(dataset_path, pdf_file)
    pmid = pdf_file.replace('.pdf', '')
    
    with pdfplumber.open(full_path) as pdf:
        # Extract title and abstract from first page
        first_page_text = pdf.pages[0].extract_text()
        
        # Get more content for analysis
        full_text = ""
        for page in pdf.pages[:3]:  # First 3 pages for context
            full_text += page.extract_text().lower() + "\n"
        
        # Count keyword occurrences
        count1 = sum(full_text.count(kw) for kw in keywords_category1)
        count2 = sum(full_text.count(kw) for kw in keywords_category2)
        count3 = sum(full_text.count(kw) for kw in keywords_category3)
        
        # Extract title from first page
        title_match = re.search(r'(?:article|title|research)\s*([^\n]+)', first_page_text, re.IGNORECASE)
        title = title_match.group(1) if title_match else "N/A"
        
        analysis_results[pmid] = {
            'title': title[:100],
            'count_category1': count1,
            'count_category2': count2,
            'count_category3': count3,
            'first_text': first_page_text[:500]
        }
        
        print(f"PMID: {pmid}")
        print(f"Title snippet: {title[:80]}")
        print(f"Category 1 (Enhance/Synergy): {count1}")
        print(f"Category 2 (Adverse reactions): {count2}")
        print(f"Category 3 (Resistance): {count3}")
        print()

import pdfplumber
import os

dataset_path = "./benchmark/dataset/NPCDR/"
pdf_files = [
    "27980427.pdf",
    "25653233.pdf",
    "23429291.pdf",
    "21088503.pdf",
    "32872376.pdf",
    "29626307.pdf"
]

# Classification keywords
keywords_category1 = ["synergy", "synergistic", "enhance", "enhancing", "sensitize", "sensitivity", "efficacy", "augment", "potentiate"]
keywords_category2 = ["adverse", "toxicity", "side effect", "toxin", "reduce toxicity", "mitigate adverse", "decrease adverse"]
keywords_category3 = ["resistance", "resistant", "overcome resistance", "reverse resistance", "reversal", "overcome"]

analysis_results = {}

print("Analyzing each PDF...\n")

for pdf_file in pdf_files:
    full_path = os.path.join(dataset_path, pdf_file)
    pmid = pdf_file.replace('.pdf', '')
    
    with pdfplumber.open(full_path) as pdf:
        # Extract text from first 3 pages
        text_list = []
        for i in range(min(3, len(pdf.pages))):
            page_text = pdf.pages[i].extract_text()
            if page_text is not None:
                text_list.append(page_text.lower())
        
        full_text = " ".join(text_list)
        
        # Count keyword occurrences
        count1 = sum(full_text.count(kw) for kw in keywords_category1)
        count2 = sum(full_text.count(kw) for kw in keywords_category2)
        count3 = sum(full_text.count(kw) for kw in keywords_category3)
        
        analysis_results[pmid] = {
            'count_category1': count1,
            'count_category2': count2,
            'count_category3': count3
        }
        
        print(f"PMID: {pmid}")
        print(f"Category 1 (Enhance/Synergy): {count1}")
        print(f"Category 2 (Adverse reactions): {count2}")
        print(f"Category 3 (Resistance): {count3}")
        print()

print("Analysis complete!")
