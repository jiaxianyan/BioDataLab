import requests

pmids = ["19033652", "19949098", "10483897"]

def fetch_abstract(pmid):
    """Fetch abstract from PubMed using E-utilities."""
    base_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esummary.fcgi"
    params = {
        "db": "pubmed",
        "id": pmid,
        "retmode": "xml"
    }
    try:
        response = requests.get(base_url, params=params, timeout=30)
        response.raise_for_status()
        root = ET.fromstring(response.content)
        # Extract abstract
        # The structure: DocSum -> Item[@Name='ArticleAbstract']
        for item in root.findall(".//Item[@Name='ArticleAbstract']"):
            abstract = item.text
            if abstract:
                return abstract.strip()
        # If not found, try alternative path
        for item in root.findall(".//Item[@Name='Abstract']"):
            abstract = item.text
            if abstract:
                return abstract.strip()
        return None
    except Exception as e:
        print(f"Error fetching PMID {pmid}: {e}")
        return None

# Fetch abstracts
abstracts = {}

    abstract = fetch_abstract(pmid)

import xml.etree.ElementTree as ET
import os

intermediate_dir = Path("/root/autodl-tmp/biomni_v20260208_tmp/deepseek-v3.2-fast/macc_extract/")

def extract_abstract_from_xml(xml_file):
    """Extract abstract text from PubMed XML."""
    try:
        tree = ET.parse(xml_file)
        root = tree.getroot()
        # PubMed XML namespace
        ns = {'pubmed': 'http://www.ncbi.nlm.nih.gov/pubmed'}
        # Find AbstractText
        abstract_elements = root.findall('.//AbstractText')
        if abstract_elements:
            # Concatenate all abstract text
            abstract_parts = []
            for elem in abstract_elements:
                if elem.text:
                    abstract_parts.append(elem.text.strip())
                # Also collect tail text if any
                if elem.tail and elem.tail.strip():
                    abstract_parts.append(elem.tail.strip())
            abstract = ' '.join(abstract_parts)
            return abstract if abstract else None
        else:
            return None
    except Exception as e:
        print(f"Error parsing {xml_file}: {e}")
        return None

abstracts = {}
for pmid in ["19033652", "19949098", "10483897"]:
    xml_path = intermediate_dir / f"{pmid}.xml"
    if xml_path.exists():
        abstract = extract_abstract_from_xml(xml_path)
        abstracts[pmid] = abstract
        print(f"PMID {pmid}: {'Abstract extracted' if abstract else 'No abstract found'}")
        if abstract:
            # Save abstract as text
            with open(intermediate_dir / f"{pmid}_abstract.txt", "w") as f:
                f.write(abstract)
    else:
        print(f"XML file not found for {pmid}")
