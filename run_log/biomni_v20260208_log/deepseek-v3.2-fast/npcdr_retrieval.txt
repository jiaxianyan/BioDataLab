from Bio import Entrez
import time

# Set email for Entrez (required)
Entrez.email = "example@example.com"

# Define query
query = '("Curcumin"[Mesh] OR Curcumin[Title/Abstract]) AND ("Drug Therapy, Combination"[Mesh] OR "Drug Synergism"[Mesh] OR combin*[Title/Abstract] OR synerg*[Title/Abstract])'

# Add date range: November 1, 2025 to November 5, 2025
# PubMed date format: YYYY/MM/DD
date_range = '2025/11/01:2025/11/05'
query_with_date = f'{query} AND ("{date_range}"[Date - Publication])'

# Also filter for Review articles
query_final = f'{query_with_date} AND "Review"[Publication Type]'

print("Query:", query_final)

# Perform search
try:
    handle = Entrez.esearch(db="pubmed", term=query_final, retmax=1000)
    record = Entrez.read(handle)
    handle.close()
except Exception as e:
    print("Error during search:", e)
    record = None

if record:
    id_list = record["IdList"]
    count = int(record["Count"])
    print(f"Found {count} articles.")
    print("First few PMIDs:", id_list[:10])
else:
    id_list = []
    count = 0
    print("No articles found.")

# Save intermediate: list of PMIDs
import json
pmid_file = os.path.join(intermediate_dir, "pmids.json")
with open(pmid_file, 'w') as f:
    json.dump(id_list, f)
print(f"Saved PMIDs to {pmid_file}")
