import os
import json

benchmark_dir = "benchmark/dataset/PCMDB"
paper_files = ['example_1.md', 'example_2.md', 'example_3.md']

# Simple substring search
def contains_experiment(text):
    keywords = [
        'GFP', 'green fluorescent protein', 'GFP reporter', 'GFP fusion',
        'GUS', 'β-glucuronidase', 'GUS reporter', 'GUS gene',
        'western blot', 'western blotting', 'immunoblot', 'immunoblotting'
    ]
    text_lower = text.lower()
    for kw in keywords:
        if kw.lower() in text_lower:
            return True
    return False

kept_papers = []
for fname in paper_files:
    filepath = os.path.join(benchmark_dir, fname)
    with open(filepath, 'r', encoding='utf-8') as f:
        content = f.read()
    if contains_experiment(content):
        kept_papers.append(fname.replace('.md', ''))

print("Kept papers:", kept_papers)

# Save result
