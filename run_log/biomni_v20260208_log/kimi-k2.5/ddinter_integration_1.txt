# Function to query PubChem for drug information
def query_pubchem(drug_name):
    """
    Query PubChem API to get drug information including generic and brand names.
    """
    base_url = "https://pubchem.ncbi.nlm.nih.gov/rest/pug"
    
    try:
        # First, search for the compound by name
        search_url = f"{base_url}/compound/name/{drug_name}/cids/JSON"
        response = requests.get(search_url, timeout=30)
        
        if response.status_code == 200:
            data = response.json()
            if 'IdentifierList' in data and 'CID' in data['IdentifierList']:
                cid = data['IdentifierList']['CID'][0]
                print(f"  Found CID {cid} for {drug_name}")
                
                # Get synonyms (includes generic and brand names)
                synonyms_url = f"{base_url}/compound/cid/{cid}/synonyms/JSON"
                syn_response = requests.get(synonyms_url, timeout=30)
                
                if syn_response.status_code == 200:
                    syn_data = syn_response.json()
                    synonyms = syn_data.get('InformationList', {}).get('Information', [{}])[0].get('Synonym', [])
                    return {
                        'cid': cid,
                        'synonyms': synonyms,
                        'status': 'success'
                    }
                else:
                    return {
                        'cid': cid,
                        'synonyms': [],
                        'status': f'synonyms query failed: {syn_response.status_code}'
                    }
            else:
                return {
                    'cid': None,
                    'synonyms': [],
                    'status': 'No CID found'
                }
        else:
            return {
                'cid': None,
                'synonyms': [],
                'status': f'search failed: {response.status_code}'
            }
    except Exception as e:
        return {
            'cid': None,
            'synonyms': [],
            'status': f'error: {str(e)}'
        }

# Query each drug
results = {}
for drug in drug_names:
    print(f"\nQuerying: {drug}")
    result = query_pubchem(drug)
    results[drug] = result
    print(f"  Status: {result['status']}")
    if result['synonyms']:
        print(f"  Top synonyms: {result['synonyms'][:5]}")
    time.sleep(0.5)  # Be nice to the API

# Save intermediate results
intermediate_file = os.path.join(intermediate_dir, 'pubchem_raw_results.json')

import os
import json
import pandas as pd

# Load intermediate results
intermediate_dir = '/root/autodl-tmp/biomni_v20260208_tmp/kimi-k2.5/ddinter_integration_1'
intermediate_file = os.path.join(intermediate_dir, 'pubchem_raw_results.json')

with open(intermediate_file, 'r') as f:
    results = json.load(f)

print("Processing results to extract generic and brand names...")

# Common brand name patterns to identify
common_brand_names = {
    'Tucatinib': {'generic': 'Tucatinib', 'brand': 'Tukysa'},
    'Vismodegibum': {'generic': 'Vismodegib', 'brand': 'Erivedge'},
    'Zopiclonum': {'generic': 'zopiclone', 'brand': 'Imovane'},
}

# Process results and extract generic and brand names
processed_results = []

for drug_name, data in results.items():
    synonyms = data.get('synonyms', [])
    cid = data.get('cid', '')
    
    print(f"\nProcessing: {drug_name}")
    print(f"  CID: {cid}")
    print(f"  All synonyms: {synonyms}")
    
    # Determine generic name (usually the INN/USAN name - first synonym is often the standard name)
    if synonyms:
        # The first synonym is typically the standard name
        generic_name = synonyms[0]
        
        # Look for brand names in synonyms
        brand_names = []
        known_brands = {
            'Tucatinib': ['Tukysa', 'ONT-380'],
            'Vismodegibum': ['Erivedge', 'GDC-0449'],
            'Zopiclonum': ['Imovane', 'Amoban', 'Zimovane'],
        }
        
        # Check if any known brand names are in synonyms
        if drug_name in known_brands:
            for brand in known_brands[drug_name]:
                if brand in synonyms:
                    brand_names.append(brand)
        
        # If no known brand names found, look for capitalized names that might be brands
        if not brand_names:
            for syn in synonyms[1:]:  # Skip the first one (usually generic)
                # Brand names are typically capitalized and don't look like chemical formulas
                if syn[0].isupper() and not any(c.isdigit() for c in syn) and len(syn) > 2:
                    if syn not in ['Irbinitinib', 'Tucatinib']:  # Skip other generic names
                        brand_names.append(syn)
                        break  # Take the first one as primary brand
        
        brand_name = '; '.join(brand_names) if brand_names else 'N/A'
    else:
        generic_name = 'N/A'
        brand_name = 'N/A'
    
    processed_results.append({
        'query_name': drug_name,
        'generic_name': generic_name,
        'brand_name': brand_name
    })
    
    print(f"  -> Generic: {generic_name}")
    print(f"  -> Brand: {brand_name}")

# Create final DataFrame
df = pd.DataFrame(processed_results)
print("\nFinal results:")
print(df)
