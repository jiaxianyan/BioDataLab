import requests
import os
import json
import time

tmp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/kimi-k2.5/a3d_modb_retrieval"

# Step 2: Query UniProt API with pagination to get all results
base_url = "https://rest.uniprot.org/uniprotkb/search"
query = "organism_id:83333 AND reviewed:true AND protein_name:transmembrane"
fields = "accession"

all_accessions = []
cursor = None
page = 1

while True:
    params = {
        "query": query,
        "fields": fields,
        "format": "json",
        "size": 500
    }
    
    if cursor:
        params["cursor"] = cursor
    
    print(f"Fetching page {page}...")
    response = requests.get(base_url, params=params)
    
    if response.status_code != 200:
        print(f"Error: {response.status_code}")
        print(f"Response: {response.text}")
        break
    
    # Get total results from header
    total_results = response.headers.get("X-Total-Results")
    if total_results and page == 1:
        print(f"Total results available: {total_results}")
    
    data = response.json()
    results = data.get("results", [])
    
    print(f"  - Results on this page: {len(results)}")
    
    for result in results:
        accession = result.get("primaryAccession")
        if accession:
            all_accessions.append(accession)
    
    # Check for next page
    link_header = response.headers.get("link")
    if link_header:
        # Parse the link header for next cursor
        links = link_header.split(",")
        next_cursor = None
        for link in links:
            if 'rel="next"' in link:
                # Extract cursor from URL
                import re
                match = re.search(r'cursor=([^&>]+)', link)
                if match:
                    next_cursor = match.group(1)
                    break
        
        if next_cursor:
            cursor = next_cursor
            page += 1
            time.sleep(0.5)  # Be polite to the API
        else:
            break
    else:
        break

print(f"\nTotal accessions collected: {len(all_accessions)}")

# Save all accessions to intermediate file
all_accessions_file = os.path.join(tmp_dir, "all_accessions.json")
with open(all_accessions_file, "w") as f:
    json.dump(all_accessions, f, indent=2)
print(f"All accessions saved to: {all_accessions_file}")
