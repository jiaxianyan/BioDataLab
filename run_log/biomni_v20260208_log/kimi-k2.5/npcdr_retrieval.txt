from Bio import Entrez
from Bio import Medline
import json
import time

# Set up email for NCBI API
Entrez.email = "user@example.com"

# Load PMIDs
output_dir = "/root/autodl-tmp/biomni_v20260208_tmp/kimi-k2.5/npcdr_retrieval/"
with open(f"{output_dir}search_pmids.json", "r") as f:

    pmid_list = search_data["pmids"]

print(f"Fetching details for {len(pmid_list)} articles...")
print(f"PMIDs: {pmid_list}")

# Fetch article details
all_articles = []
for pmid in pmid_list:
    try:
        print(f"\nFetching article PMID: {pmid}")
        handle = Entrez.efetch(db="pubmed", id=pmid, rettype="medline", retmode="text")
        records = Medline.parse(handle)
        for record in records:
            article_info = {
                "PMID": record.get("PMID", ""),
                "Title": record.get("TI", ""),
                "Abstract": record.get("AB", ""),
                "Journal": record.get("TA", ""),
                "PublicationDate": record.get("DP", ""),
                "MeshTerms": record.get("MH", [])
            }
            all_articles.append(article_info)
            print(f"  Title: {article_info['Title'][:100]}...")
        handle.close()
        time.sleep(0.5)  # Be polite to NCBI servers
    except Exception as e:
        print(f"  Error fetching PMID {pmid}: {str(e)}")

# Save all articles
with open(f"{output_dir}article_details.json", "w") as f:
    json.dump(all_articles, f, indent=2)

print(f"\n\nTotal articles fetched: {len(all_articles)}")
print(f"Articles saved to {output_dir}article_details.json")
