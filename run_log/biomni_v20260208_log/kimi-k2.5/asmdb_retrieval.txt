# Step 2: Search NCBI GEO for WGBS Series entries from Oct 1-30, 2019

# Define the search parameters
# Search for whole-genome bisulfite sequencing related terms
search_terms = [
    "whole genome bisulfite sequencing[Title/Abstract]",
    "WGBS[Title/Abstract]",
    "bisulfite sequencing[Title/Abstract]",
    "methylome[Title/Abstract]",
    "methylation sequencing[Title/Abstract]",
    "whole-genome bisulfite[Title/Abstract]"
]

# Combine with date filter for October 2019
# Publication date filter: 2019/10/01 to 2019/10/30
date_filter = "2019/10/01:2019/10/30[PDAT]"

# Use esearch to find all Series entries
def esearch_geo(query, db="gds", retmax=10000):
    """Search NCBI GEO using E-utilities esearch"""
    base_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi"
    
    params = {
        "db": db,
        "term": query,
        "retmax": retmax,
        "retmode": "json",
        "sort": "date"
    }
    
    url = f"{base_url}?{urllib.parse.urlencode(params)}"
    
    try:
        with urllib.request.urlopen(url, timeout=60) as response:
            data = json.loads(response.read().decode('utf-8'))
            return data
    except Exception as e:
        print(f"Error searching: {e}")
        return None

# Build comprehensive search query for WGBS
search_query = f"({{' OR '.join(search_terms)}}) AND gse[Filter] AND {date_filter}"

print("=" * 60)
print("Searching NCBI GEO for WGBS Series from October 2019...")
print("=" * 60)
print(f"\nSearch query: {search_query}")

# Perform the search
result = esearch_geo(search_query, db="gds", retmax=10000)

if result and 'esearchresult' in result:
    idlist = result['esearchresult'].get('idlist', [])
    count = result['esearchresult'].get('count', 0)
    print(f"\n✓ Found {count} entries matching the search criteria")
    print(f"  ID list contains {len(idlist)} IDs")
else:
    print("No results found or error occurred")
    idlist = []
