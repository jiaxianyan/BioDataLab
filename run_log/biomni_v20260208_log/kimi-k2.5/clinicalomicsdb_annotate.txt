results = []

# Process each gene
for i, gene in enumerate(expr_data.index):
    # Get expression values for responders and non-responders
    responder_values = expr_data.iloc[i, responder_indices].values
    non_responder_values = expr_data.iloc[i, non_responder_indices].values
    
    # Wilcoxon rank-sum test (Mann-Whitney U test)
    statistic, p_value = stats.mannwhitneyu(responder_values, non_responder_values, alternative='two-sided')
    
    # Calculate AUROC
    # Combine data: 1 for responders, 0 for non-responders
    all_values = np.concatenate([responder_values, non_responder_values])
    labels = np.concatenate([np.ones(len(responder_values)), np.zeros(len(non_responder_values))])
    
    auroc = roc_auc_score(labels, all_values)
    
    results.append({
        'gene': gene,
        'p_value': p_value,
        'auroc': auroc
    })
    
    if (i + 1) % 10 == 0:
