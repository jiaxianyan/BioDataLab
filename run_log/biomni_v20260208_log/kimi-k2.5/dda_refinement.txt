print("=== Step 3 & 4: Parsing FASTQ and filtering for telomeric motifs ===")

# Read the input FASTQ file
with gzip.open(INPUT_FILE, 'rt') as f:
    content = f.read()

lines = content.strip().split('\n')
print(f"Total lines in file: {len(lines)}")

# Parse FASTQ records
reads = []
i = 0
while i < len(lines):
    header = lines[i]
    sequence = lines[i+1]
    plus_line = lines[i+2]
    quality = lines[i+3]
    
    # Fix quality length to match sequence length
    seq_len = len(sequence)
    qual_len = len(quality)
    if qual_len < seq_len:
        # Pad quality with lowest quality score (!)
        quality = quality + '!' * (seq_len - qual_len)
    elif qual_len > seq_len:
        # Truncate quality
        quality = quality[:seq_len]
    
    reads.append({
        'header': header,
        'sequence': sequence,
        'plus': plus_line,
        'quality': quality
    })
    i += 4

print(f"Total reads parsed: {len(reads)}")

# Filter for G-rich motif
g_rich_reads = []
c_rich_reads = []

for read in reads:
