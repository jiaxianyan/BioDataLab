# Step 4: Fetch abstracts for all PMIDs using efetch
base_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils"

print("Fetching abstracts for all PMIDs...")

# Fetch article details in batches (PubMed recommends batches of 100-200)
batch_size = 100
all_articles = []

for i in range(0, len(pmids), batch_size):
    batch_pmids = pmids[i:i+batch_size]
    pmid_str = ','.join(batch_pmids)
    
    efetch_params = {
        'db': 'pubmed',
        'id': pmid_str,
        'retmode': 'xml'
    }
    
    print(f"  Fetching batch {i//batch_size + 1}: PMIDs {batch_pmids[0]} to {batch_pmids[-1]}")
    
    response = requests.get(f"{base_url}/efetch.fcgi", params=efetch_params, timeout=120)
    
    if response.status_code == 200:
        root = ET.fromstring(response.content)
        
        # Parse each article
        for article in root.findall('.//PubmedArticle'):
            article_data = {}
            
            # Get PMID
            pmid_elem = article.find('.//PMID')
            if pmid_elem is not None:
                article_data['PMID'] = pmid_elem.text
            
            # Get Title
            title_elem = article.find('.//ArticleTitle')
            if title_elem is not None:
                article_data['Title'] = ''.join(title_elem.itertext()) if len(list(title_elem)) > 0 else title_elem.text
            
            # Get Abstract
            abstract_elem = article.find('.//Abstract')
            if abstract_elem is not None:
                abstract_texts = []
                for text_elem in abstract_elem.findall('.//AbstractText'):
                    text = ''.join(text_elem.itertext()) if len(list(text_elem)) > 0 else text_elem.text
                    if text:
                        label = text_elem.get('Label', '')
                        if label:
                            abstract_texts.append(f"{label}: {text}")
                        else:
                            abstract_texts.append(text)
                article_data['Abstract'] = ' '.join(abstract_texts)
            else:
                article_data['Abstract'] = ''
            
            # Get Mesh Terms
            mesh_terms = []
            for mesh in article.findall('.//MeshHeading/DescriptorName'):
                mesh_terms.append(mesh.text)
            article_data['MeshTerms'] = mesh_terms
            
            # Get Keywords
            keywords = []
            for keyword in article.findall('.//Keyword'):
                keywords.append(keyword.text)
            article_data['Keywords'] = keywords
            
            all_articles.append(article_data)
        
        print(f"    Fetched {len(batch_pmids)} articles")
    else:
        print(f"    Error: Status code {response.status_code}")
    
    # Be nice to NCBI servers
    time.sleep(0.5)

print(f"\nTotal articles fetched: {len(all_articles)}")

# Save articles to intermediate file
articles_file = os.path.join(intermediate_dir, "articles.json")
with open(articles_file, 'w', encoding='utf-8') as f:
    json.dump(all_articles, f, indent=2, ensure_ascii=False)
print(f"Saved articles to {articles_file}")

# Display summary
for article in all_articles:
    print(f"\nPMID: {article.get('PMID')}")
    print(f"Title: {article.get('Title', 'N/A')[:100]}...")
