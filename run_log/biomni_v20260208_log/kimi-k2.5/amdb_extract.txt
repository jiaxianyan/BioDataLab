from Bio import Entrez

def get_biosample_info(accession):
    """Query NCBI for biosample information"""
    try:
        # Search for the biosample
        handle = Entrez.esearch(db="biosample", term=accession)
        record = Entrez.read(handle)
        handle.close()
        
        if record['IdList']:
            biosample_id = record['IdList'][0]
            
            # Fetch the biosample details
            handle = Entrez.esummary(db="biosample", id=biosample_id)
            summary = Entrez.read(handle)
            handle.close()
            
            # Also fetch full details
            handle = Entrez.efetch(db="biosample", id=biosample_id, retmode="text")
            details = handle.read()
            handle.close()
            
            return {
                'id': biosample_id,
                'summary': str(summary),
                'details': details
            }
        else:
            return None
    except Exception as e:
        print(f"Error: {e}")
        return None
