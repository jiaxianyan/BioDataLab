edqtls = pd.read_csv("benchmark/dataset/GPEdit/edqtls.tsv", sep='\t')

# Step 4: Filter edQTLs that overlap with GWAS SNPs
print("=" * 50)
print("Step 4: Finding overlapping edQTLs with GWAS SNPs")
print("=" * 50)

print("\nGWAS SNPs:")
print(gwas_snps)
print("\nedQTLs:")
print(edqtls)

# For each GWAS SNP, check if it falls within any edQTL range on the same chromosome
overlapping_edqtls = []

for _, gwas_row in gwas_snps.iterrows():
    gwas_chr = gwas_row['chr']
    gwas_pos = gwas_row['pos']
    gwas_snp_id = gwas_row['snp_id']
    
    # Find edQTLs on the same chromosome where start <= pos <= end
    matching_edqtls = edqtls[
        (edqtls['chr'] == gwas_chr) & 
        (edqtls['start'] <= gwas_pos) & 
        (edqtls['end'] >= gwas_pos)
    ]
    
    if len(matching_edqtls) > 0:
        print(f"\nGWAS SNP {gwas_snp_id} (chr={gwas_chr}, pos={gwas_pos}) overlaps with:")
        print(matching_edqtls[['chr', 'start', 'end', 'edqtl_id', 'gene']])
        
        for _, edqtl_row in matching_edqtls.iterrows():
            overlapping_edqtls.append(edqtl_row)

# Remove duplicates and create final dataframe
if overlapping_edqtls:
    overlapping_edqtls_df = pd.DataFrame(overlapping_edqtls).drop_duplicates()
    print(f"\n\nTotal overlapping edQTLs found: {len(overlapping_edqtls_df)}")
    print("\nFinal overlapping edQTLs:")
    print(overlapping_edqtls_df[['chr', 'start', 'end', 'edqtl_id', 'gene']])
else:
    print("\nNo overlapping edQTLs found")
    overlapping_edqtls_df = pd.DataFrame(columns=edqtls.columns)
