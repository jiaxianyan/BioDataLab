# Create directories
tmp_dir = "/root/autodl-tmp/biomni_v20260208_tmp/kimi-k2.5/scapaatlas_annotate"
output_file = "/root/autodl-tmp/biomni_v20260208_pred_results/kimi-k2.5/scapaatlas_annotate.bed"

os.makedirs(tmp_dir, exist_ok=True)

# Read RBP binding sites
rbp_sites = []
with open("benchmark/dataset/scAPAatlas/rbp_binding_sites.bed", "r") as f:
    for line in f:
        parts = line.strip().split("\t")
        if len(parts) >= 10:
            chrom = parts[0]
            start = int(parts[1])
            end = int(parts[2])
            rbp_sites.append({
                "chrom": chrom,
                "start": start,
                "end": end,
                "data": parts  # store all original data
            })

print(f"Loaded {len(rbp_sites)} RBP binding sites")

# Read APA regions
apa_regions = []
with open("benchmark/dataset/scAPAatlas/apa_regions.bed", "r") as f:
    for line in f:
        parts = line.strip().split("\t")
        if len(parts) >= 6:
            chrom = parts[0]
            start = int(parts[1])
            end = int(parts[2])
            apa_regions.append({
                "chrom": chrom,
                "start": start,
                "end": end,
                "data": parts
            })

print(f"Loaded {len(apa_regions)} APA regions")

# Find intersections
intersections = []
for rbp in rbp_sites:
    for apa in apa_regions:
        # Check if same chromosome and overlapping
        if rbp["chrom"] == apa["chrom"]:
            # Check for overlap: max(start) < min(end)
            if max(rbp["start"], apa["start"]) < min(rbp["end"], apa["end"]):
                intersections.append(rbp["data"])
                break  # Avoid duplicate entries if RBP overlaps multiple APA regions

print(f"Found {len(intersections)} intersecting regions")

# Save to output file
with open(output_file, "w") as f:
    for intersect_data in intersections:
        f.write("\t".join(intersect_data) + "\n")

print(f"Saved intersected regions to {output_file}")
print("\nFirst 10 lines of output:")

            break
        print(line.strip())
