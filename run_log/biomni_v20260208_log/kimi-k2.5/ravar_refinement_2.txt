    trait = row['Reported Trait']
    ont_id = row['Trait Ontology ID']
    
    print(f"\nTrait {idx+1}: {trait}")
    print(f"  Ontology ID: {ont_id}")
    
    # Ontological categorization logic:
    # material property: qualities/characteristics that inhere in material entities
    # information entity: informational content/data
    # material entity: physical objects/substances
    
    if "Abnormality of" in trait:
        # HP ontology phenotypes are qualities (material properties) that inhere in organisms
        category = "material property"
        reasoning = "Phenotype in HP ontology describing a quality/characteristic of a biological entity"
    elif "measurement" in trait.lower():
        # Measurements in EFO are information entities
        category = "information entity"
        reasoning = "Measurement is data/information about a quantity"
