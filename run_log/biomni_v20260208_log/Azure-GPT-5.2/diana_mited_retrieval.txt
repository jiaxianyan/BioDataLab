Retrieve the accession numbers and sort them alphabetically. 

Save the accession numbers as a list to a json file named `/root/biodatalab/log/biomni_v20260209_pred_results/OpenAI/Azure-GPT-5.2/diana_mited_retrieval.json`

Output format (for reference):
[
'SRRXXXXXXX',
'SRXXXXXXXX',
...
]

All intermediate files should be saved in the `/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/diana_mited_retrieval` directory.

<execute>#!BASH
set -euo pipefail

echo "[Research log] Step 0: Prepare directories"
TMP_DIR="/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/diana_mited_retrieval"
OUT_JSON="/root/biodatalab/log/biomni_v20260209_pred_results/OpenAI/Azure-GPT-5.2/diana_mited_retrieval.json"
mkdir -p "$TMP_DIR"
mkdir -p "$(dirname "$OUT_JSON")"
echo "TMP_DIR=$TMP_DIR"
echo "OUT_JSON=$OUT_JSON"

echo "[Research log] Step 1: Query SRA via NCBI Entrez E-utilities (esearch + efetch)"
QUERY='("Homo sapiens"[Organism] AND "miRNA-Seq"[Strategy] AND ("2021/10/01"[PDAT] : "2021/10/05"[PDAT]))'
echo "Query: $QUERY"

ESRCH_XML="$TMP_DIR/esearch.xml"
EFETCH_XML="$TMP_DIR/efetch.xml"

# Use retmax large to ensure all results retrieved (expected modest size for 5-day window)
curl -sG "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi" \
  --data-urlencode db=sra \
  --data-urlencode term="$QUERY" \
  --data-urlencode retmax=100000 \
  --data-urlencode retmode=xml > "$ESRCH_XML"

echo "[Research log] Saved esearch XML: $ESRCH_XML"
echo "esearch.xml first lines:"
head -n 20 "$ESRCH_XML"

# Extract IDs
IDLIST="$TMP_DIR/idlist.txt"
python - <<'PY'
import xml.etree.ElementTree as ET, pathlib
esearch = pathlib.Path("/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/diana_mited_retrieval/esearch.xml")
root = ET.parse(esearch).getroot()
ids = [e.text for e in root.findall(".//IdList/Id") if e.text]
out = pathlib.Path("/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/diana_mited_retrieval/idlist.txt")
out.write_text("\n".join(ids) + ("\n" if ids else ""))
print(f"[Research log] Found {len(ids)} SRA records (UIDs). Wrote {out}")
PY

UID_COUNT=$(wc -l < "$IDLIST" || true)
echo "[Research log] UID count: $UID_COUNT"

# If no UIDs, still produce empty output
if [ "$UID_COUNT" -eq 0 ]; then
  echo "[Research log] No records found; writing empty JSON list."
  python - <<'PY'
import json, pathlib
out = pathlib.Path("/root/biodatalab/log/biomni_v20260209_pred_results/OpenAI/Azure-GPT-5.2/diana_mited_retrieval.json")
out.write_text(json.dumps([], indent=2))
print(f"[Research log] Wrote {out}")
PY
  exit 0
fi

# Fetch all records in one go (may be large but for 5 days should be OK). Use POST to avoid URL length limits.
curl -s "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/efetch.fcgi" \
  -d db=sra \
  -d id="$(paste -sd, "$IDLIST")" \
  -d retmode=xml > "$EFETCH_XML"

echo "[Research log] Saved efetch XML: $EFETCH_XML"
echo "efetch.xml first lines:"
head -n 20 "$EFETCH_XML"

echo "[Research log] Step 2: Parse SRR accessions from efetch XML; sort; save JSON"
python - <<'PY'
import xml.etree.ElementTree as ET
import pathlib, json, re

efetch = pathlib.Path("/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/diana_mited_retrieval/efetch.xml")
root = ET.parse(efetch).getroot()

# Collect run accessions (SRR/ERR/DRR). We will filter to SRR only as per requested output example.
runs = []
for run in root.findall(".//RUN"):
    acc = run.attrib.get("accession")
    if acc:
        runs.append(acc)

# Fallback regex search if XML structure differs
if not runs:
    text = efetch.read_text(errors="ignore")
    runs = re.findall(r'\b[SED]RR\d+\b', text)

srrs = sorted({r for r in runs if r.startswith("SRR")})
tmp_dir = pathlib.Path("/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/diana_mited_retrieval")
(tmp_dir / "accessions_unsorted.txt").write_text("\n".join(sorted(set(runs))) + ("\n" if runs else ""))
(tmp_dir / "srr_accessions_sorted.txt").write_text("\n".join(srrs) + ("\n" if srrs else ""))

out_json = pathlib.Path("/root/biodatalab/log/biomni_v20260209_pred_results/OpenAI/Azure-GPT-5.2/diana_mited_retrieval.json")
out_json.write_text(json.dumps(srrs, indent=2))
print(f"[Research log] Parsed {len(runs)} run accessions total; {len(srrs)} SRR accessions.")
print(f"[Research log] Wrote sorted SRR JSON to: {out_json}")
print("[Research log] First 20 SRR accessions:")
print("\n".join(srrs[:20]))
PY

echo "[Research log] Done."
</execute>
