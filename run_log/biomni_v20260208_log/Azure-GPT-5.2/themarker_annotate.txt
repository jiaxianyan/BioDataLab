import csv, json, os, re

tmp_dir = "/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/themarker_annotate"
in_file = "benchmark/dataset/TheMarker/biomarker.txt"

def assign_label(row):
    text = " ".join([
        row.get("Biomarker Name",""),
        row.get("Gene Name",""),
        row.get("Synonyms",""),
        row.get("Biomarker Class",""),
        row.get("Function",""),
    ]).lower()

    # Heuristics (not triggered for these rows, but implemented and logged)
    if any(k in text for k in ["toxicity", "adverse", "hepatotoxic", "nephrotox", "safety", "qt prolong", "teratogen"]):
        return "SAF"
    if any(k in text for k in ["predict", "predictive", "respond", "responder", "sensitive", "resistance", "companion diagnostic"]):
        return "PRD"
    if any(k in text for k in ["surrogate endpoint", "validated surrogate", "endpoint"]):
        return "SUR"
    if any(k in text for k in ["target engagement", "pharmacodynamic", "drug effect", "inhibition", "activation by treatment"]):
        return "PDY"
    # Default: monitoring biomarker
    return "MOI"

rows = []
with open(in_file, "r", encoding="utf-8") as f:
    reader = csv.DictReader(f, delimiter="\t")
    for r in reader:
        r2 = dict(r)
        r2["AssignedLabel"] = assign_label(r)
        rows.append(r2)

lab_tsv = os.path.join(tmp_dir, "labeled_rows.tsv")
with open(lab_tsv, "w", encoding="utf-8", newline="") as w:
    writer = csv.DictWriter(w, fieldnames=list(rows[0].keys()), delimiter="\t")
    writer.writeheader()
    writer.writerows(rows)

lab_jsonl = os.path.join(tmp_dir, "labeled_rows.jsonl")
with open(lab_jsonl, "w", encoding="utf-8") as w:
    for r in rows:
        w.write(json.dumps(r, ensure_ascii=False) + "\n")

labels = [r["AssignedLabel"] for r in rows]
with open(os.path.join(tmp_dir, "labels_list.json"), "w", encoding="utf-8") as w:
    json.dump(labels, w, ensure_ascii=False)

print("Assigned labels in order:", labels)
print("Saved:", lab_tsv)
print("Saved:", lab_jsonl)
print("Saved:", os.path.join(tmp_dir, "labels_list.json"))
