from pathlib import Path
import json, re
import requests
from xml.etree import ElementTree as ET

print("Research log: Step 3-5 Fetch abstracts, classify, and write JSON.")

tmp_dir = Path("/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/adcdb_extract/")
out_path = Path("/root/biodatalab/log/biomni_v20260209_pred_results/OpenAI/Azure-GPT-5.2/adcdb_extract_2.json")
pmids = json.loads((tmp_dir/"pmids.json").read_text())

base = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/"

def efetch_pubmed_xml(id_list):
    params = {"db":"pubmed", "id": ",".join(id_list), "retmode":"xml"}
    r = requests.get(base+"efetch.fcgi", params=params, timeout=60)
    r.raise_for_status()
    return r.text

# Fetch in batches
batch_size = 200
all_records = []
for i in range(0, len(pmids), batch_size):
    batch = pmids[i:i+batch_size]
    xml_text = efetch_pubmed_xml(batch)
    (tmp_dir/f"efetch_{i//batch_size+1}.xml").write_text(xml_text)
    all_records.append(xml_text)

combined_xml = "\n".join(all_records)
(tmp_dir/"efetch_all.xml").write_text(combined_xml)

# Parse articles
root = ET.fromstring(combined_xml)
articles = root.findall(".//PubmedArticle")

def get_text(elem, path):
    x = elem.find(path)
    return "" if x is None or x.text is None else x.text.strip()

def get_abstract_text(article):
    abs_elems = article.findall(".//Abstract/AbstractText")
    if not abs_elems:
        return ""
    parts = []
    for a in abs_elems:
        label = a.attrib.get("Label")
        t = "".join(a.itertext()).strip()
        if label:
            parts.append(f"{label}: {t}")
        else:
            parts.append(t)
    return "\n".join([p for p in parts if p])

# Heuristic acceptance: must be original research and discuss biological activity (MOA/efficacy/pharm/preclinical/ADC activity)
activity_re = re.compile(r"\b(mechanism|mechanism of action|pharmacolog|efficac|anti-?tumou?r|cytotox|payload|adc|antibody[- ]drug conjugate|in vitro|in vivo|xenograft|preclinical|tumor regression|binding|internalization)\b", re.I)
exclude_re = re.compile(r"\b(review|systematic review|meta-analysis|comment|editorial|protocol)\b", re.I)

results = []
screen_log = []
pmid_seen = set()

for art in articles:
    pmid = get_text(art, ".//MedlineCitation/PMID")
    if not pmid:
        continue
    pmid_seen.add(pmid)
    title = get_text(art, ".//Article/ArticleTitle")
    abstract = get_abstract_text(art)
    pub_types = [ "".join(pt.itertext()).strip().lower() for pt in art.findall(".//PublicationType") ]
    pub_types_str = "; ".join(pub_types)

    text = (title + "\n" + abstract).strip()
    has_drug = bool(re.search(r"(zilovertamab vedotin|mk-2140|vls-101|vls 101)", text, re.I))
    has_activity = bool(activity_re.search(text))
    excluded = bool(exclude_re.search(pub_types_str))  # based on publication types

    accepted = bool(has_drug and has_activity and not excluded and len(abstract) > 0)

    results.append({"PMID": str(pmid), "Accepted": bool(accepted)})
    screen_log.append({
        "PMID": pmid, "Accepted": accepted, "HasDrug": has_drug, "HasActivityTerms": has_activity,
        "PubTypes": pub_types_str, "Title": title[:200]
    })

# Ensure we include all PMIDs from initial search, even if efetch missed any
missing = [p for p in pmids if p not in pmid_seen]
if missing:
    print("WARNING: Some PMIDs missing from efetch parse:", missing)
    for p in missing:
        results.append({"PMID": str(p), "Accepted": False})

# Preserve initial search ordering
order = {p:i for i,p in enumerate(pmids)}
results = sorted(results, key=lambda d: order.get(d["PMID"], 10**9))

# Save intermediates
(tmp_dir/"screen_log.json").write_text(json.dumps(screen_log, indent=2))
out_path.write_text(json.dumps(results, indent=2))

print(f"Parsed PubmedArticle records: {len(articles)}")
print(f"Results objects written: {len(results)}")
print("Accepted count:", sum(1 for r in results if r["Accepted"]))
print("Output saved to:", out_path)
print("Intermediate files in:", tmp_dir)

print("\nUpdated plan:")
plan = [
"1. [✓] Create working directories and define PubMed title/abstract query",
