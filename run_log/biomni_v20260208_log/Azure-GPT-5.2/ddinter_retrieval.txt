# Core Task
Systematically search the PubMed database for research published between 2021/01/01 and 2021/03/01 to identify studies reporting drug-drug interactions (DDIs) involving Ciprofloxacin, acting as either the precipitant or the object drug. The search should employ keywords such as "Drug Interactions"[MeSH] or "drug-drug interaction", while strictly excluding drug-food and drug-gene interactions by filtering out terms like "Food-Drug Interactions"[MeSH Terms], "food"[Title], "Pharmacogenetics"[MeSH Terms], or "gene"[Title/Abstract]. Following the initial search, conduct a detailed abstract analysis to screen for and isolate literature that specifically focuses on clinical or pharmacological interactions between Ciprofloxacin and other pharmaceutical agents, ensuring all irrelevant topics are removed from the final collection.

# Output Requirements & File Path
- Output format: JSON list of objects. The list must include all articles obtained from the initial PubMed search.
- Each object must contain exactly two fields:
  - `"PMID"`: PubMed ID as a string
  - `"Accepted"`: Boolean value (`true` if the article meets all inclusion and filtering criteria, `false` otherwise)
- Save Path: Save the complete list to: `/root/biodatalab/log/biomni_v20260209_pred_results/OpenAI/Azure-GPT-5.2/ddinter_retrieval.json`
- Intermediate Storage Path: All intermediate code generated during execution, downloaded files, and temporary results must be stored in `/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/ddinter_retrieval/`

#!/usr/bin/env python3
import os, json, re, textwrap, pathlib
from datetime import datetime
from urllib.parse import quote
import requests
from bs4 import BeautifulSoup

# ========== Plan ==========
plan = [
"Set up required directories and define the PubMed query with inclusion/exclusion filters and date limits.",
"Run the initial PubMed search (ESearch) and collect all PMIDs.",
"Fetch article metadata/abstracts in batches (EFetch) and parse titles/abstracts.",
"Screen abstracts for Ciprofloxacin DDIs (exclude food/gene topics; include clinical/pharmacological DDI with another drug) and assign Accepted flags.",
"Write the full PMID list with Accepted flags to the required JSON path."
]

print("PLAN (checklist):")
for i,s in enumerate(plan,1):
    print(f"{i}. [ ] {s}")

# ========== Step 1: setup ==========
tmp_dir = "/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/ddinter_retrieval/"
out_path = "/root/biodatalab/log/biomni_v20260209_pred_results/OpenAI/Azure-GPT-5.2/ddinter_retrieval.json"
os.makedirs(tmp_dir, exist_ok=True)
os.makedirs(os.path.dirname(out_path), exist_ok=True)

# PubMed query per instructions
# - core terms: ciprofloxacin AND (Drug Interactions[MeSH] OR "drug-drug interaction")
# - exclude: Food-Drug Interactions[MeSH], food[Title], Pharmacogenetics[MeSH], gene[Title/Abstract]
# - date filter: 2021/01/01 to 2021/03/01
query = (
    '(ciprofloxacin[Title/Abstract] OR Ciprofloxacin[MeSH Terms]) AND '
    '("Drug Interactions"[MeSH Terms] OR "drug-drug interaction"[Title/Abstract] OR "drug drug interaction"[Title/Abstract] OR DDI[Title/Abstract]) '
    'NOT ("Food-Drug Interactions"[MeSH Terms] OR food[Title] OR "Pharmacogenetics"[MeSH Terms] OR gene[Title/Abstract])'
)
date_params = {"mindate":"2021/01/01","maxdate":"2021/03/01","datetype":"pdat"}

with open(os.path.join(tmp_dir,"query.txt"),"w") as f:
    f.write(query+"\n")
    f.write(json.dumps(date_params, indent=2))

print("\nSTEP 1 complete: directories created and query saved.")
print("\nUPDATED PLAN:")
for i,s in enumerate(plan,1):
    chk = "✓" if i==1 else " "
    print(f"{i}. [{chk}] {s}")

# ========== Step 2: ESearch ==========
base = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/"
esearch_url = base + "esearch.fcgi"
params = {
    "db":"pubmed",
    "term":query,
    "retmax":100000,
    "retmode":"json",
    **date_params
}
r = requests.get(esearch_url, params=params, timeout=60)
r.raise_for_status()
data = r.json()
pmids = data["esearchresult"].get("idlist", [])
count = int(data["esearchresult"].get("count","0"))
print(f"\nSTEP 2: ESearch returned count={count}, retrieved PMIDs={len(pmids)}")
with open(os.path.join(tmp_dir,"esearch.json"),"w") as f:
    json.dump(data, f, indent=2)

print("\nUPDATED PLAN:")
for i,s in enumerate(plan,1):
    chk = "✓" if i<=2 else " "
    print(f"{i}. [{chk}] {s}")

# ========== Step 3: EFetch batch fetch ==========
efetch_url = base + "efetch.fcgi"

def fetch_batch(id_batch):
    params = {"db":"pubmed","id":",".join(id_batch),"retmode":"xml"}
    rr = requests.get(efetch_url, params=params, timeout=60)
    rr.raise_for_status()
    return rr.text

articles = {}  # PMID -> dict(title, abstract)
batch_size = 200
xml_files = []
for i in range(0, len(pmids), batch_size):
    batch = pmids[i:i+batch_size]
    xml_text = fetch_batch(batch)
    xml_path = os.path.join(tmp_dir, f"efetch_{i//batch_size:04d}.xml")
    with open(xml_path,"w") as f:
        f.write(xml_text)
    xml_files.append(xml_path)

    soup = BeautifulSoup(xml_text, "xml")
    for art in soup.find_all("PubmedArticle"):
        pmid = art.find("PMID").get_text(strip=True) if art.find("PMID") else None
        if not pmid: 
            continue
        title = art.find("ArticleTitle").get_text(" ", strip=True) if art.find("ArticleTitle") else ""
        abst_nodes = art.find_all("AbstractText")
        abstract = " ".join([n.get_text(" ", strip=True) for n in abst_nodes]) if abst_nodes else ""
        articles[pmid] = {"title": title, "abstract": abstract}

print(f"\nSTEP 3: fetched and parsed {len(articles)} articles (xml files: {len(xml_files)})")
with open(os.path.join(tmp_dir,"parsed_articles.json"),"w") as f:
    json.dump(articles, f, indent=2)

print("\nUPDATED PLAN:")
for i,s in enumerate(plan,1):
    chk = "✓" if i<=3 else " "
    print(f"{i}. [{chk}] {s}")

# ========== Step 4: Screening ==========
# Heuristic rules:
# Accept if:
#  - mentions ciprofloxacin (title/abstract)
#  - AND indicates interaction with another drug (look for "interaction", "co-admin", "concomitant", "AUC", "Cmax", "pharmacokinetic", "inhibition", "induction")
#  - AND contains another drug-like entity mention (simple: presence of at least one other known drug keyword pattern: letter+...; 
#    We'll use a small curated list of common interacting drugs plus generic "warfarin", "theophylline", "tizanidine", etc)
# Reject if:
#  - food/nutrition terms prominent or gene/pharmacogenetics terms (already filtered but double-check in abstract)
#  - focuses on bacteria resistance/antimicrobial efficacy without DDI focus
#  - in vitro only with no drug-drug focus (e.g., "synergy with antibiotic") unless specifically PK interaction.

other_drug_markers = [
"warfarin","theophylline","tizanidine","cyclosporine","clozapine","duloxetine","methotrexate",
"metronidazole","rifampin","rifampicin","probenecid","phenytoin","digoxin","insulin",
"caffeine","omeprazole","sucralfate","antacid","iron","zinc","calcium","magnesium",
"clopidogrel","apixaban","rivaroxaban","dabigatran","amiodarone","fluconazole",
"azithromycin","clarithromycin","linezolid","vancomycin","gentamicin"
]
# Note: minerals/antacids are drug products; food excluded separately.

exclude_terms = [
"food-drug","food drug","diet","meal","nutrition","grapefruit",
"pharmacogen","genetic","gene ","genes ","genotype","polymorphism","cyp2","cyp3","cyp1a2"
]
non_ddi_focus_terms = [
"antimicrobial susceptibility","minimum inhibitory concentration","MIC",
"biofilm","resistance","resistant","synergistic activity","checkerboard","time-kill"
]

def accepted(pmid, title, abstract):
    text = (title + " " + abstract).lower()
    if "ciprofloxacin" not in text:
        return False
    if any(t in text for t in exclude_terms):
        return False
    # must contain DDI concepts
    ddi_signal = any(k in text for k in ["drug-drug", "drug drug", "interaction", "concomitant", "coadministr", "co-admin", "pharmacokinetic", "auc", "cmax", "clearance", "inhibit", "induc", "exposure"])
    if not ddi_signal:
        return False
    # avoid purely antimicrobial synergy/resistance unless clear PK/clinical DDI language
    if any(t.lower() in text for t in non_ddi_focus_terms) and not any(k in text for k in ["pharmacokinetic","auc","cmax","clearance","qt","toxicity","bleeding","serum concentration"]):
        return False
    # need evidence of another drug
    other = any(m in text for m in other_drug_markers)
    # alternatively detect pattern "ciprofloxacin and X" where X is capitalized in title (fallback)
    if not other:
        if re.search(r"ciprofloxacin (and|with) [a-z0-9\-]{3,}", text):
            other = True
    return other

results = []
for pmid in pmids:
    meta = articles.get(pmid, {"title":"","abstract":""})
    acc = accepted(pmid, meta["title"], meta["abstract"])
    results.append({"PMID": str(pmid), "Accepted": bool(acc)})

with open(os.path.join(tmp_dir,"screening_results.json"),"w") as f:
    json.dump(results, f, indent=2)

n_acc = sum(1 for x in results if x["Accepted"])
print(f"\nSTEP 4: screening complete. Accepted={n_acc}/{len(results)}")
print("Sample Accepted PMIDs:", [x["PMID"] for x in results if x["Accepted"]][:10])

print("\nUPDATED PLAN:")
for i,s in enumerate(plan,1):
    chk = "✓" if i<=4 else " "
    print(f"{i}. [{chk}] {s}")

# ========== Step 5: Write output ==========
with open(out_path, "w") as f:
    json.dump(results, f, indent=2)

print(f"\nSTEP 5: wrote final JSON to {out_path}")
print("\nUPDATED PLAN:")
for i,s in enumerate(plan,1):
    chk = "✓"
    print(f"{i}. [{chk}] {s}")

# Also write a tiny log summary
summary = {
    "query": query,
    "date_params": date_params,
    "initial_count": count,
    "retrieved_pmids": len(pmids),
    "accepted": n_acc,
    "timestamp": datetime.utcnow().isoformat()+"Z"
}
with open(os.path.join(tmp_dir,"run_summary.json"),"w") as f:
    json.dump(summary, f, indent=2)
print("\nRun summary:", summary)
