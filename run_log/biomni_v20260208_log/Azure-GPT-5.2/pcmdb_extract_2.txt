import os, re, json, textwrap
from pathlib import Path

TMP_DIR=Path("/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2pcmdb_extract_2")
OUT_JSON=Path("/root/biodatalab/log/biomni_v20260209_pred_results/OpenAI/Azure-GPT-5.2/pcmdb_extract_2.json")

files=[TMP_DIR/"example_1.md", TMP_DIR/"example_2.md", TMP_DIR/"example_3.md"]

# Step 2: parse and detect experimental marker inference via GFP/GUS reporter or western blot
patterns = {
    "gfp_reporter": re.compile(r"\bGFP\b|green fluorescent protein|GFP[- ]?reporter|reporter line|fluorescent reporter", re.I),
    "gus_reporter": re.compile(r"\bGUS\b|β-?glucuronidase|GUS[- ]?reporter|uidA", re.I),
    "western_blot": re.compile(r"western blot|immunoblot|SDS-?PAGE.*blot|blotted|anti-?[A-Za-z0-9\-]+", re.I),
}
marker_context = re.compile(r"\bmarker(s)?\b|\bcell type\b|\bcell-specific\b|\bcell lineage\b|\bexpression\b|\bpromoter\b", re.I)

per_paper=[]
snippets_dir = TMP_DIR/"snippets"
snippets_dir.mkdir(exist_ok=True)

for fp in files:
    text=fp.read_text(errors="ignore")
    hits=[]
    for k,pat in patterns.items():
        for m in pat.finditer(text):
            start=max(0, m.start()-250)
            end=min(len(text), m.end()+250)
            window=text[start:end]
            # prioritize when window also hints about marker/cell expression
            score = 2 if marker_context.search(window) else 1
            hits.append((k, score, m.group(0), start, end, window))
    hits.sort(key=lambda x:(-x[1], x[0], x[3]))
    # decide keep if any hits exist (these experimental systems indicate marker inference)
    keep = len(hits)>0
    per_paper.append({
        "paper": fp.name,
        "paper_stem": fp.stem,
        "keep": keep,
        "hit_types": sorted(set(h[0] for h in hits)),
        "n_hits": len(hits),
    })
    # save top snippets for audit
    top=hits[:10]
    out_snip = snippets_dir/(fp.stem+"_hits.txt")
    with out_snip.open("w") as f:
        f.write(f"Paper: {fp.name}\nKeep: {keep}\nTotal hits: {len(hits)}\nHit types: {sorted(set(h[0] for h in hits))}\n\n")
        for i,(k,score,token,s,e,window) in enumerate(top,1):
            f.write(f"== Hit {i} | type={k} | score={score} | token={token} | span=({s},{e}) ==\n")
            f.write(window.replace("\n"," ")+"\n\n")

# Save intermediate summary
summary_path = TMP_DIR/"per_paper_detection_summary.json"
summary_path.write_text(json.dumps(per_paper, indent=2))
print("Step 2 results saved:")
print(" -", summary_path)
print(" - snippets:", snippets_dir)

print("\nPer-paper summary:")
for r in per_paper:
    print(r)
