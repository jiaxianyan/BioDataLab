text = paper

sentences = re.split(r'(?<=[\.\!\?])\s+(?=[A-Z0-9\(\[])', text.replace("\n", " "))
sentences = [s.strip() for s in sentences if len(s.strip())>0]

# collect gene symbols from entries

        if g: gene_symbols.append(g)

        gene_symbols.append(str(e))
gene_symbols = [g.strip() for g in gene_symbols]
gene_set = set(gene_symbols)

# also include some common aliases in paper like HNF3A in parentheses etc? We'll just match word boundaries.
def find_sentences_for_gene(gene):
    patt = re.compile(rf'\b{re.escape(gene)}\b')
    return [s for s in sentences if patt.search(s)]

evidence = {g: find_sentences_for_gene(g) for g in gene_symbols}

# Save evidence snippets
with open(os.path.join(workdir, "gene_sentence_evidence.json"), "w") as f:

# Build rule-based label signals based on keywords around genes
categories = ["TF","T Marker","I Marker","TFMarker","TF Pmarker"]

def label_from_sentences(gene, sents):
    joined = " ".join(sents).lower()
    # priority: TFMarker (definitive marker), TF Pmarker (potential marker), I Marker (marker activates TF),
    # TF (regulates marker), T Marker (regulated by TF)
    if re.search(r'\b(potential marker|candidate marker|may serve as a marker|could be a marker)\b', joined) and re.search(r'\btranscription factor|tf\b', joined):
        return "TF Pmarker"
    if re.search(r'\b(marker of|biomarker|cell marker|definitive marker|specific marker)\b', joined) and re.search(r'\btranscription factor|tf\b', joined):
        return "TFMarker"
    if re.search(r'\b(activate|activates|activated|induces|induced)\b', joined) and re.search(r'\btranscription factor|tf\b', joined):
        # gene is marker influencing TF
        return "I Marker"
    if re.search(r'\b(regulate|regulates|regulated|controls|binds to|binding to)\b', joined) and re.search(r'\b(promoter|enhancer|expression of)\b', joined) and re.search(r'\btranscription factor|tf\b', joined):

            return "TF"
    if re.search(r'\b(controlled by|regulated by|downstream of)\b', joined) and re.search(r'\btranscription factor|tf\b', joined):
        return "T Marker"
    return None

import json, os, re

print("Step 4: Classify each entry")
# We'll attempt to use structured fields if present
def get_gene(entry):
    if isinstance(entry, dict):
        return entry.get("gene") or entry.get("symbol") or entry.get("Gene") or entry.get("name") or ""
    return str(entry)

def get_desc(entry):
    if isinstance(entry, dict):
        for k in ["description","desc","function","functional_description","text","context"]:
            if k in entry and entry[k]:
                return str(entry[k])
    return ""

def classify(entry):
    gene = get_gene(entry).strip()
    desc = get_desc(entry).lower()
    # If paper-derived signal exists, use it
    sig = signals.get(gene)
    if sig:
        return sig
    # Heuristics from entry description
    if re.search(r'\btranscription factor\b|\btf\b', desc):
        if re.search(r'\bpotential marker|candidate marker|may serve as a marker|could be a marker\b', desc):
            return "TF Pmarker"
        if re.search(r'\bmarker\b|\bbiomarker\b|\bcell marker\b', desc):
            return "TFMarker"
        if re.search(r'\bregulat|control|binds to|promoter\b', desc):
            return "TF"
        return "TF"
    # marker regulated by TF
    if re.search(r'\bregulated by\b|\bdownstream of\b|\btarget of\b', desc) and re.search(r'\btranscription factor\b|\btf\b', desc):
        return "T Marker"
    # marker influences TF activity
    if re.search(r'\bactivat|induc|inhibit|modulat\b', desc) and re.search(r'\btranscription factor\b|\btf\b', desc):
        return "I Marker"
    # fallback: if gene appears in paper evidence with TF context but no signal, decide by patterns
    ev = " ".join(evidence.get(gene, [])).lower()
    if ev:
        if re.search(r'\bpotential marker|candidate marker\b', ev) and re.search(r'\btranscription factor|tf\b', ev):
            return "TF Pmarker"
        if re.search(r'\bmarker\b|\bbiomarker\b', ev) and re.search(r'\btranscription factor|tf\b', ev):
            return "TFMarker"
        if re.search(r'\bregulat|control|binds to|promoter\b', ev) and re.search(r'\btranscription factor|tf\b', ev):
            return "TF"
        if re.search(r'\bregulated by|target of\b', ev) and re.search(r'\btranscription factor|tf\b', ev):
            return "T Marker"
        if re.search(r'\bactivat|induc\b', ev) and re.search(r'\btranscription factor|tf\b', ev):
            return "I Marker"
    # ultimate fallback: assume T Marker (a marker gene) if not TF
    return "T Marker"

pred_categories = [classify(e) for e in entries]
