import os
import pandas as pd

print("Research log: Spearman correlation between protein abundance and transcript levels")

# Paths
protein_fp = "benchmark/dataset/CancerProteome/protein_abundance.csv"
transcript_fp = "benchmark/dataset/CancerProteome/transcript_levels.csv"

tmp_dir = "/root/biodatalab/log/biomni_v20260209_tmp/OpenAI/Azure-GPT-5.2/cancerproteome_annotate"
out_fp = "/root/biodatalab/log/biomni_v20260209_pred_results/OpenAI/Azure-GPT-5.2/cancerproteome_annotate.txt"

os.makedirs(tmp_dir, exist_ok=True)
os.makedirs(os.path.dirname(out_fp), exist_ok=True)

# Plan
plan = [
    "[ ] Load the protein and transcript CSV files; validate required columns",
    "[ ] Merge the two tables on GeneID; save merged intermediate file",
    "[ ] Compute Spearman correlation; save scalar output to the required .txt file"
]
print("\nPlan:")
for i, s in enumerate(plan, 1):
    print(f"{i}. {s}")

# Step 1
protein = pd.read_csv(protein_fp)
transcript = pd.read_csv(transcript_fp)

required_protein = {"GeneID", "Protein_Abundance"}
required_transcript = {"GeneID", "Transcript_Level"}

missing_p = required_protein - set(protein.columns)
missing_t = required_transcript - set(transcript.columns)
if missing_p or missing_t:
    raise ValueError(f"Missing columns. protein_missing={missing_p}, transcript_missing={missing_t}")

plan[0] = "[✓] Load the protein and transcript CSV files; validate required columns"
print("\nUpdated plan:")
for i, s in enumerate(plan, 1):
    print(f"{i}. {s}")

print(f"\nLoaded protein: {protein.shape}, transcript: {transcript.shape}")
print("Protein head:\n", protein.head(3))
print("Transcript head:\n", transcript.head(3))

# Step 2
merged = protein.merge(transcript, on="GeneID", how="inner")
merged_fp = os.path.join(tmp_dir, "merged_protein_transcript.csv")
merged.to_csv(merged_fp, index=False)

plan[1] = "[✓] Merge the two tables on GeneID; save merged intermediate file"
print("\nUpdated plan:")
for i, s in enumerate(plan, 1):
    print(f"{i}. {s}")

print(f"\nMerged shape: {merged.shape}")
print(f"Saved merged file: {merged_fp}")

# Step 3
# Ensure numeric
merged["Protein_Abundance"] = pd.to_numeric(merged["Protein_Abundance"], errors="coerce")
merged["Transcript_Level"] = pd.to_numeric(merged["Transcript_Level"], errors="coerce")

clean = merged.dropna(subset=["Protein_Abundance", "Transcript_Level"]).copy()
clean_fp = os.path.join(tmp_dir, "merged_clean_numeric.csv")
clean.to_csv(clean_fp, index=False)

print(f"\nAfter numeric coercion + dropna: {clean.shape}")
print(f"Saved cleaned merged file: {clean_fp}")

spearman_corr = clean["Protein_Abundance"].corr(clean["Transcript_Level"], method="spearman")
print(f"\nSpearman correlation: {spearman_corr}")

with open(out_fp, "w") as f:
    f.write(f"{spearman_corr}\n")

plan[2] = "[✓] Compute Spearman correlation; save scalar output to the required .txt file"
print("\nUpdated plan:")
for i, s in enumerate(plan, 1):
    print(f"{i}. {s}")

print(f"\nSaved final correlation to: {out_fp}")
